 67class Precision(contextlib.AbstractContextManager):
 68    """
 69    Context Manager to locally redefine floating-point precision.
 70
 71    Use this object via a with-block.
 72
 73    Example
 74    -------
 75    .. code-block:: python3
 76
 77       import pyxu.runtime as pxrt
 78
 79       pxrt.getPrecision()                      # Width.DOUBLE
 80       with pxrt.Precision(pxrt.Width.SINGLE):
 81           pxrt.getPrecision()                  # Width.SINGLE
 82       pxrt.getPrecision()                      # Width.DOUBLE
 83    """
 84
 85    def __init__(self, width: Width):
 86        self._width = width
 87        self._width_prev = getPrecision()
 88
 89    def __enter__(self) -> "Precision":
 90        _setPrecision(self._width)
 91        return self
 92
 93    def __exit__(self, exc_type, exc_value, traceback) -> bool:
 94        exc_raised = any(_ is not None for _ in [exc_type, exc_value, traceback])
 95        if exc_raised:
 96            pass
 97
 98        _setPrecision(self._width_prev)
 99        return False if exc_raised else True

139def enforce_precision(
140    i: pxt.VarName = frozenset(),
141    o: bool = True,
142    allow_None: bool = True,
143) -> cabc.Callable:
144    """
145    Decorator to pre/post-process function parameters to enforce runtime FP-precision.
146
147    Parameters
148    ----------
149    i: VarName
150        Function parameters for which precision must be enforced to runtime's FP-precision.
151        Function parameter values must have a NumPy API, or be scalars.
152        None-valued parameters are allowed if `allow_None` is True (default).
153    o: bool
154        If True (default), ensure function's output (if any) has runtime's FP-precision.
155        If function's output does not have a NumPy API or is not scalar-valued, set `o` explicitly
156        to False.
157    allow_None: bool
158
159    Example
160    -------
161    .. code-block:: python3
162
163       import pyxu.runtime as pxrt
164
165       @pxrt.enforce_precision(i='y', o=False)  # `i` can process multiple args: `i=('x','y')`.
166       def f(x, y, z=1):
167           print(x.dtype, y.dtype)
168           return x + y + z
169
170       x, y = np.arange(5), np.r_[0.5]
171       print(x.dtype, y.dtype)  # int64, float64
172
173       with pxrt.Precision(pxrt.Width.SINGLE):
174           out = f(x,y)                         # (int64, float32) [printed inside f-call.]
175       print(out.dtype)                         # float64 [would have been float32 if `o=True`.]
176    """
177
178    def decorator(func: cabc.Callable) -> cabc.Callable:
179        @functools.wraps(func)
180        def wrapper(*ARGS, **KWARGS):
181            func_args = pxi.parse_params(func, *ARGS, **KWARGS)
182
183            for k in [i] if isinstance(i, str) else i:
184                if k not in func_args:
185                    error_msg = f"Parameter[{k}] not part of {func.__qualname__}() parameter list."
186                    raise ValueError(error_msg)
187                elif func_args[k] is None:
188                    if not allow_None:
189                        raise ValueError(f"Parameter[{k}] cannot be None-valued.")
190                else:
191                    func_args[k] = coerce(func_args[k])
192
193            out = func(**func_args)
194            if o and (out is not None):
195                out = coerce(out)
196            return out
197
198        return wrapper
199
200    return decorator

213def coerce(x):
214    """
215    Transform input to match runtime FP-precision.
216
217    Parameters
218    ----------
219    x: Real, NDArray
220
221    Returns
222    -------
223    y: Real, NDArray
224        Input cast to the runtime FP-precision.
225        Fails if operation is impossible or unsafe. (I.e. casting complex-valued data.)
226
227    Note
228    ----
229    This method is a NO-OP if :py:func:`~pyxu.runtime.getCoerceState()` returns ``False``.
230    """
231    if getCoerceState() is False:
232        return x
233    else:
234        dtype = getPrecision().value
235        try:
236            if isinstance(x, pxt.Real):
237                return np.array(x, dtype=dtype)[()]
238            elif isinstance(x, nb.Number):
239                raise  # other number categories cannot be converted.
240            elif np.can_cast(x.dtype, dtype, casting="same_kind"):
241                return x.astype(dtype, copy=False)  # cast warnings impossible
242            else:
243                raise
244        except Exception:
245            raise TypeError(f"Cannot coerce {type(x)} to scalar/array of precision {dtype}.")

248def _setPrecision(width: Width):
249    # For internal use only. It is recommended to modify FP-precision locally using the `Precision`
250    # context manager.
251    state = globals()
252    state["__width"] = width

258    state = globals()
