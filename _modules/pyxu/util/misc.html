 12def as_canonical_shape(x: pxt.NDArrayShape) -> pxt.NDArrayShape:
 13    # Transform a lone integer into a valid tuple-based shape specifier.
 14    if isinstance(x, cabc.Iterable):
 15        x = tuple(x)
 16    else:
 17        x = (x,)
 18    sh = tuple(map(int, x))
 19    return sh

 22def next_fast_len(N: pxt.Integer, even: bool = False) -> pxt.Integer:
 23    # Compute the next 5-smooth number greater-or-equal to `N`.
 24    # If `even=True`, then ensure the next 5-smooth number is even-valued.
 25    #
 26    # ref: https://en.wikipedia.org/wiki/Smooth_number
 27
 28    # warning: scipy.fftpack.next_fast_len() != scipy.fft.next_fast_len()
 29    from scipy.fftpack import next_fast_len
 30
 31    N5 = next_fast_len(int(N))
 32
 33    is_even = lambda n: n % 2 == 0
 34    if (not is_even(N5)) and even:
 35        while not is_even(N5 := next_fast_len(N5)):
 36            N5 += 1
 37
 38    return N5

 41def peaks(x: pxt.NDArray, y: pxt.NDArray) -> pxt.NDArray:
 42    r"""
 43    Matlab 2D peaks function.
 44
 45    Peaks is a function of two variables, obtained by translating and scaling Gaussian
 46    distributions.
 47    (See `Matlab's peaks function <https://www.mathworks.com/help/matlab/ref/peaks.html>`_.)
 48
 49    This function is useful for testing purposes.
 50
 51    Parameters
 52    ----------
 53    x: NDArray
 54        X coordinates.
 55    y: NDArray
 56        Y coordinates.
 57
 58    Returns
 59    -------
 60    z: NDArray
 61        Values of the 2D function ``peaks`` at the points specified by the entries of `x` and `y`.
 62
 63    Examples
 64    --------
 65
 66    .. plot::
 67
 68       import numpy as np
 69       import matplotlib.pyplot as plt
 70       from pyxu.util.misc import peaks
 71
 72       x = np.linspace(-3, 3, 1000)
 73       xx, yy = np.meshgrid(x, x)
 74       z = peaks(xx, yy)
 75       plt.figure()
 76       plt.imshow(z)
 77    """
 78    ndi = pxd.NDArrayInfo.from_obj(x)
 79    xp = ndi.module()
 80
 81    a = 3 * ((1 - x) ** 2) * xp.exp(-(x**2) - (y + 1) ** 2)
 82    b = -10 * ((x / 5) - x**3 - y**5) * xp.exp(-(x**2) - (y**2))
 83    c = -xp.exp(-((x + 1) ** 2) - (y**2)) / 3
 84    z = a + b + c
 85    return z

 88def star_like_sample(
 89    N: pxt.Integer,
 90    w: pxt.Integer,
 91    s: pxt.Real,
 92    po: pxt.Integer,
 93    x0: pxt.Real,
 94    ndi: pxd.NDArrayInfo = pxd.NDArrayInfo.NUMPY,
 95) -> pxt.NDArray:
 96    r"""
 97    Star-like test image.
 98
 99    Generates a (N, N) square image of a star-like object normalized between 0 and 1.
100    Based on `GlobalBioIm's StarLikeSample function
101    <https://github.com/Biomedical-Imaging-Group/GlobalBioIm/blob/master/Util/StarLikeSample.m>`_.
102    This function is useful for testing purposes as it contains high-frequency information.
103
104    Parameters
105    ----------
106    N: Integer
107        Size of the image (must be an even number).
108    w: Integer
109        The number of branches of the sample will be 4*w.
110    s: Real
111        Slope of the sigmoid function :math:`\frac1{1+\exp[s (x-x_{0})]}` attenuating the boundaries.
112    po: Integer
113        Power-raising factor for the final image (to have smoother edges).
114    x0: Real
115        Radial shift of the sigmoid function :math:`\frac1{1+\exp[s (x-x_{0})]}`.
116    ndi: NDArrayInfo
117        Desired array module for the output.
118
119    Returns
120    -------
121    image: NDArray
122        (N, N) image of star-like sample.
123
124    Examples
125    --------
126    .. plot::
127
128       import numpy as np
129       import matplotlib.pyplot as plt
130       from pyxu.util.misc import star_like_sample
131
132       star = star_like_sample(N=256, w=8, s=20, po=3, x0=0.7)
133       plt.figure()
134       plt.imshow(star)
135    """
136    assert N % 2 == 0
137    xp = ndi.module()
138
139    grid = xp.linspace(-1, 1, N)
140    x, y = xp.meshgrid(grid, grid)
141    theta = xp.arctan2(y, x)
142    image = 1 + xp.cos(4 * w * theta)
143    image /= 1 + xp.exp(s * (xp.sqrt(x**2 + y**2) - x0))
144    image = xp.maximum(image, 0) / 2
145    image **= po
146    image /= xp.amax(image)
147    return image
