 20def get_array_module(x, fallback: pxt.ArrayModule = None) -> pxt.ArrayModule:
 21    """
 22    Get the array namespace corresponding to a given object.
 23
 24    Parameters
 25    ----------
 26    x: object
 27        Any object compatible with the interface of NumPy arrays.
 28    fallback: ArrayModule
 29        Fallback module if `x` is not a NumPy-like array.
 30        Default behaviour: raise error if fallback used.
 31
 32    Returns
 33    -------
 34    namespace: ArrayModule
 35        The namespace to use to manipulate `x`, or `fallback` if provided.
 36    """
 37
 38    def infer_api(y):
 39        try:
 40            return pxd.NDArrayInfo.from_obj(y).module()
 41        except ValueError:
 42            return None
 43
 44    if (xp := infer_api(x)) is not None:
 45        return xp
 46    elif fallback is not None:
 47        return fallback
 48    else:
 49        raise ValueError(f"Could not infer array module for {type(x)}.")

 52def compute(*args, mode: str = "compute", **kwargs):
 53    r"""
 54    Force computation of Dask collections.
 55
 56    Parameters
 57    ----------
 58    \*args: object, list
 59        Any number of objects.
 60        If it is a dask object, it is evaluated and the result is returned.
 61        Non-dask arguments are passed through unchanged.
 62        Python collections are traversed to find/evaluate dask objects within.
 63        (Use `traverse` =False to disable this behavior.)
 64    mode: str
 65        Dask evaluation strategy: compute or persist.
 66    \*\*kwargs: dict
 67        Extra keyword parameters forwarded to :py:func:`dask.compute` or :py:func:`dask.persist`.
 68
 69    Returns
 70    -------
 71    \*cargs: object, list
 72        Evaluated objects. Non-dask arguments are passed through unchanged.
 73    """
 74    try:
 75        func = dict(compute=dask.compute, persist=dask.persist)[mode.lower()]
 76    except Exception:
 77        raise ValueError(f"mode: expected compute/persist, got {mode}.")
 78
 79    cargs = func(*args, **kwargs)
 80    if len(args) == 1:
 81        cargs = cargs[0]
 82    return cargs

 85def to_NUMPY(x: pxt.NDArray) -> pxt.NDArray:
 86    """
 87    Convert an array from a specific backend to NUMPY.
 88
 89    Parameters
 90    ----------
 91    x: NDArray
 92        Array to be converted.
 93
 94    Returns
 95    -------
 96    y: NDArray
 97        Array with NumPy backend.
 98
 99    Notes
100    -----
101    This function is a no-op if the array is already a NumPy array.
102    """
103    N = pxd.NDArrayInfo
104    ndi = N.from_obj(x)
105    if ndi == N.NUMPY:
106        y = x
107    elif ndi == N.DASK:
108        y = compute(x)
109    elif ndi == N.CUPY:
110        y = x.get()
111    else:
112        msg = f"Dev-action required: define behaviour for {ndi}."
113        raise ValueError(msg)
114    return y

117def redirect(
118    i: pxt.VarName,
119    **kwargs: cabc.Mapping[str, cabc.Callable],
120) -> cabc.Callable:
121    """
122    Change codepath for supplied array backends.
123
124    Some functions/methods cannot be written in module-agnostic fashion. The action of this
125    decorator is summarized below:
126
127    * Analyze an array-valued parameter (`x`) of the wrapped function/method (`f`).
128    * If `x` lies in one of the supplied array namespaces: re-route execution to the specified
129      function.
130    * If `x` lies in none of the supplied array namespaces: execute `f`.
131
132    Parameters
133    ----------
134    i: VarName
135        name of the array-like variable in `f` to base dispatch on.
136    kwargs: ~collections.abc.Mapping
137
138        * key[:py:class:`str`]: array backend short-name as defined in :py:class:`~pyxu.info.deps.NDArrayInfo`.
139        * value[:py:class:`collections.abc.Callable`]: function/method to dispatch to.
140
141    Notes
142    -----
143    Auto-dispatch via :py:func:`redirect` assumes the dispatcher/dispatchee have the same
144    parameterization, i.e.:
145
146    * if `f` is a function -> dispatch possible to another callable with identical signature (i.e.,
147      function or staticmethod)
148    * if `f` is a staticmethod -> dispatch possible to another callable with identical signature
149      (i.e., function or staticmethod)
150    * if `f` is an instance-method -> dispatch to another instance-method of the class with
151      identical signature.
152
153    Example
154    -------
155    .. code-block:: python3
156
157       def f(x, y): return "f"
158
159       @redirect('x', NUMPY=f)    # if 'x' is of type NDArrayInfo.NUMPY, i.e. has
160       def g(x, y): return "g"    # short-name 'NUMPY' -> reroute execution to `f`
161
162       x1 = np.arange(5)
163       x2 = da.array(x1)
164       y = 1
165       g(x1, y), g(x2, y)  # 'f', 'g'
166    """
167
168    def decorator(func: cabc.Callable) -> cabc.Callable:
169        @functools.wraps(func)
170        def wrapper(*ARGS, **KWARGS):
171            try:
172                func_args = pxi.parse_params(func, *ARGS, **KWARGS)
173            except Exception as e:
174                error_msg = f"Could not parameterize {func}()."
175                raise ValueError(error_msg) from e
176
177            if i not in func_args:
178                error_msg = f"Parameter[{i}] not part of {func.__qualname__}() parameter list."
179                raise ValueError(error_msg)
180
181            ndi = pxd.NDArrayInfo.from_obj(func_args[i])
182            if (alt_func := kwargs.get(ndi.name)) is not None:
183                out = alt_func(**func_args)
184            else:
185                out = func(**func_args)
186
187            return out
188
189        return wrapper
190
191    return decorator

194def copy_if_unsafe(x: pxt.NDArray) -> pxt.NDArray:
195    """
196    Copy array if it is unsafe to do in-place updates on it, i.e.
197
198    * if the array is read-only, OR
199    * if the array is a view onto another array.
200
201    Parameters
202    ----------
203    x: NDArray
204
205    Returns
206    -------
207    y: NDArray
208    """
209    N = pxd.NDArrayInfo
210    ndi = N.from_obj(x)
211    if ndi == N.DASK:
212        # Dask operations span a graph -> always safe.
213        y = x
214    elif ndi == N.NUMPY:
215        read_only = not x.flags.writeable
216        reference = not x.flags.owndata
217        y = x.copy() if (read_only or reference) else x
218    elif ndi == N.CUPY:
219        read_only = False  # https://github.com/cupy/cupy/issues/2616
220        reference = not x.flags.owndata
221        y = x.copy() if (read_only or reference) else x
222    else:
223        msg = f"copy_if_unsafe() not yet defined for {ndi}."
224        raise NotImplementedError(msg)
225    return y

228def read_only(x: pxt.NDArray) -> pxt.NDArray:
229    """
230    Make an array read-only.
231
232    Parameters
233    ----------
234    x: NDArray
235
236    Returns
237    -------
238    y: NDArray
239    """
240    N = pxd.NDArrayInfo
241    ndi = N.from_obj(x)
242    if ndi == N.DASK:
243        # Dask operations are read-only since graph-backed.
244        y = x
245    elif ndi == N.NUMPY:
246        y = x.view()
247        y.flags.writeable = False
248    elif ndi == N.CUPY:
249        y = x.view()
250        # y.flags.writeable = False  # https://github.com/cupy/cupy/issues/2616
251    else:
252        msg = f"read_only() not yet defined for {ndi}."
253        raise NotImplementedError(msg)
254    return y
