 21def infer_sum_shape(sh1: pxt.OpShape, sh2: pxt.OpShape) -> pxt.OpShape:
 22    A, B, C, D = *sh1, *sh2
 23    if B == D:
 24        return np.broadcast_shapes((A,), (C,)) + (B,)
 25    else:
 26        raise ValueError(f"Addition of {sh1} and {sh2} operators forbidden.")

 29def infer_composition_shape(sh1: pxt.OpShape, sh2: pxt.OpShape) -> pxt.OpShape:
 30    A, B, C, D = *sh1, *sh2
 31    if B == C:
 32        return (A, D)
 33    else:
 34        raise ValueError(f"Composition of {sh1} and {sh2} operators forbidden.")

 37def vectorize(
 38    i: pxt.VarName,
 39    method: str = "scan",
 40    codim: pxt.Integer = None,
 41) -> cabc.Callable:
 42    """
 43    Decorator to auto-vectorize an array function to abide by
 44    :py:class:`~pyxu.abc.operator.Property` API rules.
 45
 46    Parameters
 47    ----------
 48    i: VarName
 49        Function parameter to vectorize. This variable must hold an object with a NumPy API.
 50    method: str
 51        Vectorization strategy:
 52
 53        * `scan` computes outputs using a for-loop.
 54        * `parallel` passes inputs to DASK and evaluates them in parallel.
 55        * `scan_dask` passes inputs to DASK but evaluates inputs in sequence.
 56          This is useful if the function being vectorized has a shared resource, i.e. is not
 57          thread-safe.
 58          It effectively gives a DASK-unaware function the ability to work with DASK inputs.
 59    codim: Integer
 60        Size of the function's core dimension output.
 61
 62        This parameter is only required in "parallel" and "scan_dask" modes.
 63
 64    Example
 65    -------
 66    .. code-block:: python3
 67
 68       import pyxu.util as pxu
 69
 70       @pxu.vectorize('x')
 71       def f(x):
 72           return x.sum(keepdims=True)
 73
 74       x = np.arange(10).reshape((2, 5))
 75       f(x[0]), f(x[1])  #  [10], [35]
 76       f(x)              #  [10, 35] -> would have retured [45] if not decorated.
 77    """
 78    method = method.strip().lower()
 79    assert method in ("scan", "scan_dask", "parallel"), f"Unknown vectorization method '{method}'."
 80    if using_dask := (method in ("scan_dask", "parallel")):
 81        assert isinstance(codim, pxt.Integer), f"Parameter[codim] must be specified for DASK-backed '{method}'."
 82
 83    def decorator(func: cabc.Callable) -> cabc.Callable:
 84        sig = inspect.Signature.from_callable(func)
 85        if i not in sig.parameters:
 86            error_msg = f"Parameter[{i}] not part of {func.__qualname__}() parameter list."
 87            raise ValueError(error_msg)
 88
 89        @functools.wraps(func)
 90        def wrapper(*ARGS, **KWARGS):
 91            func_args = pxi.parse_params(func, *ARGS, **KWARGS)
 92
 93            x = func_args.pop(i)
 94            *sh, dim = x.shape
 95            x = x.reshape((-1, dim))
 96            N, xp = len(x), pxa.get_array_module(x)
 97
 98            if using_dask:
 99                f = lambda _: func(**{i: _, **func_args})
100                blks = _dask_zip(
101                    func=(f,) * N,
102                    data=x,
103                    out_shape=[(codim,)] * N,  # can't infer codim -> user-specified
104                    out_dtype=(x.dtype,) * N,
105                    parallel=method == "parallel",
106                )
107                y = xp.stack(blks, axis=0)  # (N, codim)
108            else:  # method = "scan"
109                # infer output dimensions + allocate
110                func_args[i] = x[0]
111                y0 = func(**func_args)
112                y = xp.zeros((N, y0.size), dtype=y0.dtype)  # (N, codim)
113
114                y[0] = y0
115                for k in range(1, N):
116                    func_args[i] = x[k]
117                    y[k] = func(**func_args)
118
119            y = y.reshape(*sh, -1)
120            return y
121
122        return wrapper
123
124    return decorator

127def _dask_zip(
128    func: list[cabc.Callable],
129    data: list[pxt.NDArray],
130    out_shape: list[pxt.NDArrayShape],
131    out_dtype: list[pxt.DType],
132    parallel: bool,
133) -> list[pxt.NDArray]:
134    """
135    (This is a Low-Level function.)
136
137    Computes the equivalent of ``out = [f(x) for (f, x) in zip(func, data)]``, with the following semantics:
138
139    * If `data` contains only NUMPY/CUPY arrays, then ``out`` is computed as above.
140    * If `data` contains only DASK arrays, then entries of ``out`` are computed:
141
142      * if `parallel` enabled  -> dask-delay each `func`, then evaluate in parallel.
143      * if `parallel` disabled -> dask-delay each `func`, then evaluate in sequence.
144        (This is useful if `func` s share a common resource, thus not thread-safe.)
145
146    For Dask-array inputs, this amounts to creating a task graph with virtual dependencies
147    between successive `func` calls. In other words, the task graph looks like::
148
149        _dask_zip(func, data, parallel=True) -> out
150
151                      +----+              +----+
152            data[0]-->|func|-->blk[0]-+-->|list|-->out
153                .     +----+          |   +----+
154                .                     |
155                .     +----+          |
156            data[n]-->|func|-->blk[n]-+
157                      +----+
158
159        ==========================================================================================================
160        _dask_zip(func, data, parallel=False) -> out
161                                                                                               +----+
162                                +-------------------+----------------+--------------------+----+list|-->out
163                                |                   |                |                    |    +----+
164                                |                   |                |                    |
165                      +----+    |        +----+     |      +---+     |         +----+     |
166            data[0]-->|func|-->out[0]-+->|func|-->out[1]-->|...|-->out[n-1]-+->|func|-->out[n]
167                      +----+          |  +----+            +---+            |  +----+
168                                      |                                     |
169            data[1]-------------------+                                     |
170                .                                                           |
171                .                                                           |
172                .                                                           |
173            data[n]---------------------------------------------------------+
174
175    Parameters
176    ----------
177    func: list
178        Functions to apply to each element of `data`.
179
180        Function signatures are ``Callable[[NDArray], NDArray]``.
181    data: list
182        (N_data,) arrays to act on.
183    out_shape: list
184        Shapes of ``func[i](data[i])``.
185
186        This parameter is only used if inputs are DASK arrays.
187        Its goal is to transform :py:class:`~dask.delayed.Delayed` objects back to array form.
188    out_dtype: list
189        Dtypes of ``func[i](data[i])``.
190
191        This parameter is only used if inputs are DASK arrays.
192        Its goal is to transform :py:class:`~dask.delayed.Delayed` objects back to array form.
193
194    Returns
195    -------
196    out: list
197        (N_data,) objects acted upon.
198
199        Outputs have the same backend/dtype as inputs, or as specified by `out_[shape,dtype]`.
200    """
201    assert all(len(_) == len(func) for _ in [data, out_shape, out_dtype])
202
203    NDI = pxd.NDArrayInfo
204    dask_input = lambda obj: NDI.from_obj(obj) == NDI.DASK
205    if all(map(dask_input, data)):
206        out = []
207        for i in range(len(data)):
208            if parallel:
209                # If parallel=True, the user guarantees that functions CAN be executed in parallel,
210                # i.e. no side-effects induced by calling the func[k] if arbitrary order.
211                # The functions are hence PURE.
212                _func = dask.delayed(func[i], pure=True)
213            else:
214                # If parallel=False, side-effects MAY influence func[k] outputs.
215                # The functions are hence IMPURE.
216                _func = dgm.bind(
217                    children=dask.delayed(func[i], pure=False),
218                    parents=out[i - 1] if (i > 0) else [],
219                )
220            blk = _array_ize(_func(data[i]), out_shape[i], out_dtype[i])
221            out.append(blk)
222    else:
223        out = [_func(arr) for (_func, arr) in zip(func, data)]
224    return out

227def _array_ize(
228    data,
229    shape: pxt.NDArrayShape,
230    dtype: pxt.DType,
231) -> pxt.NDArray:
232    """
233    (This is a Low-Level function.)
234
235    Transform a Dask-delayed object into its Dask-array counterpart.
236    This function is a no-op if `data` is not a :py:class:`~dask.delayed.Delayed` object.
237
238    Parameters
239    ----------
240    data: NDArray, :py:class:`~dask.delayed.Delayed`
241        Object to act upon.
242    shape: NDArrayShape
243        Shape of `data`.
244
245        This parameter is only used if `data` is a :py:class:`~dask.delayed.Delayed` object.
246        Its goal is to transform the former back to array form.
247    dtype: DType
248        Dtype of `data`.
249
250        This parameter is only used if `data` is a :py:class:`~dask.delayed.Delayed` object.
251        Its goal is to transform the former back to array form.
252
253    Returns
254    -------
255    arr: NDArray
256        Dask-backed NDArray if `data` was a :py:class:`~dask.delayed.Delayed` object; no-op otherwise.
257    """
258    from dask.delayed import Delayed
259
260    if isinstance(data, Delayed):
261        xp = pxd.NDArrayInfo.DASK.module()
262        arr = xp.from_delayed(
263            data,
264            shape=shape,
265            dtype=dtype,
266        )
267    else:
268        arr = data
269    return arr
