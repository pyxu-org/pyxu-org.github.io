 13def view_as_complex(x: pxt.NDArray) -> pxt.NDArray:
 14    r"""
 15    View real-valued array as its complex-valued bijection.
 16    (Inverse of :py:func:`~pyxu.util.complex.view_as_real`.)
 17
 18    Parameters
 19    ----------
 20    x: NDArray
 21        (..., 2N) real-valued array.
 22
 23    Returns
 24    -------
 25    y: NDArray
 26        (..., N) complex-valued array.
 27
 28    Examples
 29    --------
 30
 31    .. code-block:: python3
 32
 33       from pyxu.util import view_as_real, view_as_complex
 34       x = np.arange(6.0)      # array([0., 1., 2., 3., 4., 5.])
 35       y = view_as_complex(x)  # array([0.+1.j, 2.+3.j, 4.+5.j])
 36       view_as_real(y) == x    # True
 37
 38    Notes
 39    -----
 40    Complex-valued inputs are returned unchanged.
 41    For real-valued inputs, this function acts on the last axis as:
 42
 43    .. math::
 44
 45       y_n = x_{2n-1}+j \, x_{2n}, \qquad 1\leq n\leq N.
 46
 47    See Also
 48    --------
 49    :py:func:`~pyxu.util.complex.view_as_real`,
 50    :py:func:`~pyxu.util.complex.view_as_real_mat`,
 51    :py:func:`~pyxu.util.complex.view_as_complex_mat`
 52    """
 53    if _is_complex(x):
 54        return x
 55
 56    try:
 57        r_dtype = x.dtype
 58        r_width = pxrt.Width(r_dtype)
 59        c_dtype = r_width.complex.value
 60    except Exception:
 61        raise ValueError(f"Unsupported dtype {r_dtype}.")
 62    assert x.shape[-1] % 2 == 0, "Last array dimension should be even-valued."
 63
 64    c_sh = (*x.shape[:-1], x.shape[-1] // 2)
 65    y = x.view(c_dtype).reshape(c_sh)
 66    return y

 69def view_as_real(x: pxt.NDArray) -> pxt.NDArray:
 70    r"""
 71    View complex-valued array as its real-valued bijection.
 72    (Inverse of :py:func:`~pyxu.util.complex.view_as_complex`.)
 73
 74    Parameters
 75    ----------
 76    x: NDArray
 77        (..., N) complex-valued array.
 78
 79    Returns
 80    -------
 81    y: NDArray
 82        (..., 2N) real-valued array.
 83
 84    Examples
 85    --------
 86
 87    .. code-block:: python3
 88
 89       from pyxu.util import view_as_real, view_as_complex
 90       x = np.r_[:3] + 1j * np.r_[2:5]   # array([0.+2.j, 1.+3.j, 2.+4.j])
 91       y = view_as_real(x)               # array([0., 2., 1., 3., 2., 4.])
 92       view_as_complex(y) == x           # True
 93
 94    Notes
 95    -----
 96    Real-valued inputs are returned unchanged.
 97    For complex-valued inputs, this function acts on the last axis as:
 98
 99    .. math::
100
101        y_{2n-1} = \mathcal{R}(x_n),
102        \quad
103        y_{2n} = \mathcal{I}(x_n),
104        \quad 1\leq n\leq N,
105
106    where :math:`\mathcal{R}, \mathcal{I}` denote the real/imaginary parts respectively.
107
108    See Also
109    --------
110    :py:func:`~pyxu.util.complex.view_as_complex`,
111    :py:func:`~pyxu.util.complex.view_as_real_mat`,
112    :py:func:`~pyxu.util.complex.view_as_complex_mat`
113    """
114    if _is_real(x):
115        return x
116
117    try:
118        c_dtype = x.dtype
119        c_width = pxrt.CWidth(c_dtype)
120        r_dtype = c_width.real.value
121    except Exception:
122        raise ValueError(f"Unsupported dtype {c_dtype}.")
123
124    r_sh = (*x.shape[:-1], 2 * x.shape[-1])
125    y = x.view(r_dtype).reshape(r_sh)
126    return y

143def view_as_real_mat(
144    cmat: pxt.NDArray,
145    real_input: bool = False,
146    real_output: bool = False,
147) -> pxt.NDArray:
148    r"""
149    View complex-valued matrix as its real-valued equivalent.
150    (Inverse of :py:func:`~pyxu.util.complex.view_as_complex_mat`.)
151
152    Useful to transform complex-valued matrix/vector products to their real-valued counterparts.
153
154    Parameters
155    ----------
156    cmat: NDArray
157        (M, N) complex-valued matrix.
158
159    Returns
160    -------
161    rmat: NDArray
162        The output shape depends on the values of `real_input` and `real_output`::
163
164           | real_input | real_output | rmat.shape |
165           |------------|-------------|------------|
166           | False      | False       | (2M, 2N)   |
167           | False      | True        | ( M, 2N)   |
168           | True       | False       | (2M,  N)   |
169           | True       | True        | ( M,  N)   |
170
171    Examples
172    --------
173
174    .. code-block:: python3
175
176       from pyxu.util import view_as_real_mat, view_as_complex_mat
177
178       A = np.reshape(
179           np.r_[:6] + 1j * np.r_[2:8], # array([[0.+2.j, 1.+3.j, 2.+4.j],
180           newshape=(2, 3),             #        [3.+5.j, 4.+6.j, 5.+7.j]])
181       )
182       B = view_as_real_mat(A)          # array([[ 0., -2.,  1., -3.,  2., -4.],
183                                        #        [ 2.,  0.,  3.,  1.,  4.,  2.],
184                                        #        [ 3., -5.,  4., -6.,  5., -7.],
185                                        #        [ 5.,  3.,  6.,  4.,  7.,  5.]])
186
187    Notes
188    -----
189    * Real-valued matrices are returned unchanged.
190    * Complex-valued matrices :math:`A\in\mathbb{C}^{M\times N}` are mapped into a real-valued
191      matrix :math:`\hat{A}\in\mathbb{R}^{2M\times 2N}` defined, for :math:`1\leq n \leq N`,
192      :math:`1\leq m\leq M` as
193
194      .. math::
195
196          \hat{A}_{2m-1,2n-1} = \mathcal{R}(A_{m,n}),
197          & \quad
198          \hat{A}_{2m-1,2n} = -\mathcal{I}(A_{m,n}),\\
199          \hat{A}_{2m,2n-1} = \mathcal{I}(A_{m,n}),
200          & \quad
201          \hat{A}_{2m,2n}=\mathcal{R}(A_{m,n}).
202
203      If ``real_[in|out]put=True``, then even columns/rows (or both) are furthermore dropped.
204      We have ``view_as_real(A @ x) = view_as_real_mat(A) @ view_as_real(x)``.
205
206    See Also
207    --------
208    :py:func:`~pyxu.util.complex.view_as_real`,
209    :py:func:`~pyxu.util.complex.view_as_complex`,
210    :py:func:`~pyxu.util.complex.view_as_complex_mat`
211    """
212    assert cmat.ndim == 2, f"cmat: expected a 2D array, got {cmat.ndim}-D."
213    if _is_real(cmat):
214        return cmat
215
216    try:
217        c_dtype = cmat.dtype
218        c_width = pxrt.CWidth(c_dtype)
219        r_dtype = c_width.real.value
220    except Exception:
221        raise ValueError(f"Unsupported dtype {c_dtype}.")
222
223    xp = pxd.NDArrayInfo.from_obj(cmat).module()
224    rmat = xp.zeros((2 * cmat.shape[0], 2 * cmat.shape[1]), dtype=r_dtype)
225    rmat[::2, ::2], rmat[1::2, 1::2] = cmat.real, cmat.real
226    rmat[::2, 1::2], rmat[1::2, ::2] = -cmat.imag, cmat.imag
227    if real_input:
228        rmat = rmat[:, ::2]
229    if real_output:
230        rmat = rmat[::2, :]
231    return rmat

234def view_as_complex_mat(
235    rmat: pxt.NDArray,
236    real_input: bool = False,
237    real_output: bool = False,
238) -> pxt.NDArray:
239    r"""
240    View real-valued matrix as its complex-valued equivalent.
241    (Inverse of :py:func:`~pyxu.util.complex.view_as_real_mat`.)
242
243    Useful to transform real-valued matrix/vector products to their complex-valued counterparts.
244
245    Parameters
246    ----------
247    rmat: NDArray
248        Real-valued matrix.
249        Accepted dimensions depend on the values of `real_input` and `real_output`::
250
251           | real_input | real_output | rmat.shape |
252           |------------|-------------|------------|
253           | False      | False       | (2M, 2N)   |
254           | False      | True        | ( M, 2N)   |
255           | True       | False       | (2M,  N)   |
256           | True       | True        | ( M,  N)   |
257
258    Returns
259    -------
260    cmat: NDArray
261        (M, N) complex-valued matrix.
262
263    Examples
264    --------
265
266    .. code-block:: python3
267
268       from pyxu.util import view_as_real_mat, view_as_complex_mat
269
270       A = np.array([[ 0., -2.,  1., -3.,  2., -4.],
271                     [ 2.,  0.,  3.,  1.,  4.,  2.],
272                     [ 3., -5.,  4., -6.,  5., -7.],
273                     [ 5.,  3.,  6.,  4.,  7.,  5.]])
274
275       B = view_as_complex(A)  # array([[0.+2.j, 1.+3.j, 2.+4.j],
276                               #        [3.+5.j, 4.+6.j, 5.+7.j]])
277
278    Notes
279    -----
280    * Complex-valued matrices are returned unchanged.
281    * Real-valued matrices are mapped into a complex-valued matrix :math:`{A}\in\mathbb{C}^{M\times
282      N}` as follows:
283
284      * :math:`\hat{A}\in\mathbb{R}^{2M\times 2N}`:  :math:`A_{m,n} = \hat{A}_{2m-1,2n-1} + j \hat{A}_{2m,2n-1}`,
285      * :math:`\hat{A}\in\mathbb{R}^{M\times 2N}`:  :math:`A_{m,n} = \hat{A}_{m,2n-1} - j \hat{A}_{m,2n}`,
286      * :math:`\hat{A}\in\mathbb{R}^{2M\times N}`:  :math:`A_{m,n} = \hat{A}_{2m-1,n} + j \hat{A}_{2m,n}`,
287      * :math:`\hat{A}\in\mathbb{R}^{M\times N}`:  :math:`A_{m,n} = \hat{A}_{m,n} + 0j`,
288
289      for :math:`1\leq n \leq N`, :math:`1\leq m\leq M`.
290
291    See Also
292    --------
293    :py:func:`~pyxu.util.complex.view_as_real`,
294    :py:func:`~pyxu.util.complex.view_as_complex`,
295    :py:func:`~pyxu.util.complex.view_as_real_mat`
296    """
297    assert rmat.ndim == 2, f"rmat: expected a 2D array, got {rmat.ndim}-D."
298    if _is_complex(rmat):
299        return rmat
300
301    try:
302        r_dtype = rmat.dtype
303        r_width = pxrt.Width(r_dtype)
304        c_dtype = r_width.complex.value
305    except Exception:
306        raise ValueError(f"Unsupported dtype {r_dtype}.")
307
308    N_row, N_col = rmat.shape
309    error_msg = lambda _: f"{_} array dimension should be even-valued."
310    if real_input and real_output:
311        cmat = rmat.astype(c_dtype)
312    elif real_input and (not real_output):
313        assert N_row % 2 == 0, error_msg("First")
314        cmat = rmat[::2, :] + (1j * rmat[1::2, :])
315    elif (not real_input) and real_output:
316        assert N_col % 2 == 0, error_msg("Last")
317        cmat = rmat[:, ::2] - (1j * rmat[:, 1::2])
318    else:  # (not real_input) and (not real_output)
319        assert N_row % 2 == 0, error_msg("First")
320        assert N_col % 2 == 0, error_msg("Last")
321        cmat = rmat[::2, ::2] + (1j * rmat[1::2, ::2])
322    return cmat
