  1r"""
  2This sampler module implements state-of-the-art algorithms that generate samples from probability distributions.
  3These algorithms are particularly well-suited for high-dimensional distributions such as posterior distributions of inverse
  4problems in imaging, which is a notoriously difficult task.
  5The ability to sample from the posterior distribution is extremely valuable, as it allows to explore the landscape of the
  6objective function instead of having a single point estimate (the maximum a posteriori solution, i.e. the mode of the posterior).
  7This is useful for uncertainty quantification (UQ) purposes [UQ_MCMC]_, and it allows compute Monte Carlo estimates of expected
  8values with respect to the posterior.
  9For example, the mean of samples from the posterior is an approximation of the minimum mean-square error (MMSE) estimator that
 10can be used for image reconstruction.
 11Higher-order pixel-wise statistics (e.g., the variance) can also be computed in an online fashion
 12(see :py:mod:`~pyxu.experimental.sampler.statistics`) and provide useful diagnostic tools for uncertainty quantification.
 13
 14In the following example, we showcase the unajusted Langevin algorithm (:py:class:`~pyxu.experimental.sampler._sampler.ULA`)
 15applied to a total-variation denoising problem.
 16We show the MMSE estimator as well as the pixelwise variance of the samples.
 17As expected, the variance is higher around edges than in the smooth regions, indicating that there is higher uncertainty
 18in these regions.
 19
 20.. code-block:: python3
 21
 22   import matplotlib.pyplot as plt
 23   import numpy as np
 24   import pyxu.experimental.sampler as pxes
 25   import pyxu.operator as pxo
 26   import pyxu.opt.solver as pxs
 27   import skimage as skim
 28
 29   sh_im = (128,) * 2
 30   gt = skim.transform.resize(skim.data.shepp_logan_phantom(), sh_im)  # Ground-truth image
 31   N = np.prod(sh_im)  # Number of pixels
 32
 33   # Noisy data
 34   rng = np.random.default_rng(seed=0)
 35   sigma = 1e-1  # Noise standard deviation
 36   y = gt + sigma * rng.standard_normal(sh_im)  # Noisy image
 37   f = 1 / 2 * pxo.SquaredL2Norm(dim=N).argshift(-y.ravel()) / sigma**2  # Data fidelity loss
 38
 39   # Smoothed TV regularization
 40   g = pxo.L21Norm(arg_shape=(2, *sh_im)).moreau_envelope(1e-2) * pxo.Gradient(arg_shape=sh_im)
 41   theta = 10  # Regularization parameter
 42
 43   # Compute MAP estimator
 44   pgd = pxs.PGD(f=f + theta * g)
 45   pgd.fit(x0=y.ravel())
 46   im_MAP = pgd.solution().reshape(sh_im)
 47
 48   fig, ax = plt.subplots(1, 3)
 49   ax[0].imshow(gt)
 50   ax[0].set_title("Ground truth")
 51   ax[0].axis("off")
 52   ax[1].imshow(y, vmin=0, vmax=1)
 53   ax[1].set_title("Noisy image")
 54   ax[1].axis("off")
 55   ax[2].imshow(im_MAP, vmin=0, vmax=1)
 56   ax[2].set_title("MAP reconstruction")
 57   ax[2].axis("off")
 58
 59   ula = pxes.ULA(f=f + theta * g)  # ULA sampler
 60
 61   n = int(1e4)  # Number of samples
 62   burn_in = int(1e3)  # Number of burn-in iterations
 63   gen = ula.samples(x0=np.zeros(N), rng=rng)  # Generator for ULA samples
 64   # Objects for computing online statistics based on samples
 65   online_mean = pxes.OnlineMoment(order=1)
 66   online_var = pxes.OnlineVariance()
 67
 68   i = 0  # Number of samples
 69   for sample in gen:  # Draw ULA sample
 70       i += 1
 71       if i > burn_in + n:
 72           break
 73       if i > burn_in:
 74           mean = online_mean.update(sample)  # Update online mean
 75           var = online_var.update(sample)  # Update online variance
 76
 77   fig, ax = plt.subplots(1, 2)
 78   mean_im = ax[0].imshow(mean.reshape(sh_im), vmin=0, vmax=1)
 79   fig.colorbar(mean_im, fraction=0.05, ax=ax[0])
 80   ax[0].set_title("Mean (MMSE estimator)")
 81   ax[0].axis("off")
 82   var_im = ax[1].imshow(var.reshape(sh_im))
 83   fig.colorbar(var_im, fraction=0.05, ax=ax[1])
 84   ax[1].set_title("Variance")
 85   ax[1].axis("off")
 86   fig.suptitle("Pixel-wise statistics of ULA samples")

108    def samples(self, rng=None, **kwargs) -> cabc.Generator:
109        """Returns a generator; samples are drawn by calling next(generator)."""
110        self._sample_init(rng, **kwargs)
111
112        def _generator():
113            while True:
114                yield self._sample()

116        return _generator()

127class ULA(_Sampler):
128    r"""
129    Unajusted Langevin algorithm (ULA).
130
131    Generates samples from the distribution
132
133    .. math::
134
135       p(\mathbf{x})
136       =
137       \frac{\exp(-\mathcal{F}(\mathbf{x}))}{\int_{\mathbb{R}^N} \exp(-\mathcal{F}(\tilde{\mathbf{x}}))
138       \mathrm{d} \tilde{\mathbf{x}} },
139
140    where :math:`\mathcal{F}: \mathbb{R}^N \to \mathbb{R}` is *differentiable* with :math:`\beta`-*Lipschitz continuous*
141    gradient.
142
143    Notes
144    -----
145    ULA is a Monte-Carlo Markov chain (MCMC) method that derives from the discretization of overdamped Langevin diffusions.
146    More specifically, it relies on the Langevin stochastic differential equation (SDE):
147
148    .. math::
149
150       \mathrm{d} \mathbf{X}_t
151       =
152       - \nabla \mathcal{F}(\mathbf{X}_t) \mathrm{d}t + \sqrt{2} \mathrm{d} \mathbf{B}_t,
153
154    where :math:`(\mathbf{B}_t)_{t \geq 0}` is a :math:`N`-dimensional Brownian motion.
155    It is well known that under mild technical assumptions, this SDE has a unique strong solution whose invariant distribution
156    is :math:`p(\mathbf{x}) \propto \exp(-\mathcal{F}(\mathbf{x}))`.
157    The discrete-time Euler-Maruyama discretization of this SDE then yields the ULA Markov chain
158
159    .. math::
160
161       \mathbf{X}_{k+1} = \mathbf{X}_{k} - \gamma \nabla \mathcal{F}(\mathbf{X}_k) + \sqrt{2 \gamma} \mathbf{Z}_{k+1}
162
163    for all :math:`k \in \mathbb{Z}`, where :math:`\gamma` is the discretization step size and :math:`(\mathbf{Z}_k)_{k
164    \in \mathbb{Z}}` is a sequence of independant and identically distributed :math:`N`-dimensional standard Gaussian
165    distributions.
166    When :math:`\mathcal{F}` is differentiable with :math:`\beta`-Lipschitz continuous gradient and :math:`\gamma \leq \frac{1}{\beta}`,
167    the ULA Markov chain converges (see [ULA]_) to a unique stationary distribution :math:`p_\gamma` such that
168
169    .. math::
170
171       \lim_{\gamma \to 0} \Vert p_\gamma - p \Vert_{\mathrm{TV}} = 0.
172
173    The discretization step :math:`\gamma` is subject to the bias-variance tradeoff:
174    a larger step will lead to faster convergence of the Markov chain at the expense of a larger bias in the
175    approximation of the distribution :math:`p`.
176    Setting :math:`\gamma` as large as possible (default behavior) is recommended for large-scale problems, since
177    convergence speed (rather than approximation bias) is then typically the main bottelneck.
178    See `Example` section below for a concrete illustration of this tradeoff.
179
180    Remarks
181    -------
182    Like all MCMC sampling methods, ULA comes with the following challenges:
183
184    * The first few samples of the chain may not be adequate for computing statistics, as they might be located in low
185      probability regions.
186      This challenge can either be alleviated by selecting a representative starting point to the chain, or by having a
187      `burn-in` phase where the first few samples are discarded.
188
189    * Consecutive samples are typically correlated, which can deteriorate the Monte-Carlo estimation of quantities of
190      interest.
191      This issue can be alleviated by `thinning` the chain, i.e., selecting only every :math:`k` samples, at the expense
192      of an increased computational cost.
193      Useful diagnostic tools to quantify this correlation between samples include the pixel-wise autocorrelation function
194      and the `effective sample size <https://mc-stan.org/docs/reference-manual/effective-sample-size.html>`_.
195
196    Example
197    -------
198    We illustrate ULA on a 1D example (:math:`N = 1`) where :math:`\mathcal{F}(x) = \frac{x^2}{2}`; the target
199    distribution :math:`p(x)` is thus the 1D standard Gaussian.
200    In this toy example, the biased distribution :math:`p_\gamma(x)` can be computed in closed form.
201    The ULA Markov chain is given by
202
203    .. math::
204
205       \mathbf{X}_{k+1} &= \mathbf{X}_{k} - \gamma \nabla\mathcal{F}(\mathbf{X}_k) + \sqrt{2\gamma}\mathbf{Z}_{k+1} \\
206       &= \mathbf{X}_{k} (1 - \gamma) + \sqrt{2 \gamma} \mathbf{Z}_{k+1}.
207
208    Assuming for simplicity that :math:`\mathbf{X}_0` is Gaussian with mean :math:`\mu_0` and variance
209    :math:`\sigma_0^2`, :math:`\mathbf{X}_k` is Gaussian for any :math:`k \in \mathbb{Z}` as a linear combination of
210    Gaussians.
211    Taking the expected value of the recurrence relation yields
212
213    .. math::
214
215       \mu_k := \mathbb{E}(\mathbf{X}_{k}) = \mathbb{E}(\mathbf{X}_{k-1}) (1 - \gamma) = \mu_0 (1 - \gamma)^k
216
217    (geometric sequence).
218    Taking the expected value of the square of the recurrence relation yields
219
220    .. math::
221
222       \mu^{(2)}_k := \mathbb{E}(\mathbf{X}_{k}^2) = \mathbb{E}(\mathbf{X}_{k-1}^2) (1 - \gamma)^2 + 2 \gamma =
223       (1 - \gamma)^{2k} (\sigma_0^2 - b) + b
224
225    with :math:`b = \frac{2 \gamma}{1 - (1 - \gamma)^{2}} = \frac{1}{1-\frac{\gamma}{2}}` (arithmetico-geometric
226    sequence) due to the independence of :math:`\mathbf{X}_{k-1}` and :math:`\mathbf{Z}_{k}`.
227    Hence, :math:`p_\gamma(x)` is a Gaussian with mean :math:`\mu_\gamma= \lim_{k \to \infty} \mu_k = 0` and
228    variance :math:`\sigma_\gamma^2 = \lim_{k \to \infty} \mu^{(2)}_k - \mu_k^2 = \frac{1}{1-\frac{\gamma}{2}}`.
229    As expected, we have :math:`\lim_{\gamma \to 0} \sigma_\gamma^2 = 1`, which is the variance of the target
230    distribution :math:`p(x)`.
231
232    We plot the distribution of the samples of ULA for one large (:math:`\gamma_1 \approx 1`, i.e.
233    :math:`\sigma_{\gamma_1}^2 \approx 2`) and one small (:math:`\gamma_2 = 0.1`, i.e. :math:`\sigma_{\gamma_2}^2
234    \approx 1.05`) step size.
235    As expected, the larger step size :math:`\gamma_1` leads to a larger bias in the approximation of :math:`p(x)`.
236    To quantify the speed of convergence of the Markov chains, we compute the `Cramér-von Mises
237    <https://en.wikipedia.org/wiki/Cram%C3%A9r%E2%80%93von_Mises_criterion>`_ tests of goodness of fit of the
238    empirical distributions to the stationary distributions of ULA :math:`p_{\gamma_1}(x)` and :math:`p_{\gamma_2}(x)`.
239    We observe that the larger step :math:`\gamma_1` leads to a better fit (lower Cramér-von Mises criterion), which
240    illustrates the aforementioned bias-variance tradeoff for the choice of the step size.
241
242    .. plot::
243
244       import matplotlib.pyplot as plt
245       import numpy as np
246       import pyxu.experimental.sampler as pxes
247       import pyxu.operator as pxo
248       import scipy as sp
249
250       f = pxo.SquaredL2Norm(dim=1) / 2  # To sample 1D normal distribution (mean 0, variance 1)
251       ula = pxes.ULA(f=f)  # Sampler with maximum step size
252       ula_lb = pxes.ULA(f=f, gamma=1e-1)  # Sampler with small step size
253
254       gen_ula = ula.samples(x0=np.zeros(1))
255       gen_ula_lb = ula_lb.samples(x0=np.zeros(1))
256       n_burn_in = int(1e3)  # Number of burn-in iterations
257       for i in range(n_burn_in):
258           next(gen_ula)
259           next(gen_ula_lb)
260
261       # Online statistics objects
262       mean_ula = pxes.OnlineMoment(order=1)
263       mean_ula_lb = pxes.OnlineMoment(order=1)
264       var_ula = pxes.OnlineVariance()
265       var_ula_lb = pxes.OnlineVariance()
266
267       n = int(1e4)  # Number of samples
268       samples_ula = np.zeros(n)
269       samples_ula_lb = np.zeros(n)
270       for i in range(n):
271           sample = next(gen_ula)
272           sample_lb = next(gen_ula_lb)
273           samples_ula[i] = sample
274           samples_ula_lb[i] = sample_lb
275           mean = float(mean_ula.update(sample))
276           var = float(var_ula.update(sample))
277           mean_lb = float(mean_ula_lb.update(sample_lb))
278           var_lb = float(var_ula_lb.update(sample_lb))
279
280       # Theoretical variances of biased stationary distributions of ULA
281       biased_var = 1 / (1 - ula._gamma / 2)
282       biased_var_lb = 1 / (1 - ula_lb._gamma / 2)
283
284       # Quantify goodness of fit of empirical distribution with theoretical distribution (Cramér-von Mises test)
285       cvm = sp.stats.cramervonmises(samples_ula, "norm", args=(0, np.sqrt(biased_var)))
286       cvm_lb = sp.stats.cramervonmises(samples_ula_lb, "norm", args=(0, np.sqrt(biased_var_lb)))
287
288       # Plots
289       grid = np.linspace(-4, 4, 1000)
290
291       plt.figure()
292       plt.title(
293           f"ULA samples (large step size) \n Empirical mean: {mean:.3f} (theoretical: 0) \n "
294           f"Empirical variance: {var:.3f} (theoretical: {biased_var:.3f}) \n"
295           f"Cramér-von Mises goodness of fit: {cvm.statistic:.3f}"
296       )
297       plt.hist(samples_ula, range=(min(grid), max(grid)), bins=100, density=True)
298       plt.plot(grid, sp.stats.norm.pdf(grid), label=r"$p(x)$")
299       plt.plot(grid, sp.stats.norm.pdf(grid, scale=np.sqrt(biased_var)), label=r"$p_{\gamma_1}(x)$")
300       plt.legend()
301       plt.show()
302
303       plt.figure()
304       plt.title(
305           f"ULA samples (small step size) \n Empirical mean: {mean_lb:.3f} (theoretical: 0) \n "
306           f"Empirical variance: {var_lb:.3f} (theoretical: {biased_var_lb:.3f}) \n"
307           f"Cramér-von Mises goodness of fit: {cvm_lb.statistic:.3f}"
308       )
309       plt.hist(samples_ula_lb, range=(min(grid), max(grid)), bins=100, density=True)
310       plt.plot(grid, sp.stats.norm.pdf(grid), label=r"$p(x)$")
311       plt.plot(grid, sp.stats.norm.pdf(grid, scale=np.sqrt(biased_var_lb)), label=r"$p_{\gamma_2}(x)$")
312       plt.legend()
313       plt.show()

316    def __init__(self, f: pxa.DiffFunc, gamma: pxt.Real = None):
317        r"""
318        Parameters
319        ----------
320        f: :py:class:`~pyxu.abc.operator.DiffFunc`
321            Differentiable functional.
322        gamma: Real
323            Euler-Maruyama discretization step of the Langevin equation (see `Notes`).
324        """
325        self._f = f
326        self._beta = f.diff_lipschitz
327        self._gamma = self._set_gamma(gamma)
328        self._rng = None
329        self.x = None


330
331    def _sample_init(self, rng, x0: pxt.NDArray):
332        r"""
333        Parameters
334        ----------
335        rng:
336            Internal random generator.
337        x0: NDArray
338            Starting point of the Markov chain.
339        """
340        self.x = x0.copy()
341        if rng is None:
342            xp = pxu.get_array_module(x0)
343            self._rng = xp.random.default_rng(None)
344        else:
345            self._rng = rng
346
347    @pxrt.enforce_precision()
348    def _sample(self) -> pxt.NDArray:
349        x = self.x.copy()
350        x -= self._gamma * pxu.copy_if_unsafe(self._f.grad(self.x))
351        x += math.sqrt(2 * self._gamma) * self._rng.standard_normal(size=self.x.shape, dtype=self.x.dtype)
352        self.x = x
353        return x

355    def objective_func(self) -> pxt.Real:
356        r"""
357        Negative logarithm of the target ditribution (up to the a constant) evaluated at the current state of the
358        Markov chain.
359
360        Useful for diagnostics purposes to monitor whether the Markov chain is sufficiently warm-started.
361        If so, the samples should accumulate around the modes of the target distribution, i.e., toward the minimum of
362        :math:`\mathcal{F}`.
363        """
364        return pxu.copy_if_unsafe(self._f.apply(self.x))


365
366    def _set_gamma(self, gamma: pxt.Real = None) -> pxt.Real:
367        if gamma is None:
368            if math.isfinite(self._beta):
369                return pxrt.coerce(0.98 / self._beta)
370            else:
371                msg = "If f has unbounded Lipschitz gradient, the gamma parameter must be provided."
372            raise ValueError(msg)
373        else:
374            try:
375                assert gamma > 0
376            except Exception:
377                raise ValueError(f"gamma must be positive, got {gamma}.")
378            return pxrt.coerce(gamma)

381class MYULA(ULA):
382    r"""
383    Moreau-Yosida unajusted Langevin algorithm (MYULA).
384
385    Generates samples from the distribution
386
387    .. math::
388
389       p(\mathbf{x}) = \frac{\exp(-\mathcal{F}(\mathbf{x}) - \mathcal{G}(\mathbf{x}))}{\int_{\mathbb{R}^N}
390       \exp(-\mathcal{F}(\tilde{\mathbf{x}}) - \mathcal{G}(\tilde{\mathbf{x}})) \mathrm{d} \tilde{\mathbf{x}} },
391
392    where :math:`\mathcal{F}: \mathbb{R}^N \to \mathbb{R}` is *convex* and *differentiable* with :math:`\beta`-
393    *Lipschitz continuous* gradient, and :math:`\mathcal{G}: \mathbb{R}^N \to \mathbb{R}` is *proper*, *lower semi-
394    continuous* and *convex* with *simple proximal operator*.
395
396    Notes
397    -----
398    MYULA is an extension of :py:class:`~pyxu.experimental.sampler._sampler.ULA` to sample from distributions whose
399    logarithm is nonsmooth.
400    It consists in applying ULA to the differentiable functional :math:`\mathcal{U}^\lambda = \mathcal{F} + \mathcal{G}^\lambda`
401    for some :math:`\lambda > 0`, where
402
403     .. math::
404
405        \mathcal{G}^\lambda (\mathbf{x}) = \inf_{\tilde{\mathbf{x}} \in \mathbb{R}^N} \frac{1}{2 \lambda} \Vert
406        \tilde{\mathbf{x}} - \mathbf{x} \Vert_2^2 + \mathcal{G}(\tilde{\mathbf{x}})
407
408    is the Moreau-Yosida envelope of :math:`\mathcal{G}` with parameter :math:`\lambda`.
409    We then have
410
411    .. math::
412
413       \nabla \mathcal{U}^\lambda (\mathbf{x}) = \nabla \mathcal{F}(\mathbf{x}) + \frac{1}{\lambda} (\mathbf{x} -
414       \mathrm{prox}_{\lambda \mathcal{G}}(\mathbf{x})),
415
416    hence :math:`\nabla \mathcal{U}^\lambda` is :math:`(\beta + \frac{1}{\lambda})`-Lipschitz continuous, where
417    :math:`\beta` is the Lipschitz constant of :math:`\nabla \mathcal{F}`.
418    Note that the target distribution of the underlying ULA Markov chain is not exactly :math:`p(\mathbf{x})`, but the distribution
419
420    .. math::
421
422       p^\lambda(\mathbf{x}) \propto \exp(-\mathcal{F}(\mathbf{x})-\mathcal{G}^\lambda(\mathbf{x})),
423
424    which introduces some additional bias on top of the bias of ULA related to the step size :math:`\gamma` (see `Notes`
425    of :py:class:`~pyxu.experimental.sampler._sampler.ULA` documentation).
426    MYULA is guaranteed to converges when :math:`\gamma \leq \frac{1}{\beta + \frac{1}{\lambda}}`, in which case it converges
427    toward the stationary distribution :math:`p^\lambda_\gamma(\mathbf{x})` that satisfies
428
429    .. math::
430
431       \lim_{\gamma, \lambda \to 0} \Vert p^\lambda_\gamma - p \Vert_{\mathrm{TV}} = 0
432
433    (see [MYULA]_).
434    The parameter :math:`\lambda` parameter is subject to a similar bias-variance tradeoff as :math:`\gamma`.
435    It is recommended to set it in the order of :math:`\frac{1}{\beta}`, so that the contributions of :math:`\mathcal{F}`
436    and :math:`\mathcal{G}^\lambda` to the Lipschitz constant of :math:`\nabla \mathcal{U}^\lambda` is well balanced.
437    """

439    def __init__(
440        self,
441        f: pxa.DiffFunc = None,
442        g: pxa.ProxFunc = None,
443        gamma: pxt.Real = None,
444        lamb: pxt.Real = None,
445    ):
446        r"""
447        Parameters
448        ----------
449        f: :py:class:`~pyxu.abc.operator.DiffFunc` | None
450            Differentiable functional.
451        g: :py:class:`~pyxu.abc.operator.ProxFunc` | None
452            Proximable functional.
453        gamma: Real
454            Euler-Maruyama discretization step of the Langevin equation (see `Notes` of
455            :py:class:`~pyxu.experimental.sampler._sampler.ULA` documentation).
456        lamb: Real
457            Moreau-Yosida envelope parameter for `g`.
458        """
459        dim = None
460        if f is not None:
461            dim = f.dim
462        if g is not None:
463            if dim is None:
464                dim = g.dim
465            else:
466                assert g.dim == dim
467        if dim is None:
468            raise ValueError("One of f or g must be nonzero.")
469
470        self._f_diff = pxo.NullFunc(dim=dim) if (f is None) else f
471        self._g = pxo.NullFunc(dim=dim) if (g is None) else g
472
473        self._lambda = self._set_lambda(lamb)
474        f = self._f_diff + self._g.moreau_envelope(self._lambda)
475        f.diff_lipschitz = f.estimate_diff_lipschitz()
476        super().__init__(f, gamma)


477
478    def _set_lambda(self, lamb: pxt.Real = None) -> pxt.Real:
479        if lamb is None:
480            if self._g._name == "NullFunc":
481                return pxrt.coerce(1)  # Lambda is irrelevant if g is a NullFunc, but it must be positive
482            elif math.isfinite(dl := self._f_diff.diff_lipschitz):
483                return pxrt.coerce(2) if dl == 0 else pxrt.coerce(min(2, 1 / dl))
484            else:
485                msg = "If f has unbounded Lipschitz gradient, the lambda parameter must be provided."
486            raise ValueError(msg)
487        else:
488            return pxrt.coerce(lamb)
