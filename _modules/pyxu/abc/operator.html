 142    def asop(self, cast_to: pxt.OpC) -> pxt.OpT:
 143        r"""
 144        Recast an :py:class:`~pyxu.abc.operator.Operator` (or subclass thereof) to another
 145        :py:class:`~pyxu.abc.operator.Operator`.
 146
 147        Users may call this method if the arithmetic API yields sub-optimal return types.
 148
 149        This method is a no-op if `cast_to` is a parent class of ``self``.
 150
 151        Parameters
 152        ----------
 153        cast_to: OpC
 154            Target type for the recast.
 155
 156        Returns
 157        -------
 158        op: OpT
 159            Operator with the new interface.
 160
 161            Fails when cast is forbidden.
 162            (Ex: :py:class:`~pyxu.abc.operator.Map` -> :py:class:`~pyxu.abc.operator.Func` if codim > 1)
 163
 164        Notes
 165        -----
 166        * The interface of `cast_to` is provided via encapsulation + forwarding.
 167        * If ``self`` does not implement all methods from `cast_to`, then unimplemented methods
 168          will raise :py:class:`NotImplementedError` when called.
 169        """
 170        if cast_to not in _core_operators():
 171            raise ValueError(f"cast_to: expected a core base-class, got {cast_to}.")
 172
 173        p_core = frozenset(self.properties())
 174        p_shell = frozenset(cast_to.properties())
 175        if p_shell <= p_core:
 176            # Trying to cast `self` to it's own class or a parent class.
 177            # Inheritance rules mean the target object already satisfies the intended interface.
 178            return self
 179        else:
 180            # (p_shell > p_core) -> specializing to a sub-class of ``self``
 181            # OR
 182            # len(p_shell ^ p_core) > 0 -> specializing to another branch of the class hierarchy.
 183            op = cast_to(shape=self.shape)
 184            op._core = self  # for debugging
 185
 186            # Forward shared arithmetic fields from core to shell.
 187            for p in p_shell & p_core:
 188                for m in p.arithmetic_methods():
 189                    m_core = getattr(self, m)
 190                    setattr(op, m, m_core)
 191            return op

 409    def squeeze(self) -> pxt.OpT:
 410        r"""
 411        Cast an :py:class:`~pyxu.abc.operator.Operator` to the right core operator sub-type given
 412        codomain dimension.
 413        """
 414        p = set(self.properties())
 415        if self.codim == 1:
 416            p.add(Property.FUNCTIONAL)
 417            if Property.DIFFERENTIABLE in self.properties():
 418                p.add(Property.DIFFERENTIABLE_FUNCTION)
 419            if Property.LINEAR in self.properties():
 420                for p_ in Property:
 421                    if p_.name.startswith("LINEAR_"):
 422                        p.discard(p_)
 423                p.add(Property.PROXIMABLE)
 424        elif self.codim == self.dim:
 425            if Property.LINEAR in self.properties():
 426                p.add(Property.LINEAR_SQUARE)
 427        klass = self._infer_operator_type(p)
 428        return self.asop(klass)

 447    def expr(self, level: int = 0, strip: bool = True) -> str:
 448        """
 449        Pretty-Print the expression representation of the operator.
 450
 451        Useful for debugging arithmetic-induced expressions.
 452
 453        Example
 454        -------
 455
 456        .. code-block:: python3
 457
 458           import numpy as np
 459           import pyxu.abc as pxa
 460
 461           N = 5
 462           op1 = pxa.LinFunc.from_array(np.arange(N))
 463           op2 = pxa.LinOp.from_array(np.ones((N, N)))
 464           op = ((2 * op1) + (op2 ** 3)).argshift(np.full(N, 4))
 465
 466           print(op.expr())
 467           # [argshift, ==> DiffMap(5, 5)
 468           # .[add, ==> SquareOp(5, 5)
 469           # ..[scale, ==> LinFunc(1, 5)
 470           # ...LinFunc(1, 5),
 471           # ...2.0],
 472           # ..[exp, ==> SquareOp(5, 5)
 473           # ...LinOp(5, 5),
 474           # ...3]],
 475           # .(5,)]
 476        """
 477        fmt = lambda obj, lvl: ("." * lvl) + str(obj)
 478        lines = []
 479
 480        head, *tail = self._expr()
 481        if len(tail) == 0:
 482            head = f"{repr(head)},"
 483        else:
 484            head = f"[{head}, ==> {repr(self)}"
 485        lines.append(fmt(head, level))
 486
 487        for t in tail:
 488            if isinstance(t, Operator):
 489                lines += t.expr(level=level + 1, strip=False).split("\n")
 490            else:
 491                t = f"{t},"
 492                lines.append(fmt(t, level + 1))
 493        if len(tail) > 0:
 494            # Drop comma for last tail item, then close the sub-expression.
 495            lines[-1] = lines[-1][:-1]
 496            lines[-1] += "],"
 497
 498        out = "\n".join(lines)
 499        if strip:
 500            out = out.strip(",")  # drop comma at top-level tail.
 501        return out

 548    def lipschitz(self) -> pxt.Real:
 549        r"""
 550        Return the last computed Lipschitz constant of :math:`\mathbf{f}`.
 551
 552        Notes
 553        -----
 554        * If a Lipschitz constant is known apriori, it can be stored in the instance as follows:
 555
 556          .. code-block:: python3
 557
 558             class TestOp(Map):
 559                 def __init__(self, shape):
 560                     super().__init__(shape=shape)
 561                     self.lipschitz = 2
 562
 563             op = TestOp(shape=(2, 3))
 564             op.lipschitz  # => 2
 565
 566
 567          Alternatively the Lipschitz constant can be set manually after initialization:
 568
 569          .. code-block:: python3
 570
 571             class TestOp(Map):
 572                 def __init__(self, shape):
 573                     super().__init__(shape=shape)
 574
 575             op = TestOp(shape=(2,3))
 576             op.lipschitz  # => inf, since unknown apriori
 577
 578             op.lipschitz = 2  # post-init specification
 579             op.lipschitz  # => 2
 580
 581        * :py:meth:`~pyxu.abc.operator.Map.lipschitz` **never** computes anything:
 582          call :py:meth:`~pyxu.abc.operator.Map.estimate_lipschitz` manually to *compute* a new Lipschitz estimate:
 583
 584          .. code-block:: python3
 585
 586             op.lipschitz = op.estimate_lipschitz()
 587        """
 588        if not hasattr(self, "_lipschitz"):
 589            self._lipschitz = self.estimate_lipschitz()
 590
 591        return pxrt.coerce(self._lipschitz)

 606    def estimate_lipschitz(self, **kwargs) -> pxt.Real:
 607        r"""
 608        Compute a Lipschitz constant of the operator.
 609
 610        Parameters
 611        ----------
 612        kwargs: ~collections.abc.Mapping
 613            Class-specific kwargs to configure Lipschitz estimation.
 614
 615        Notes
 616        -----
 617        * This method should always be callable without specifying any kwargs.
 618
 619        * A constant :math:`L_{\mathbf{f}} > 0` is said to be a *Lipschitz constant* for a map
 620          :math:`\mathbf{f}: \mathbb{R}^{M} \to \mathbb{R}^{N}` if:
 621
 622          .. math::
 623
 624             \|\mathbf{f}(\mathbf{x}) - \mathbf{f}(\mathbf{y})\|_{\mathbb{R}^{N}}
 625             \leq
 626             L_{\mathbf{f}} \|\mathbf{x} - \mathbf{y}\|_{\mathbb{R}^{M}},
 627             \qquad
 628             \forall \mathbf{x}, \mathbf{y}\in \mathbb{R}^{M},
 629
 630          where
 631          :math:`\|\cdot\|_{\mathbb{R}^{M}}` and
 632          :math:`\|\cdot\|_{\mathbb{R}^{N}}`
 633          are the canonical norms on their respective spaces.
 634
 635          The smallest Lipschitz constant of a map is called the *optimal Lipschitz constant*.
 636        """
 637        raise NotImplementedError

 727        multi-dimensional map.
 728        The *Jacobian* (or *differential*) of :math:`\mathbf{f}` at
 729        :math:`\mathbf{z} \in \mathbb{R}^{M}` is defined as the best linear approximator of
 730        :math:`\mathbf{f}` near :math:`\mathbf{z}`, in the following sense:
 731
 732        .. math::
 733
 734           \mathbf{f}(\mathbf{x}) - \mathbf{f}(\mathbf{z})
 735           =
 736           \mathbf{J}_{\mathbf{f}}(\mathbf{z}) (\mathbf{x} - \mathbf{z})
 737           +
 738           o(\| \mathbf{x} - \mathbf{z} \|)
 739           \quad
 740           \text{as} \quad \mathbf{x} \to \mathbf{z}.

 743
 744        .. math::
 745
 746           [\mathbf{J}_{\mathbf{f}}(\mathbf{x})]_{ij}
 747           :=
 748           \frac{\partial f_{i}}{\partial x_{j}}(\mathbf{x}),

 946    def moreau_envelope(self, mu: pxt.Real) -> pxt.OpT:
 947        r"""
 948        Approximate proximable functional :math:`f` by its *Moreau envelope* :math:`f^{\mu}`.
 949
 950        Parameters
 951        ----------
 952        mu: Real
 953            Positive regularization parameter.
 954
 955        Returns
 956        -------
 957        op: OpT
 958            Differential Moreau envelope.
 959
 960        Notes
 961        -----
 962        Consider a convex non-smooth proximable functional
 963        :math:`f: \mathbb{R}^{M} \to \mathbb{R} \cup \{+\infty\}`
 964        and a regularization parameter :math:`\mu > 0`.
 965        The :math:`\mu`-*Moreau envelope* (or *Moreau-Yoshida envelope*) of :math:`f` is given by
 966
 967        .. math::
 968
 969           f^{\mu}(\mathbf{x})
 970           =
 971           \min_{\mathbf{z} \in \mathbb{R}^{M}}
 972           f(\mathbf{z})
 973           \quad + \quad
 974           \frac{1}{2\mu} \|\mathbf{x} - \mathbf{z}\|^{2}.
 975
 976        The parameter :math:`\mu` controls the trade-off between the regularity properties of :math:`f^{\mu}` and
 977        the approximation error incurred by the Moreau-Yoshida regularization.
 978
 979        The Moreau envelope inherits the convexity of :math:`f` and is gradient-Lipschitz (with Lipschitz constant
 980        :math:`\mu^{-1}`), even if :math:`f` is non-smooth.
 981        Its gradient is moreover given by:
 982
 983        .. math::
 984
 985           \nabla f^{\mu}(\mathbf{x})
 986           =
 987           \mu^{-1} \left(\mathbf{x} - \text{prox}_{\mu f}(\mathbf{x})\right).
 988
 989        In addition, :math:`f^{\mu}` envelopes :math:`f` from below:
 990        :math:`f^{\mu}(\mathbf{x}) \leq f(\mathbf{x})`.
 991        This envelope becomes tighter as :math:`\mu \to 0`:
 992
 993        .. math::
 994
 995           \lim_{\mu \to 0} f^{\mu}(\mathbf{x}) = f(\mathbf{x}).
 996
 997        Finally, it can be shown that the minimizers of :math:`f` and :math:`f^{\mu}` coincide, and
 998        that the Fenchel conjugate of :math:`f^{\mu}` is strongly-convex.
 999
1000        Example
1001        -------
1002        Construct and plot the Moreau envelope of the :math:`\ell_{1}`-norm:
1003
1004        .. plot::
1005
1006           import numpy as np
1007           import matplotlib.pyplot as plt
1008           from pyxu.abc import ProxFunc
1009
1010           class L1Norm(ProxFunc):
1011               def __init__(self, dim: int):
1012                   super().__init__(shape=(1, dim))
1013               def apply(self, arr):
1014                   return np.linalg.norm(arr, axis=-1, keepdims=True, ord=1)
1015               def prox(self, arr, tau):
1016                   return np.clip(np.abs(arr)-tau, a_min=0, a_max=None) * np.sign(arr)
1017
1018           N = 512
1019           l1_norm = L1Norm(dim=N)
1020           mus = [0.1, 0.5, 1]
1021           smooth_l1_norms = [l1_norm.moreau_envelope(mu) for mu in mus]
1022
1023           x = np.linspace(-1, 1, N)[:, None]
1024           labels=['mu=0']
1025           labels.extend([f'mu={mu}' for mu in mus])
1026           plt.figure()
1027           plt.plot(x, l1_norm(x))
1028           for f in smooth_l1_norms:
1029               plt.plot(x, f(x))
1030           plt.legend(labels)
1031           plt.title('Moreau Envelope')
1032
1033           labels=[f'mu={mu}' for mu in mus]
1034           plt.figure()
1035           for f in smooth_l1_norms:
1036               plt.plot(x, f.grad(x))
1037           plt.legend(labels)
1038           plt.title('Derivative of Moreau Envelope')
1039        """
1040        from pyxu.operator.interop.source import from_source
1041
1042        assert mu > 0, f"mu: expected positive, got {mu}"
1043
1044        @pxrt.enforce_precision(i="arr")
1045        def op_apply(_, arr):
1046            from pyxu.math.linalg import norm
1047
1048            x = self.prox(arr, tau=_._mu)
1049            out = pxu.copy_if_unsafe(self.apply(x))
1050            out += (0.5 / _._mu) * norm(arr - x, axis=-1, keepdims=True) ** 2
1051            return out
1052
1053        @pxrt.enforce_precision(i="arr")
1054        def op_grad(_, arr):
1055            x = arr.copy()
1056            x -= self.prox(arr, tau=_._mu)
1057            x /= _._mu
1058            return x
1059
1060        op = from_source(
1061            cls=DiffFunc,
1062            shape=self.shape,
1063            embed=dict(
1064                _name="moreau_envelope",
1065                _mu=mu,
1066                _diff_lipschitz=float(1 / mu),
1067            ),
1068            apply=op_apply,
1069            grad=op_grad,
1070            _expr=lambda _: ("moreau_envelope", _, _._mu),
1071        )
1072        return op

1440    def estimate_lipschitz(self, **kwargs) -> pxt.Real:
1441        r"""
1442        Compute a Lipschitz constant of the operator.
1443
1444        Parameters
1445        ----------
1446        method: "svd" | "trace"
1447
1448            * If `svd`, compute the optimal Lipschitz constant.
1449            * If `trace`, compute an upper bound. (Default)
1450
1451        kwargs:
1452            Optional kwargs passed on to:
1453
1454            * `svd`: :py:func:`~pyxu.abc.operator.LinOp.svdvals`
1455            * `trace`: :py:func:`~pyxu.math.linalg.hutchpp`
1456
1457        Notes
1458        -----
1459        * The tightest Lipschitz constant is given by the spectral norm of the operator :math:`\mathbf{A}`:
1460          :math:`\|\mathbf{A}\|_{2}`.
1461          It can be computed via the SVD, which is compute-intensive task for large operators.
1462          In this setting, it may be advantageous to overestimate the Lipschitz constant with the
1463          Frobenius norm of :math:`\mathbf{A}` since :math:`\|\mathbf{A}\|_{F} \geq \|\mathbf{A}\|_{2}`.
1464
1465          :math:`\|\mathbf{A}\|_{F}` can be efficiently approximated by computing the trace of :math:`\mathbf{A}^{\ast} \mathbf{A}`
1466          (or :math:`\mathbf{A}\mathbf{A}^{\ast}`) via the `Hutch++ stochastic algorithm <https://arxiv.org/abs/2010.09649>`_.
1467
1468        * :math:`\|\mathbf{A}\|_{F}` is upper-bounded by :math:`\|\mathbf{A}\|_{F} \leq \sqrt{n} \|\mathbf{A}\|_{2}`,
1469          where the equality is reached (worst-case scenario) when the eigenspectrum of the linear operator is flat.
1470        """
1471        method = kwargs.get("method", "trace").lower().strip()
1472
1473        if method == "svd":
1474            # svdvals() may have alternative signature in specialized classes, but we must always use
1475            # the LinOp.svdvals() interface below for kwargs-filtering.
1476            func, sig_func = self.__class__.svdvals, LinOp.svdvals
1477            kwargs.update(
1478                k=1,
1479                which="LM",
1480            )
1481            estimate = lambda: func(self, **kwargs).item()
1482        elif method == "trace":
1483            if self.shape == (1, 1):
1484                # [Sepand] Special case of degenerate LinOps (which are not LinFuncs).
1485                # Cannot call .gram() below since it will recurse indefinitely.
1486                # [Reason: asop().squeeze() combinations don't work for (1, 1) degenerate LinOps.]
1487                func = sig_func = LinFunc.estimate_lipschitz
1488                estimate = lambda: func(self.squeeze(), **kwargs)
1489            else:
1490                from pyxu.math.linalg import hutchpp as func
1491
1492                sig_func = func
1493
1494                kwargs.update(
1495                    op=self.gram() if (self.codim >= self.dim) else self.cogram(),
1496                    m=kwargs.get("m", 126),
1497                )
1498                estimate = lambda: np.sqrt(func(**kwargs)).item()
1499        else:
1500            raise NotImplementedError
1501
1502        # Filter unsupported kwargs
1503        sig = inspect.Signature.from_callable(sig_func)
1504        kwargs = {k: v for (k, v) in kwargs.items() if (k in sig.parameters)}
1505
1506        L = estimate()
1507        return L

1510    def svdvals(
1511        self,
1512        k: pxt.Integer,
1513        which: str = "LM",
1514        gpu: bool = False,
1515        **kwargs,
1516    ) -> pxt.NDArray:
1517        r"""
1518        Compute singular values of a linear operator.
1519
1520        Parameters
1521        ----------
1522        k: Integer
1523            Number of singular values to compute.
1524        which: "LM" | "SM"
1525            Which `k` singular values to find:
1526
1527                * `LM`: largest magnitude
1528                * `SM`: smallest magnitude
1529        gpu: bool
1530            If ``True`` the singular value decomposition is performed on the GPU.
1531        kwargs: ~collections.abc.Mapping
1532            Additional kwargs accepted by :py:func:`~scipy.sparse.linalg.svds`.
1533
1534        Returns
1535        -------
1536        D: NDArray
1537            (k,) singular values in ascending order.
1538        """
1539        which = which.upper().strip()
1540        assert which in {"SM", "LM"}
1541
1542        def _dense_eval():
1543            if gpu:
1544                import cupy as xp
1545                import cupy.linalg as spx
1546            else:
1547                import numpy as xp
1548                import scipy.linalg as spx
1549            op = self.asarray(xp=xp, dtype=pxrt.getPrecision().value)
1550            return spx.svd(op, compute_uv=False)
1551
1552        def _sparse_eval():
1553            if gpu:
1554                assert pxd.CUPY_ENABLED
1555                import cupyx.scipy.sparse.linalg as spx
1556
1557                self._warn_vals_sparse_gpu()
1558            else:
1559                spx = spsl
1560            op = self.to_sciop(gpu=gpu, dtype=pxrt.getPrecision().value)
1561            kwargs.update(
1562                k=k,
1563                which=which,
1564                return_singular_vectors=False,
1565                # random_state=0,  # unsupported by CuPy
1566            )
1567            return spx.svds(op, **kwargs)
1568
1569        if k >= min(self.shape) // 2:
1570            msg = "Too many svdvals wanted: using matrix-based ops."
1571            warnings.warn(msg, pxw.DenseWarning)
1572            D = _dense_eval()
1573        else:
1574            D = _sparse_eval()
1575
1576        # Filter to k largest/smallest magnitude + sorted
1577        xp = pxu.get_array_module(D)
1578        D = D[xp.argsort(D)]
1579        return D[:k] if (which == "SM") else D[-k:]

1665    def pinv(
1666        self,
1667        arr: pxt.NDArray,
1668        damp: pxt.Real,
1669        kwargs_init=None,
1670        kwargs_fit=None,
1671    ) -> pxt.NDArray:
1672        r"""
1673        Evaluate the Moore-Penrose pseudo-inverse :math:`\mathbf{A}^{\dagger}` at specified point(s).
1674
1675        Parameters
1676        ----------
1677        arr: NDArray
1678            (..., N) input points.
1679        damp: Real
1680            Positive dampening factor regularizing the pseudo-inverse.
1681        kwargs_init: ~collections.abc.Mapping
1682            Optional kwargs to be passed to :py:meth:`~pyxu.opt.solver.cg.CG`'s ``__init__()`` method.
1683        kwargs_fit: ~collections.abc.Mapping
1684            Optional kwargs to be passed to :py:meth:`~pyxu.opt.solver.cg.CG`'s ``fit()`` method.
1685
1686        Returns
1687        -------
1688        out: NDArray
1689            (..., M) pseudo-inverse(s).
1690
1691        Notes
1692        -----
1693        The Moore-Penrose pseudo-inverse of an operator :math:`\mathbf{A}: \mathbb{R}^{M} \to \mathbb{R}^{N}` is
1694        defined as the operator :math:`\mathbf{A}^{\dagger}: \mathbb{R}^{N} \to \mathbb{R}^{M}` verifying the
1695        Moore-Penrose conditions:
1696
1697            1. :math:`\mathbf{A} \mathbf{A}^{\dagger} \mathbf{A} = \mathbf{A}`,
1698            2. :math:`\mathbf{A}^{\dagger} \mathbf{A} \mathbf{A}^{\dagger} = \mathbf{A}^{\dagger}`,
1699            3. :math:`(\mathbf{A}^{\dagger} \mathbf{A})^{\ast} = \mathbf{A}^{\dagger} \mathbf{A}`,
1700            4. :math:`(\mathbf{A} \mathbf{A}^{\dagger})^{\ast} = \mathbf{A} \mathbf{A}^{\dagger}`.
1701
1702        This operator exists and is unique for any finite-dimensional linear operator.
1703        The action of the pseudo-inverse :math:`\mathbf{A}^{\dagger} \mathbf{y}` for every
1704        :math:`\mathbf{y} \in \mathbb{R}^{N}` can be computed in matrix-free fashion by solving the
1705        *normal equations*:
1706
1707        .. math::
1708
1709           \mathbf{A}^{\ast} \mathbf{A} \mathbf{x} = \mathbf{A}^{\ast} \mathbf{y}
1710           \quad\Leftrightarrow\quad
1711           \mathbf{x} = \mathbf{A}^{\dagger} \mathbf{y},
1712           \quad
1713           \forall (\mathbf{x}, \mathbf{y}) \in \mathbb{R}^{M} \times \mathbb{R}^{N}.
1714
1715        In the case of severe ill-conditioning, it is possible to consider the dampened normal
1716        equations for a numerically-stabler approximation of :math:`\mathbf{A}^{\dagger} \mathbf{y}`:
1717
1718        .. math::
1719
1720           (\mathbf{A}^{\ast} \mathbf{A} + \tau I) \mathbf{x} = \mathbf{A}^{\ast} \mathbf{y},
1721
1722        where :math:`\tau > 0` corresponds to the `damp` parameter.
1723        """
1724        from pyxu.operator.linop import HomothetyOp
1725        from pyxu.opt.solver import CG
1726        from pyxu.opt.stop import MaxIter
1727
1728        kwargs_fit = dict() if kwargs_fit is None else kwargs_fit
1729        kwargs_init = dict() if kwargs_init is None else kwargs_init
1730        kwargs_init.update(show_progress=kwargs_init.get("show_progress", False))
1731
1732        if np.isclose(damp, 0):
1733            A = self.gram()
1734        else:
1735            A = self.gram() + HomothetyOp(cst=damp, dim=self.dim)
1736
1737        cg = CG(A, **kwargs_init)
1738        if "stop_crit" not in kwargs_fit:
1739            # .pinv() may not have sufficiently converged given the default CG stopping criteria.
1740            # To avoid infinite loops, CG iterations are thresholded.
1741            sentinel = MaxIter(n=20 * A.dim)
1742            kwargs_fit["stop_crit"] = cg.default_stop_crit() | sentinel
1743
1744        b = self.adjoint(arr)
1745        cg.fit(b=b, **kwargs_fit)
1746        return cg.solution()

2046    Base class for real-valued linear functionals :math:`f: \mathbb{R}^{M} \to \mathbb{R}`.

2048    Instances of this class must implement
2049    :py:meth:`~pyxu.abc.operator.Map.apply`, and
2050    :py:meth:`~pyxu.abc.operator.LinOp.adjoint`.
2051
2052    If known, the Lipschitz constant :math:`L` should be stored in the
2053    :py:attr:`~pyxu.abc.operator.Map.lipschitz` property.
2054    """

2056    @classmethod
2057    def properties(cls) -> cabc.Set[Property]:
2058        p = set()

2060            p |= klass.properties()

2071    def estimate_lipschitz(self, **kwargs) -> pxt.Real:
2072        # Try all backends until one works.
2073        for ndi in pxd.NDArrayInfo:
2074            try:
2075                xp = ndi.module()
2076                g = self.grad(xp.ones(self.dim))
2077                L = float(xp.linalg.norm(g))
2078                return L
2079            except Exception:
2080                pass

2084        xp = pxu.get_array_module(arr)
2085        x = xp.ones((*arr.shape[:-1], 1), dtype=arr.dtype)
2086        g = self.adjoint(x)
2087        return g

2091        # out = arr - tau * self.grad(arr)
2092        out = pxu.copy_if_unsafe(self.grad(arr))
