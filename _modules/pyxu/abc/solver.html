119class Solver:
120    r"""
121    Iterative solver for minimization problems of the form
122
123    .. math::
124
125       \hat{x} = \arg\min_{x \in \mathbb{R}^{N}} \mathcal{F}(x),
126
127    where the form of :math:`\mathcal{F}` is solver-dependent.
128
129    Solver provides a versatile API for solving optimisation problems, with the following features:
130
131    * manual/automatic/background execution of solver iterations via parameters provided to
132      :py:meth:`~pyxu.abc.solver.Solver.fit`. (See below.)
133    * automatic checkpointing of solver progress, providing a safe restore point in case of
134      faulty numerical code.
135      Each solver instance backs its state and final output to a folder on disk for
136      post-analysis.
137      In particular :py:meth:`~pyxu.abc.solver.Solver.fit` will never crash: detailed exception information will always
138      be available in a logfile for post-analysis.
139    * arbitrary specification of complex stopping criteria via the
140      :py:class:`~pyxu.abc.solver.StoppingCriterion` class.
141    * solve for multiple initial points in parallel.
142
143    To implement a new iterative solver, users need to sub-class
144    :py:class:`~pyxu.abc.solver.Solver`
145    and overwrite the methods below:
146
147    * :py:meth:`~pyxu.abc.solver.Solver.__init__`
148    * :py:meth:`~pyxu.abc.solver.Solver.m_init`  [i.e. math-init()]
149    * :py:meth:`~pyxu.abc.solver.Solver.m_step`  [i.e. math-step()]
150    * :py:meth:`~pyxu.abc.solver.Solver.default_stop_crit`  [optional; see method definition for details]
151    * :py:meth:`~pyxu.abc.solver.Solver.objective_func`  [optional; see method definition for details]
152
153    Advanced functionalities of
154    :py:class:`~pyxu.abc.solver.Solver`
155    are automatically inherited by sub-classes.
156
157
158    Examples
159    --------
160    Here are examples on how to solve minimization problems with this class:
161
162    .. code-block:: python3
163
164       slvr = Solver()
165
166       ### 1. Blocking Mode: .fit() does not return until solver has stopped.
167       >>> slvr.fit(mode=Mode.BLOCK, ...)
168       >>> data, hist = slvr.stats()  # final output of solver.
169
170       ### 2. Async Mode: solver iterations run in the background.
171       >>> slvr.fit(mode=Mode.ASYNC, ...)
172       >>> print('test')  # you can do something in between.
173       >>> slvr.busy()  # or check whether the solver already stopped.
174       >>> slvr.stop()  # and preemptively force it to stop.
175       >>> data, hist = slvr.stats()  # then query the result after a (potential) force-stop.
176
177       ### 3. Manual Mode: fine-grain control of solver data per iteration.
178       >>> slvr.fit(mode=Mode.MANUAL, ...)
179       >>> for data in slvr.steps():
180       ...     # Do something with the logged variables after each iteration.
181       ...     pass  # solver has stopped after the loop.
182       >>> data, hist = slvr.stats()  # final output of solver.
183    """

188    def __init__(
189        self,
190        *,
191        folder: pxt.Path = None,
192        exist_ok: bool = False,
193        stop_rate: pxt.Integer = 1,
194        writeback_rate: pxt.Integer = None,
195        verbosity: pxt.Integer = None,
196        show_progress: bool = True,
197        log_var: pxt.VarName = frozenset(),
198    ):
199        """
200        Parameters
201        ----------
202        folder: Path
203            Directory on disk where instance data should be stored.
204            A location will be automatically chosen if unspecified. (Default: OS-dependent tempdir.)
205        exist_ok: bool
206            If `folder` is specified and `exist_ok` is false (default),
207            :py:class:`FileExistsError`
208            is raised if the target directory already exists.
209        stop_rate: Integer
210            Rate at which solver evaluates stopping criteria.
211        writeback_rate: Integer
212            Rate at which solver checkpoints are written to disk.
213            No checkpointing is done if unspecified: only the final solver output will be written
214            back to disk.
215            Must be a multiple of `stop_rate` if specified.
216        verbosity: Integer
217            Rate at which stopping criteria statistics are logged.
218            Must be a multiple of `stop_rate`.
219            Defaults to `stop_rate` if unspecified.
220        show_progress: bool
221            If True (default) and :py:meth:`~pyxu.abc.solver.Solver.fit` is run with mode=BLOCK,
222            then statistics are also logged to stdout.
223        log_var: VarName
224            Variables from the solver's math-state (:py:attr:`~pyxu.abc.solver.Solver._mstate`)
225            to be logged per iteration.
226            These are the variables made available when calling
227            :py:meth:`~pyxu.abc.solver.Solver.stats`.
228
229        Notes
230        -----
231        * Partial device<>CPU synchronization takes place when stopping-criteria are evaluated.
232          Increasing `stop_rate` is advised to reduce the effect of such transfers when applicable.
233        * Full device<>CPU synchronization takes place at checkpoint-time.
234          Increasing `writeback_rate` is advised to reduce the effect of such transfers when
235          applicable.
236        """
237        self._mstate = dict()
238        self._astate = dict(
239            history=None,  # stopping criteria values per iteration
240            idx=0,  # iteration index
241            log_rate=None,
242            log_var=None,
243            logger=None,
244            stdout=None,
245            stop_crit=None,
246            stop_rate=None,
247            track_objective=None,
248            wb_rate=None,
249            workdir=None,
250            # Execution-mode related -----------
251            mode=None,
252            active=None,
253            worker=None,
254        )
255
256        try:
257            if folder is None:
258                folder = plib.Path(tempfile.mkdtemp(prefix="pyxu_"))
259            elif (folder := plib.Path(folder).expanduser().resolve()).exists() and (not exist_ok):
260                raise FileExistsError(f"{folder} already exists.")
261            else:
262                shutil.rmtree(folder, ignore_errors=True)
263                folder.mkdir(parents=True)
264            self._astate["workdir"] = folder
265        except Exception:
266            raise Exception(f"folder: expected path-like, got {type(folder)}.")
267
268        try:
269            assert stop_rate >= 1
270            self._astate["stop_rate"] = int(stop_rate)
271        except Exception:
272            raise ValueError(f"stop_rate must be positive, got {stop_rate}.")
273
274        try:
275            self._astate["wb_rate"] = writeback_rate
276            if writeback_rate is not None:
277                assert writeback_rate % self._astate["stop_rate"] == 0
278                self._astate["wb_rate"] = int(writeback_rate)
279        except Exception:
280            raise ValueError(f"writeback_rate must be a multiple of stop_rate({stop_rate}), got {writeback_rate}.")
281
282        try:
283            if verbosity is None:
284                verbosity = self._astate["stop_rate"]
285            assert verbosity % self._astate["stop_rate"] == 0
286            self._astate["log_rate"] = int(verbosity)
287            self._astate["stdout"] = bool(show_progress)
288        except Exception:
289            raise ValueError(f"verbosity must be a multiple of stop_rate({stop_rate}), got {verbosity}.")
290
291        try:
292            if isinstance(log_var, str):
293                log_var = (log_var,)
294            self._astate["log_var"] = frozenset(log_var)
295        except Exception:
296            raise ValueError(f"log_var: expected collection, got {type(log_var)}.")

298    def fit(self, **kwargs):
299        r"""
300        Solve minimization problem(s) defined in
301        :py:meth:`~pyxu.abc.solver.Solver.__init__`,
302        with the provided run-specifc parameters.
303
304        Parameters
305        ----------
306        kwargs
307            See class-level docstring for class-specific keyword parameters.
308        stop_crit: StoppingCriterion
309            Stopping criterion to end solver iterations.
310            If unspecified, defaults to :py:meth:`~pyxu.abc.solver.Solver.default_stop_crit`.
311        mode: Mode
312            Execution mode.
313            See :py:class:`~pyxu.abc.solver.Solver` for usage examples.
314
315            Useful method pairs depending on the execution mode:
316
317            * BLOCK: :py:meth:`~pyxu.abc.solver.Solver.fit`
318            * ASYNC: :py:meth:`~pyxu.abc.solver.Solver.fit`, :py:meth:`~pyxu.abc.solver.Solver.busy`, :py:meth:`~pyxu.abc.solver.Solver.stop`

320        track_objective: bool
321            Auto-compute objective function every time stopping criterion is evaluated.
322        """
323        self._fit_init(
324            mode=kwargs.pop("mode", Mode.BLOCK),
325            stop_crit=kwargs.pop("stop_crit", None),
326            track_objective=kwargs.pop("track_objective", False),
327        )
328        self.m_init(**kwargs)
329        self._fit_run()

350    def steps(self, n: pxt.Integer = None) -> cabc.Generator:
351        """
352        Generator of logged variables after each iteration.
353
354        The i-th call to
355        :py:func:`next`
356        on this object returns the logged variables after the i-th solver iteration.
357
358        This method is only usable after calling :py:meth:`~pyxu.abc.solver.Solver.fit` with mode=MANUAL.
359        See :py:class:`~pyxu.abc.solver.Solver` for usage examples.
360
361        There is no guarantee that a checkpoint on disk exists when the generator is exhausted.
362        (Reason: potential exceptions raised during solver's progress.)
363        Users should invoke :py:meth:`~pyxu.abc.solver.Solver.writeback` afterwards if needed.
364
365        Parameters
366        ----------
367        n: Integer
368            Maximum number of :py:func:`next` calls allowed before exhausting the generator.
369            Defaults to infinity if unspecified.
370
371            The generator will terminate prematurely if the solver naturally stops before `n` calls to
372            :py:func:`next` are made.
373        """
374        self._check_mode(Mode.MANUAL)
375        i = 0
376        while (n is None) or (i < n):
377            if self._step():
378                data, _ = self.stats()
379                yield data
380                i += 1
381            else:
382                self._astate["mode"] = None  # force steps() to be call-once when exhausted.
383                self._cleanup_logger()
384                return

389    def stats(self) -> tuple[_stats_data_spec, _stats_history_spec]:
390        """
391        Query solver state.
392
393        Returns
394        -------
395        data: ~collections.abc.Mapping
396            Value(s) of ``log_var`` (s) after last iteration.
397        history: numpy.ndarray, :py:obj:`None`
398            (N_iter,) records of stopping-criteria values per iteration.
399
400        Notes
401        -----
402        If any of the ``log_var`` (s) and/or ``history`` are not (yet) known at query time, ``None`` is
403        returned.
404        """
405        history = self._astate["history"]
406        if history is not None:
407            if len(history) > 0:
408                history = np.concatenate(history, dtype=history[0].dtype, axis=0)
409            else:
410                history = None
411        data = {k: self._mstate.get(k) for k in self._astate["log_var"]}
412        return data, history

471        """
472        Stop an async-running solver.
473
474        This method is only usable after calling :py:meth:`~pyxu.abc.solver.Solver.fit` with mode=ASYNC.
475        See :py:class:`~pyxu.abc.solver.Solver` for usage examples.
476
477        This method will block until the solver has stopped.
478
479        There is no guarantee that a checkpoint on disk exists once halted.
480        (Reason: potential exceptions raised during solver's progress.)
481        Users should invoke :py:meth:`~pyxu.abc.solver.Solver.writeback` afterwards if needed.
482
483        Users must call this method to terminate an async-solver, even if
484        :py:meth:`~pyxu.abc.solver.Solver.busy` is False.
485        """
486        self._check_mode(Mode.ASYNC, Mode.BLOCK)
487        self._astate["active"].clear()
488        self._astate["worker"].join()

490            mode=None,  # forces stop() to be call-once.
491            active=None,
492            worker=None,
493        )
494        self._cleanup_logger()

588    def _step(self) -> bool:
589        ast = self._astate  # shorthand
590
591        must_stop = lambda: ast["idx"] % ast["stop_rate"] == 0
592        must_log = lambda: ast["idx"] % ast["log_rate"] == 0
593        must_writeback = lambda: (ast["wb_rate"] is not None) and (ast["idx"] % ast["wb_rate"] == 0)
594
595        def _log(msg: str = None):
596            if msg is None:  # report stopping-criterion values
597                h = ast["history"][-1][0]
598                msg = [f"[{dt.datetime.now()}] Iteration {ast['idx']:>_d}"]
599                for field, value in zip(h.dtype.names, h):
600                    msg.append(f"\t{field}: {value}")
601                msg = "\n".join(msg)
602            ast["logger"].info(msg)
603
604        def _update_history():
605            def _as_struct(data: dict[str, float]) -> np.ndarray:
606                ftype = pxrt.getPrecision().value
607                spec_data = [(k, ftype) for k in data]
608
609                itype = np.int64
610                spec_iter = [("iteration", itype)]
611
612                dtype = np.dtype(spec_iter + spec_data)
613
614                utype = np.uint8
615                s = np.concatenate(  # to allow mixed int/float fields:
616                    [  # (1) cast to uint, then (2) to compound dtype.
617                        np.array([ast["idx"]], dtype=itype).view(utype),
618                        np.array(list(data.values()), dtype=ftype).view(utype),
619                    ]
620                ).view(dtype)
621                return s
622
623            h = _as_struct(ast["stop_crit"].info())
624            ast["history"].append(h)
625
626        # [Sepand] Important
627        # stop_crit.stop(), _update_history(), _log() must always be called in this order.
628
629        try:
630            _ms, _ml, _mw = must_stop(), must_log(), must_writeback()
631
632            if _ms and ast["track_objective"]:
633                self._mstate["objective_func"] = self.objective_func().reshape(-1)
634
635            if _ms and ast["stop_crit"].stop(self._mstate):
636                _update_history()
637                _log()
638                _log(msg=f"[{dt.datetime.now()}] Stopping Criterion satisfied -> END")
639                self.writeback()
640                return False
641            else:
642                if _ms:
643                    _update_history()
644                if _ml:
645                    _log()
646                if _mw:
647                    self.writeback()
648                ast["idx"] += 1
649                self.m_step()
650                if _ms:
651                    self._m_persist()
652                return True
653        except Exception as e:
654            msg = f"[{dt.datetime.now()}] Something went wrong -> EXCEPTION RAISED"
655            msg_xtra = f"More information: {self.logfile}."
656            print("\n".join([msg, msg_xtra]), file=sys.stderr)
657            if ast["wb_rate"] is not None:  # checkpointing enabled
658                _, r = divmod(ast["idx"], ast["wb_rate"])
659                idx_valid = ast["idx"] - r
660                msg_idx = f"Last valid checkpoint done at iteration={idx_valid}."
661                msg = "\n".join([msg, msg_idx])
662            ast["logger"].exception(msg, exc_info=e)
663            return False
