 12class CG(pxa.Solver):
 13    r"""
 14    Conjugate Gradient Method.
 15
 16    The Conjugate Gradient method solves the minimization problem
 17
 18    .. math::
 19
 20       \min_{x\in\mathbb{R}^{N}} \frac{1}{2} \mathbf{x}^{T} \mathbf{A} \mathbf{x} - \mathbf{x}^{T} \mathbf{b},
 21
 22    where :math:`\mathbf{A}: \mathbb{R}^{N} \to \mathbb{R}^{N}` is a *symmetric* *positive definite*
 23    operator, and :math:`\mathbf{b} \in \mathbb{R}^{N}`.
 24
 25    The norm of the `explicit residual <https://www.wikiwand.com/en/Conjugate_gradient_method>`_
 26    :math:`\mathbf {r}_{k+1}:=\mathbf{b}-\mathbf{Ax}_{k+1}` is used as the default stopping criterion.
 27    This provides a guaranteed level of accuracy both in exact arithmetic and in the presence of
 28    round-off errors.
 29    By default, the iterations stop when the norm of the explicit residual is smaller than 1e-4.
 30
 31    Parameters (``__init__()``)
 32    ---------------------------
 33    * **A** (:py:class:`~pyxu.abc.operator.PosDefOp`)
 34      --
 35      Positive-definite operator :math:`\mathbf{A}: \mathbb{R}^{N} \to \mathbb{R}^{N}`.
 36    * **\*\*kwargs** (:py:class:`~collections.abc.Mapping`)
 37      --
 38      Other keyword parameters passed on to :py:meth:`pyxu.abc.solver.Solver.__init__`.
 39
 40    Parameters (``fit()``)
 41    ----------------------
 42    * **b** (:py:attr:`~pyxu.info.ptype.NDArray`)
 43      --
 44      (..., N) :math:`\mathbf{b}` terms in the CG cost function.
 45
 46      All problems are solved in parallel.
 47    * **x0** (:py:attr:`~pyxu.info.ptype.NDArray`, :py:obj:`None`)
 48      --
 49      (..., N) initial point(s).
 50
 51      Must be broadcastable with `b` if provided.
 52      Defaults to 0.
 53    * **restart_rate** (:py:attr:`~pyxu.info.ptype.Integer`)
 54      --
 55      Number of iterations after which restart is applied.
 56
 57      By default, a restart is done after 'n' iterations, where 'n' corresponds to the dimension of :math:`\mathbf{A}`.
 58    * **\*\*kwargs** (:py:class:`~collections.abc.Mapping`)
 59      --
 60      Other keyword parameters passed on to :py:meth:`pyxu.abc.solver.Solver.fit`.
 61    """
 62
 63    def __init__(self, A: pxa.PosDefOp, **kwargs):
 64        kwargs.update(
 65            log_var=kwargs.get("log_var", ("x",)),
 66        )
 67        super().__init__(**kwargs)
 68
 69        self._A = A

 71    @pxrt.enforce_precision(i=("b", "x0"))
 72    def m_init(
 73        self,
 74        b: pxt.NDArray,
 75        x0: pxt.NDArray = None,
 76        restart_rate: pxt.Integer = None,
 77    ):
 78        r"""
 79        Parameters
 80        ----------
 81        b: NDArray
 82            (..., N) :math:`\mathbf{b}` terms in the CG cost function.
 83            All problems are solved in parallel.
 84        x0: NDArray
 85            (..., N) initial point(s).
 86
 87            Must be broadcastable with `b` if specified.
 88            Defaults to 0.
 89        restart_rate: Integer
 90            Number of iterations after which restart is applied.
 91            By default, a restart is done after 'n' iterations, where 'n' corresponds to the dimension of
 92            the linear operator :math:`\mathbf{A}`.
 93
 94        Note
 95        ----
 96        `Restarts <https://www.wikiwand.com/en/Conjugate_gradient_method>`_ may slow down convergence,
 97        but improve stability due to round-off error or ill-conditioning of the linear operator.
 98        If these issues are suspected, adjust `restart_rate` accordingly.
 99        """
100        mst = self._mstate  # shorthand
101
102        if restart_rate is not None:
103            assert restart_rate >= 1
104            mst["restart_rate"] = int(restart_rate)
105        else:
106            mst["restart_rate"] = self._A.dim
107
108        xp = pxu.get_array_module(b)
109        if x0 is None:
110            mst["b"] = b
111            mst["x"] = xp.zeros_like(b)
112        elif b.shape == x0.shape:
113            # No broadcasting involved
114            mst["b"] = b
115            mst["x"] = x0.copy()
116        else:
117            # In-place updates involving b/x won't work if shapes differ -> coerce to largest.
118            mst["b"], mst["x"] = xp.broadcast_arrays(b, x0)
119            mst["x"] = mst["x"].copy()
120
121        mst["residual"] = mst["b"].copy()
122        mst["residual"] -= self._A.apply(mst["x"])
123        mst["conjugate_dir"] = mst["residual"].copy()
124
125    def m_step(self):
126        mst = self._mstate  # shorthand
127        x, r, p = mst["x"], mst["residual"], mst["conjugate_dir"]
128        xp = pxu.get_array_module(x)
129
130        Ap = self._A.apply(p)
131        rr = pxlg.norm(r, ord=2, axis=-1, keepdims=True) ** 2
132        alpha = rr / (p * Ap).sum(axis=-1, keepdims=True)
133        x += alpha * p
134
135        if pxu.compute(xp.any(rr <= pxrt.Width(rr.dtype).eps())):  # explicit eval
136            r[:] = mst["b"]
137            r -= self._A.apply(x)
138        else:  # implicit eval
139            r -= alpha * Ap
140
141        # Because CG can only generate n conjugate vectors in an n-dimensional space, it makes sense
142        # to restart CG every n iterations.
143        if self._astate["idx"] % mst["restart_rate"] == 0:  # explicit eval
144            beta = 0
145            r[:] = mst["b"]
146            r -= self._A.apply(x)
147        else:  # implicit eval
148            beta = pxlg.norm(r, ord=2, axis=-1, keepdims=True) ** 2 / rr
149        p *= beta
150        p += r
151
152        # for homogenity with other solver code. Optional in CG due to in-place computations.
153        mst["x"], mst["residual"], mst["conjugate_dir"] = x, r, p
154
155    def default_stop_crit(self) -> pxa.StoppingCriterion:
156        from pyxu.opt.stop import AbsError
157
158        stop_crit = AbsError(
159            eps=1e-4,
160            var="residual",
161            f=None,
162            norm=2,
163            satisfy_all=True,
164        )
165        return stop_crit
166
167    def objective_func(self) -> pxt.NDArray:
168        x = self._mstate["x"]
169        b = self._mstate["b"]
170
171        f = self._A.apply(x)
172        f = pxu.copy_if_unsafe(f)
173        f /= 2
174        f -= b
175        f *= x
176
177        return f.sum(axis=-1, keepdims=True)
178
179    def solution(self) -> pxt.NDArray:
180        """
181        Returns
182        -------
183        x: NDArray
184            (..., N) solution.
185        """
186        data, _ = self.stats()
187        return data.get("x")
