 18class ProxAdam(pxa.Solver):
 19    r"""
 20    Proximal Adam solver [ProxAdam]_.
 21
 22    ProxAdam minimizes
 23
 24    .. math::
 25
 26       {\min_{\mathbf{x}\in\mathbb{R}^N} \;\mathcal{F}(\mathbf{x})\;\;+\;\;\mathcal{G}(\mathbf{x})},
 27
 28    where:
 29
 30    * :math:`\mathcal{F}:\mathbb{R}^N\rightarrow \mathbb{R}` is *convex* and *differentiable*, with
 31      :math:`\beta`-*Lipschitz continuous* gradient, for some :math:`\beta\in[0,+\infty[`.
 32    * :math:`\mathcal{G}:\mathbb{R}^N\rightarrow \mathbb{R}\cup\{+\infty\}` is a *proper*, *lower
 33      semicontinuous* and *convex function* with a *simple proximal operator*.
 34
 35    ProxAdam is a suitable alternative to Proximal Gradient Descent
 36    (:py:class:`~pyxu.opt.solver.pgd.PGD`) when:
 37
 38    * computing :math:`\beta` to optimally choose the step size is infeasible, and
 39    * line-search methods to estimate step sizes are too expensive.
 40
 41    Compared to PGD, ProxAdam:
 42
 43    * auto-tunes gradient updates based on stochastic estimates of
 44      :math:`\phi_{t} = \mathbb{E}[\nabla\mathcal{F}]` and :math:`\psi_{t} =
 45      \mathbb{E}[\nabla\mathcal{F}^{2}]` respectively;
 46    * uses a modified proximity operator at each iteration to update coordinates at varying scales:
 47
 48      .. math::
 49
 50         \text{prox}_{\alpha\mathcal{G}}(\mathbf{x}_{t})
 51         =
 52         \min_{\mathbf{z}\in\mathbb{R}^N} \;\mathcal{G}(\mathbf{z})\;\;+\;\;\frac{1}{2\alpha}||\mathbf{z}-\mathbf{x}_t||_H^2,
 53
 54      where :math:`H=\text{diag}(\psi_t)`.
 55      The modified proximity operator is computed by solving a PGD sub-problem.
 56
 57    ProxAdam has many named variants for particular choices of :math:`\phi` and :math:`\psi`:
 58
 59    * Adam:
 60
 61      .. math::
 62
 63         \phi_t
 64         =
 65         \frac{
 66           \mathbf{m}_t
 67         }{
 68           1-\beta_1^t
 69         }
 70         \qquad
 71         \psi_t
 72         =
 73         \sqrt{
 74           \frac{
 75             \mathbf{v}_t
 76           }{
 77             1-\beta_2^t
 78           }
 79         } + \epsilon,
 80
 81    * AMSGrad:
 82
 83      .. math::
 84
 85         \phi_t = \mathbf{m}_t
 86         \qquad
 87         \psi_t = \sqrt{\hat{\mathbf{v}}_t},
 88
 89    * PAdam:
 90
 91      .. math::
 92
 93         \phi_t = \mathbf{m}_t
 94         \qquad
 95         \psi_t = \hat{\mathbf{v}}_t^p,
 96
 97    where in all cases:
 98
 99    .. math::
100
101       \mathbf{m}_t
102       =
103       \beta_1\mathbf{m}_{t-1}
104       +
105       (1-\beta_1)\mathbf{g}_t \\
106       \mathbf{v}_t
107       =
108       \beta_2\mathbf{v}_{t-1}
109       +
110       (1-\beta_2)\mathbf{g}_t^2\\
111       \hat{\mathbf{v}}_t
112       =
113       \max(\hat{\mathbf{v}}_{t-1}, \mathbf{v}_t),
114
115    with :math:`\mathbf{m}_0 = \mathbf{v}_0 = \mathbf{0}`.
116
117    Remarks
118    -------
119    * The algorithm is still valid if :math:`\mathcal{G}` is zero.
120
121    * The convergence is guaranteed for step sizes :math:`\alpha\leq 2/\beta`.
122
123    * The default stopping criterion is the relative norm change of the primal variable.
124      By default, the algorithm stops when the norm of the difference between two consecutive iterates
125      :math:`\{\mathbf{x}_n\}_{n\in\mathbb{N}}` is smaller than 1e-4.
126      Different stopping criteria can be used.
127      It is recommended to change the stopping criterion when using the PAdam and AMSGrad variants to
128      avoid premature stops.
129      By default, the same stopping criterion is used for the proximal sub-problem.
130
131    Parameters (``__init__()``)
132    ---------------------------
133    * **f** (:py:class:`~pyxu.abc.operator.DiffFunc`)
134      --
135      Differentiable function :math:`\mathcal{F}`.
136    * **g** (:py:class:`~pyxu.abc.operator.ProxFunc`, :py:obj:`None`)
137      --
138      Proximable function :math:`\mathcal{G}`.
139    * **\*\*kwargs** (:py:class:`~collections.abc.Mapping`)
140      --
141      Other keyword parameters passed on to :py:meth:`pyxu.abc.solver.Solver.__init__`.
142
143    Parameters (``fit()``)
144    ----------------------
145    * **x0** (:py:attr:`~pyxu.info.ptype.NDArray`)
146      --
147      (..., N) initial point(s).
148    * **variant** ("adam", "amsgrad", "padam")
149      --
150      Name of the ProxAdam variant to use.
151      Defaults to "adam".
152    * **a** (:py:attr:`~pyxu.info.ptype.Real`, :py:obj:`None`)
153      --
154      Max normalized gradient step size.
155      Defaults to :math:`1 / \beta` if unspecified.
156    * **b1** (:py:attr:`~pyxu.info.ptype.Real`)
157      --
158      1st-order gradient exponential decay :math:`\beta_{1} \in [0, 1)`.
159    * **b2** (:py:attr:`~pyxu.info.ptype.Real`)
160      --
161      2nd-order gradient exponential decay :math:`\beta_{2} \in [0, 1)`.
162    * **m0** (:py:attr:`~pyxu.info.ptype.NDArray`, :py:obj:`None`)
163      --
164      (..., N) initial 1st-order gradient estimate corresponding to each initial point.
165      Defaults to the null vector if unspecified.
166    * **v0** (:py:attr:`~pyxu.info.ptype.NDArray`, :py:obj:`None`)
167      --
168      (..., N) initial 2nd-order gradient estimate corresponding to each initial point.
169      Defaults to the null vector if unspecified.
170    * **kwargs_sub** (:py:class:`~collections.abc.Mapping`, :py:obj:`None`)
171      --
172      Keyword parameters used to initialize :py:meth:`~pyxu.opt.solver.pgd.PGD`'s ``__init__()`` in
173      sub-problems.
174      This is an advanced option: use it with care.
175    * **stop_crit_sub** (:py:class:`~pyxu.abc.solver.StoppingCriterion`, :py:obj:`None`)
176      --
177      Sub-problem stopping criterion.
178      Default: use same stopping criterion as main problem.
179    * **p** (:py:attr:`~pyxu.info.ptype.Real`)
180      --
181      PAdam power parameter :math:`p \in (0, 0.5]`.
182      Must be specified for PAdam, unused otherwise.
183    * **eps_adam** (:py:attr:`~pyxu.info.ptype.Real`)
184      --
185      Adam noise parameter :math:`\epsilon`.
186      This term is used exclusively if `variant="adam"`.
187      Defaults to 1e-6.
188    * **eps_var** (:py:attr:`~pyxu.info.ptype.Real`)
189      --
190      Avoids division by zero if estimated gradient variance is too small.
191      Defaults to 1e-6.
192    * **\*\*kwargs** (:py:class:`~collections.abc.Mapping`)
193      --
194      Other keyword parameters passed on to :py:meth:`pyxu.abc.solver.Solver.fit`.
195
196    Note
197    ----
198    If provided, `m0` and `v0` must be broadcastable with `x0`.
199
200    Example
201    -------
202    Consider the following optimization problem:
203
204    .. math::
205
206       \min_{\mathbf{x}\in\mathbb{R}^N} \Vert{\mathbf{x}-\mathbf{1}}\Vert_2^2 + \Vert{\mathbf{x}-\mathbf{1}}\Vert_1
207
208    .. code-block:: python3
209
210       import numpy as np
211
212       from pyxu.operator.func import L1Norm, SquaredL2Norm
213       from pyxu.opt.solver import ProxAdam
214
215       N = 3
216       f = SquaredL2Norm(dim=N).asloss(1)
217       g = L1Norm(dim=N).asloss(1)
218
219       prox_adam = ProxAdam(f, g)
220       prox_adam.fit(
221           x0=np.zeros((N,)),
222           variant="padam",
223           p=0.25,
224       )
225       x_opt = prox_adam.solution()
226       np.allclose(x_opt, 1)  # True
227    """
228
229    def __init__(
230        self,
231        f: pxa.DiffFunc,
232        g: pxa.ProxFunc = None,
233        **kwargs,
234    ):
235        kwargs.update(
236            log_var=kwargs.get("log_var", ("x",)),
237        )
238        super().__init__(**kwargs)
239
240        self._f = f
241        # If f is domain-agnostic and g is unspecified, cannot auto-infer NullFunc dimension.
242        # Solution: delay initialization of g to m_init(), where x0's shape can be used.
243        self._prox_required = g is not None
244        self._g = g
245
246    @pxrt.enforce_precision(i=("x0", "a", "b1", "b2", "m0", "v0", "p", "eps_adam", "eps_var"))
247    def m_init(  # default values from https://github.com/pmelchior/proxmin/blob/master/proxmin/algorithms.py
248        self,
249        x0: pxt.NDArray,
250        variant: str = "adam",
251        a: pxt.Real = None,
252        b1: pxt.Real = 0.9,
253        b2: pxt.Real = 0.999,
254        m0: pxt.NDArray = None,
255        v0: pxt.NDArray = None,
256        kwargs_sub: dict = None,
257        stop_crit_sub: pxa.solver.StoppingCriterion = None,
258        p: pxt.Real = 0.5,
259        eps_adam: pxt.Real = 1e-6,
260        eps_var: pxt.Real = 1e-6,
261    ):
262        mst = self._mstate  # shorthand
263        mst["x"] = x0
264
265        if self._g is None:
266            self._g = pxf.NullFunc(dim=x0.shape[-1])
267
268        if a is None:
269            mst["a"] = pxrt.coerce(1 / self._f.diff_lipschitz)
270            if math.isinf(mst["a"]):
271                # _f is constant-valued: a is a free parameter.
272                mst["a"] = pxrt.coerce(1)
273                msg = "\n".join(
274                    [
275                        rf"The gradient step size `a` is auto-set to {mst['a']}.",
276                        r"Choosing a manually may lead to faster convergence.",
277                    ]
278                )
279                warnings.warn(msg, pxw.AutoInferenceWarning)
280        else:
281            try:
282                assert a > 0
283                mst["a"] = a
284            except Exception:
285                raise ValueError(f"`a` must be positive, got {a}.")
286
287        mst["variant"] = self.__parse_variant(variant)
288
289        assert 0 < p <= 0.5, f"p: expected value in (0, 0.5], got {p}."
290        mst["padam_p"] = p
291
292        assert eps_adam > 0, f"eps_adam: expected positive value, got {eps_adam}."
293        mst["eps_adam"] = eps_adam
294
295        xp = pxu.get_array_module(x0)
296
297        if m0 is None:
298            mst["mean"] = xp.zeros_like(x0)
299        elif m0.shape == x0.shape:
300            # No broadcasting involved
301            mst["mean"] = m0
302        else:
303            x0, m0 = xp.broadcast_arrays(x0, m0)
304            mst["mean"] = m0.copy()
305
306        if v0 is None:
307            mst["variance"] = xp.zeros_like(x0)
308        elif v0.shape == x0.shape:
309            # No broadcasting involved
310            mst["variance"] = v0
311        else:
312            x0, v0 = xp.broadcast_arrays(x0, v0)
313            mst["variance"] = v0.copy()
314        mst["variance_hat"] = mst["variance"]
315
316        if kwargs_sub is None:
317            kwargs_sub = dict()
318        mst["subproblem_init_kwargs"] = kwargs_sub
319        if stop_crit_sub is None:
320            stop_crit_sub = self.default_stop_crit()
321        mst["subproblem_stop_crit"] = stop_crit_sub
322
323        assert 0 <= b1 < 1, f"b1: expected value in [0, 1), got {b1}."
324        mst["b1"] = b1
325
326        assert 0 <= b2 < 1, f"b2: expected value in [0, 1), got {b2}."
327        mst["b2"] = b2
328
329        assert eps_var > 0, f"eps_var: expected positive value, got {eps_var}."
330        mst["eps_variance"] = eps_var
331
332    def m_step(self):
333        mst = self._mstate  # shorthand
334
335        x, a = mst["x"], mst["a"]
336        xp = pxu.get_array_module(x)
337        gm = pxu.copy_if_unsafe(self._f.grad(x))
338        gv = gm.copy()
339
340        ## Part 1: evaluate phi/psi ============================
341        m, b1 = mst["mean"], mst["b1"]
342        # In-place implementation of -----------------
343        #   m = b1 * m + (1 - b1) * g
344        m *= b1
345        gm *= 1 - b1
346        m += gm
347        # --------------------------------------------
348        mst["mean"] = m
349
350        v, b2 = mst["variance"], mst["b2"]
351        # In-place implementation of -----------------
352        #   v = b2 * v + (1 - b2) * (g ** 2)
353        v *= b2
354        gv **= 2
355        gv *= 1 - b2
356        v += gv
357        # --------------------------------------------
358        mst["variance"] = v.clip(mst["eps_variance"], None)  # avoid division-by-zero
359        mst["variance_hat"] = xp.maximum(mst["variance_hat"], mst["variance"])
360
361        phi = self.__phi(t=self._astate["idx"])
362        psi = self.__psi(t=self._astate["idx"])
363        ## =====================================================
364
365        ## Part 2: take a step in the gradient's direction =====
366        # In-place implementation of -----------------
367        #   x = x - a * (phi / psi)
368        phi /= psi
369        phi *= a
370        x -= phi
371        ## =====================================================
372
373        ## Part 3: eval PGD sub-problems =======================
374        if self._prox_required:
375            # Assume N initial points were provided in fit().
376            # We need to solve a PGD sub-problem for each of them.
377            #
378            # Option 1: solve one large PGD(f, g) sub-problem, with
379            #    f = hstack(f1, ..., fN)
380            #    g = hstack(g, ..., g)
381            # Option 2: solve N small PGD(f, g) sub-problems, with
382            #    f = fn
383            #    g = g
384            #
385            # The former (1) is more convenient since all sub-problems are solved in parallel, but:
386            # * convergence will be slower since diff-Lipschitz constant of hstack(f1, ..., fN) is
387            #   larger than those of fn. (Moreover the diff-Lipschitz constants of the fn may be
388            #   spread out.)
389            # * in the event `stop_crit_sub` is provided, we cannot always intercept/adapt it to
390            #   apply block-wise on `_mstate`.
391            #
392            # Each sub-problem in the latter (2) converges fast, but sub-problems may execute in
393            # sequence depending on the array backend used.
394            #
395            # In light of the above, we opt for Option 2.
396            *x_sh, N = x.shape
397            x_sub = []  # sub-problem solutions
398            scale = pxrt.coerce(0.5 / a)
399            for _x, _psi in zip(x.reshape(-1, N), psi.reshape(-1, N)):
400                gamma = pxrt.coerce(a / xp.max(_psi))
401                h_half = pxl.DiagonalOp(xp.sqrt(_psi))
402
403                f1 = pxf.SquaredL2Norm(dim=N).asloss(h_half.apply(_x)) * h_half
404                f = f1 * scale
405
406                kwargs = mst["subproblem_init_kwargs"].copy()
407                kwargs.update(show_progress=False)
408                slvr = pxs.PGD(f=f, g=self._g, **kwargs)
409
410                slvr.fit(
411                    x0=_x,
412                    tau=gamma,
413                    stop_crit=mst["subproblem_stop_crit"],
414                )
415                x_sub.append(slvr.solution())
416            x = xp.stack(x_sub, axis=0).reshape(*x_sh, N)  # (..., N) original shape
417        ## =====================================================
418
419        mst["x"] = x

421    def default_stop_crit(self) -> pxa.StoppingCriterion:
422        from pyxu.opt.stop import RelError
423
424        # Described in [ProxAdam]_ and used in their implementation:
425        # https://github.com/pmelchior/proxmin/blob/master/proxmin/algorithms.py
426        rel_error = RelError(
427            eps=1e-4,
428            var="x",
429            f=None,
430            norm=2,
431            satisfy_all=True,
432        )
433        return rel_error

435    def objective_func(self) -> pxt.NDArray:
436        func = lambda x: self._f.apply(x) + self._g.apply(x)
437        y = func(self._mstate["x"])
438        return y

440    def solution(self) -> pxt.NDArray:
441        """
442        Returns
443        -------
444        x: NDArray
445            (..., N) solution.
446        """
447        data, _ = self.stats()
448        return data.get("x")

450    def __phi(self, t: int):
451        mst = self._mstate
452        var = mst["variant"]
453        if var == "adam":
454            out = mst["mean"].copy()
455            out /= 1 - (mst["b1"] ** t)
456        elif var in ["amsgrad", "padam"]:
457            out = mst["mean"].copy()  # to allow in-place updates outside __compute_phi()
458        return out
459
460    def __psi(self, t: int):
461        mst = self._mstate
462        xp = pxu.get_array_module(mst["x"])
463        var = mst["variant"]
464        if var == "adam":
465            out = xp.sqrt(mst["variance"])
466            out /= xp.sqrt(1 - mst["b2"] ** t)
467            out += mst["eps_adam"]
468        elif var == "amsgrad":
469            out = xp.sqrt(mst["variance_hat"])
470        elif var == "padam":
471            out = mst["variance_hat"] ** mst["padam_p"]
472        return out
473
474    def __parse_variant(self, variant: str) -> str:
475        supported_variants = {"adam", "amsgrad", "padam"}
476        if (v := variant.lower().strip()) not in supported_variants:
477            raise ValueError(f"Unsupported variant '{variant}'.")
478        return v
