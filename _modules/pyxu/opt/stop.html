 29class MaxIter(pxa.StoppingCriterion):
 30    """
 31    Stop iterative solver after a fixed number of iterations.
 32
 33    .. note::
 34
 35        If you want to add a grace period to a solver, i.e. for it to do *at least* N iterations before
 36        stopping based on the value of another criteria, you can AND :py:class:`~pyxu.opt.stop.MaxIter`
 37        with the other criteria.
 38
 39        .. code-block:: python3
 40
 41           sc = MaxIter(n=5) & AbsError(eps=0.1)
 42           # If N_iter < 5  -> never stop.
 43           # If N_iter >= 5 -> stop if AbsError() decides to.
 44    """
 45


[docs]
 46    def __init__(self, n: pxt.Integer):
 47        """
 48        Parameters
 49        ----------
 50        n: Integer
 51            Max number of iterations allowed.
 52        """
 53        try:
 54            assert int(n) > 0
 55            self._n = int(n)
 56        except Exception:
 57            raise ValueError(f"n: expected positive integer, got {n}.")
 58        self._i = 0


 59
 60    def stop(self, state: cabc.Mapping) -> bool:
 61        self._i += 1
 62        return self._i > self._n
 63
 64    def info(self) -> cabc.Mapping[str, float]:
 65        return dict(N_iter=self._i)
 66
 67    def clear(self):
 68        self._i = 0

 71class ManualStop(pxa.StoppingCriterion):

 98    def __init__(self, t: dt.timedelta):
 99        """
100        Parameters
101        ----------
102        t: ~datetime.timedelta
103            Max runtime allowed.
104        """
105        try:
106            assert t > dt.timedelta()
107            self._t_max = t
108        except Exception:
109            raise ValueError(f"t: expected positive duration, got {t}.")
110        self._t_start = dt.datetime.now()
111        self._t_now = self._t_start


112
113    def stop(self, state: cabc.Mapping) -> bool:
114        self._t_now = dt.datetime.now()
115        return (self._t_now - self._t_start) > self._t_max
116
117    def info(self) -> cabc.Mapping[str, float]:
118        d = (self._t_now - self._t_start).total_seconds()
119        return dict(duration=d)
120
121    def clear(self):
122        self._t_start = dt.datetime.now()
123        self._t_now = self._t_start

126class MaxCarbon(pxa.StoppingCriterion):
127    """
128    Stop iterative solver after a threshold amount of carbon dioxide (CO2) was produced by computing
129    resources used to execute the optimization.
130
131    .. note::
132
133       Codecarbon is not a core dependency of Pyxu and must be installed separately.
134
135       On Windows and MacOS, `codecarbon <https://pypi.org/project/codecarbon/>`_ tracks power consumption of
136       Intel processors using the
137       `Intel Power Gadget (IPG) <https://www.intel.com/content/www/us/en/developer/articles/tool/power-gadget.html>`_.
138       IPG must be installed independently.
139       (MacOS-specific: IPG must also have correct security permissions.)
140    """
141


[docs]
142    def __init__(self, co2: pxt.Real):
143        """
144        Parameters
145        ----------
146        co2: Real
147            Max allowed CO2 emissions [Kg].
148        """
149        try:
150            assert float(co2) > 0
151            self._co2_max = float(co2)
152        except Exception:
153            raise ValueError(f"co2: expected positive carbon quantity, got {co2}.")
154
155        codecarbon = pxu.import_module("codecarbon")
156        self.tracker = codecarbon.EmissionsTracker(
157            api_call_interval=-1,
158            save_to_file=False,
159            log_level="warning",
160        )
161        self.tracker.start()
162        self._co2_start = self.tracker.flush()
163        self._co2_now = self._co2_start


164
165    def stop(self, state: cabc.Mapping) -> bool:
166        self._co2_now = self.tracker.flush()
167        decision = (self._co2_now - self._co2_start) > self._co2_max
168        if decision:
169            self.tracker.stop()
170        return decision
171
172    def info(self) -> cabc.Mapping[str, float]:
173        return dict(co2=self._co2_now)
174
175    def clear(self):
176        self.tracker.start()
177        self._co2_start = self.tracker.flush()
178        self._co2_now = self._co2_start

189        """
190        Parameters

222class AbsError(pxa.StoppingCriterion):
223    """
224    Stop iterative solver after absolute norm of a variable (or function thereof) reaches threshold.
225    """
226


[docs]
227    def __init__(
228        self,
229        eps: pxt.Real,
230        var: pxt.VarName = "x",
231        f: SVFunction = None,
232        norm: pxt.Real = 2,
233        satisfy_all: bool = True,
234    ):
235        """
236        Parameters
237        ----------
238        eps: Real
239            Positive threshold.
240        var: VarName
241            Variable in :py:attr:`pyxu.abc.solver.Solver._mstate` to query.
242        f: ~collections.abc.Callable
243            Optional function to pre-apply to ``_mstate[var]`` before applying the norm.
244            Defaults to the identity function. The callable should either:
245
246            * accept a scalar input -> output a scalar, or
247            * accept an NDArray input -> output an NDArray, i.e same semantics as
248              :py:meth:`~pyxu.abc.operator.Map.apply`.
249        norm: Integer, Real
250            Ln norm to use >= 0. (Default: L2.)
251        satisfy_all: bool
252            If True (default) and ``_mstate[var]`` is multi-dimensional, stop if all evaluation
253            points lie below threshold.
254        """
255        try:
256            assert eps > 0
257            self._eps = eps
258        except Exception:
259            raise ValueError(f"eps: expected positive threshold, got {eps}.")
260
261        self._var = var
262        self._f = f if (f is not None) else (lambda _: _)
263
264        try:
265            assert norm >= 0
266            self._norm = norm
267        except Exception:
268            raise ValueError(f"norm: expected non-negative, got {norm}.")
269
270        self._satisfy_all = satisfy_all
271        self._val = np.r_[0]  # last computed Ln norm(s) in stop().


272
273    def stop(self, state: cabc.Mapping) -> bool:
274        fx = self._f(state[self._var])
275        if isinstance(fx, pxt.Real):
276            fx = np.r_[fx]
277        xp = pxu.get_array_module(fx)
278
279        self._val = xp.linalg.norm(fx, ord=self._norm, axis=-1, keepdims=True)
280        rule = xp.all if self._satisfy_all else xp.any
281        decision = rule(self._val <= self._eps)
282
283        self._val, decision = pxu.compute(self._val, decision)
284        return decision
285
286    def info(self) -> cabc.Mapping[str, float]:
287        if self._val.size == 1:
288            data = {f"AbsError[{self._var}]": float(self._val.max())}  # takes the only element available.
289        else:
290            data = {
291                f"AbsError[{self._var}]_min": float(self._val.min()),
292                f"AbsError[{self._var}]_max": float(self._val.max()),
293            }
294        return data
295
296    def clear(self):
297        self._val = np.r_[0]

300class RelError(pxa.StoppingCriterion):
301    """
302    Stop iterative solver after relative norm change of a variable (or function thereof) reaches
303    threshold.
304    """
305


[docs]
306    def __init__(
307        self,
308        eps: pxt.Real,
309        var: pxt.VarName = "x",
310        f: SVFunction = None,
311        norm: pxt.Real = 2,
312        satisfy_all: bool = True,
313    ):
314        """
315        Parameters
316        ----------
317        eps: Real
318            Positive threshold.
319        var: VarName
320            Variable in :py:attr:`pyxu.abc.solver.Solver._mstate` to query.
321        f: ~collections.abc.Callable
322            Optional function to pre-apply to ``_mstate[var]`` before applying the norm.
323            Defaults to the identity function. The callable should either:
324
325            * accept a scalar input -> output a scalar, or
326            * accept an NDArray input -> output an NDArray, i.e same semantics as
327              :py:meth:`~pyxu.abc.operator.Map.apply`.
328        norm: Integer, Real
329            Ln norm to use >= 0. (Default: L2.)
330        satisfy_all: bool
331            If True (default) and ``_mstate[var]`` is multi-dimensional, stop if all evaluation
332            points lie below threshold.
333        """
334        try:
335            assert eps > 0
336            self._eps = eps
337        except Exception:
338            raise ValueError(f"eps: expected positive threshold, got {eps}.")
339
340        self._var = var
341        self._f = f if (f is not None) else (lambda _: _)
342
343        try:
344            assert norm >= 0
345            self._norm = norm
346        except Exception:
347            raise ValueError(f"norm: expected non-negative, got {norm}.")
348
349        self._satisfy_all = satisfy_all
350        self._val = np.r_[0]  # last computed Ln rel-norm(s) in stop().
351        self._x_prev = None  # buffered var from last query.


352
353    def stop(self, state: cabc.Mapping) -> bool:
354        x = state[self._var]
355        if isinstance(x, pxt.Real):
356            x = np.r_[x]
357        xp = pxu.get_array_module(x)
358
359        if self._x_prev is None:
360            self._x_prev = x.copy()
361            # force 1st .info() call to have same format as further calls.
362            self._val = np.zeros(shape=(1,) if (x.ndim == 1) else x.shape[:-1])
363            return False  # decision deferred: insufficient history to evaluate rel-err.
364        else:
365            norm = lambda _: xp.linalg.norm(_, ord=self._norm, axis=-1, keepdims=True)
366            rule = xp.all if self._satisfy_all else xp.any
367
368            fx_prev = self._f(self._x_prev)
369            numerator = norm(self._f(x) - fx_prev)
370            denominator = norm(fx_prev)
371            decision = rule(numerator <= self._eps * denominator)
372
373            with warnings.catch_warnings():
374                # Store relative improvement values for info(). Special care must be taken for the
375                # problematic case 0/0 -> NaN.
376                warnings.simplefilter("ignore")
377                self._val = numerator / denominator
378                self._val[xp.isnan(self._val)] = 0  # no relative improvement.
379            self._x_prev = x.copy()
380
381            self._x_prev, self._val, decision = pxu.compute(self._x_prev, self._val, decision)
382            return decision
383
384    def info(self) -> cabc.Mapping[str, float]:
385        if self._val.size == 1:
386            data = {f"RelError[{self._var}]": float(self._val.max())}  # takes the only element available.
387        else:
388            data = {
389                f"RelError[{self._var}]_min": float(self._val.min()),
390                f"RelError[{self._var}]_max": float(self._val.max()),
391            }
392        return data
393
394    def clear(self):
395        self._val = np.r_[0]
396        self._x_prev = None
