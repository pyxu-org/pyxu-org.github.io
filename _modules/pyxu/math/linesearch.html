19def backtracking_linesearch(
20    f: pxa.DiffFunc,
21    x: pxt.NDArray,
22    direction: pxt.NDArray,
23    gradient: pxt.NDArray = None,
24    a0: pxt.Real = None,
25    r: pxt.Real = LINESEARCH_DEFAULT_R,
26    c: pxt.Real = LINESEARCH_DEFAULT_C,
27) -> pxt.NDArray:
28    r"""
29    Backtracking line search algorithm based on the
30    `Armijo-Goldstein condition <https://www.wikiwand.com/en/Backtracking_line_search>`_.
31
32    Parameters
33    ----------
34    f: ~pyxu.abc.operator.DiffFunc
35        Differentiable functional.
36    x: NDArray
37        (..., N) initial search point(s).
38    direction: NDArray
39        (..., N) search direction(s) corresponding to initial point(s).
40    gradient: NDArray
41        (..., N) gradient of `f` at initial search point(s).
42
43        Specifying `gradient` when known is an optimization:
44        it will be autocomputed via :py:meth:`~pyxu.abc.operator.DiffFunc.grad` if unspecified.
45    a0: Real
46        Initial step size.
47
48        If unspecified and :math:`\nabla f` is :math:`\beta`-Lipschitz continuous, then `a0` is
49        auto-chosen as :math:`\frac{1}{\beta}`.
50    r: Real
51        Step reduction factor.
52    c: Real
53        Bound reduction factor.
54
55    Returns
56    -------
57    a: NDArray
58        (..., 1) step sizes.
59    """
60    assert 0 < r < 1
61    assert 0 < c < 1
62    if a0 is None:
63        a0 = pxrt.coerce(1 / f.diff_lipschitz)
64        assert a0 > 0, "a0: cannot auto-set step size."
65    else:
66        assert a0 > 0
67
68    if gradient is None:
69        gradient = f.grad(x)
70
71    f_x = f.apply(x)
72    d_f = c * (gradient * direction).sum(axis=-1, keepdims=True)  # \delta f
73
74    def refine(a: pxt.NDArray) -> pxt.NDArray:
75        # Do one iteration of the algorithm.
76        #
77        # Parameters
78        # ----------
79        # a : NDArray
80        #     (..., 1) current step size(s)
81        #
82        # Returns
83        # -------
84        # mask : NDArray[bool]
85        #     (..., 1) refinement points
86        lhs = f.apply(x + a * direction)
87        rhs = f_x + a * d_f
88        return lhs > rhs  # mask
89
90    xp = pxu.get_array_module(x)
91    a = xp.full(shape=(*x.shape[:-1], 1), fill_value=a0, dtype=x.dtype)
92    while (mask := refine(a)).any():
93        a[mask] *= r
94    return a
