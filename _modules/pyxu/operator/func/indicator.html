 26class _IndicatorFunction(pxf._ShiftLossMixin, pxa.ProxFunc):
 27    def __init__(self, dim: pxt.Integer):
 28        super().__init__(shape=(1, dim))
 29        self.lipschitz = np.inf
 30
 31    @staticmethod
 32    def _bool2indicator(x: pxt.NDArray, dtype: pxt.DType) -> pxt.NDArray:
 33        # x: NDarray[bool]
 34        # y: NDarray[(0, \inf), dtype]
 35        xp = pxu.get_array_module(x)
 36        cast = lambda _: np.array(_, dtype=dtype)[()]
 37        y = xp.where(x, cast(0), cast(np.inf))
 38        return y

 41class _NormBall(_IndicatorFunction):
 42    def __init__(
 43        self,
 44        dim: pxt.Integer,
 45        ord: pxt.Integer,
 46        radius: pxt.Real,
 47    ):
 48        super().__init__(dim=dim)
 49        self._ord = ord
 50        self._radius = float(radius)
 51
 52    @pxrt.enforce_precision(i="arr")
 53    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 54        norm = pxlg.norm(arr, ord=self._ord, axis=-1, keepdims=True)
 55        out = self._bool2indicator(norm <= self._radius, arr.dtype)
 56        return out
 57
 58    @pxrt.enforce_precision(i=("arr", "tau"))
 59    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
 60        klass = {  # class of proximal operator to use
 61            1: pxf.LInfinityNorm,
 62            2: pxf.L2Norm,
 63            np.inf: pxf.L1Norm,
 64        }[self._ord]
 65        op = klass(dim=self.dim)
 66
 67        out = arr.copy()
 68        out -= op.prox(arr, tau=self._radius)
 69        return out

 72def L1Ball(dim: pxt.Integer, radius: pxt.Real = 1) -> pxt.OpT:
 73    r"""
 74    Indicator function of the :math:`\ell_{1}`-ball.
 75
 76    .. math::
 77
 78       \iota_{1}^{r}(\mathbf{x})
 79       :=
 80       \begin{cases}
 81           0 & \|\mathbf{x}\|_{1} \le r \\
 82           \infty & \text{otherwise}.
 83       \end{cases}
 84
 85    .. math::
 86
 87       \text{prox}_{\tau\, \iota_{1}^{r}}(\mathbf{x})
 88       :=
 89       \mathbf{x} - \text{prox}_{r\, \ell_{\infty}}(\mathbf{x})
 90
 91    Parameters
 92    ----------
 93    dim: Integer
 94    radius: Real
 95        Ball radius. (Default: unit ball.)
 96
 97    Returns
 98    -------
 99    op: OpT
100    """
101    op = _NormBall(dim=dim, ord=1, radius=radius)
102    op._name = "L1Ball"
103    return op

174class PositiveOrthant(_IndicatorFunction):
175    r"""
176    Indicator function of the positive orthant.
177
178    .. math::
179
180       \iota_{+}(\mathbf{x})
181       :=
182       \begin{cases}
183           0 & \min{\mathbf{x}} \ge 0,\\
184           \infty & \text{otherwise}.
185       \end{cases}
186
187    .. math::
188
189       \text{prox}_{\tau\, \iota_{+}}(\mathbf{x})
190       :=
191       \max(\mathbf{x}, \mathbf{0})
192    """
193
194    def __init__(self, dim: pxt.Integer):
195        super().__init__(dim=dim)
196
197    @pxrt.enforce_precision(i="arr")
198    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
199        in_set = (arr >= 0).all(axis=-1, keepdims=True)
200        out = self._bool2indicator(in_set, arr.dtype)
201        return out
202
203    @pxrt.enforce_precision(i=("arr", "tau"))
204    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
205        out = arr.clip(0, None)
206        return out

209class HyperSlab(_IndicatorFunction):
210    r"""
211    Indicator function of a hyperslab.
212
213    .. math::
214
215       \iota_{\mathbf{a}}^{l,u}(\mathbf{x})
216       :=
217       \begin{cases}
218           0 & l \le \langle \mathbf{a}, \mathbf{x} \rangle \le u \\
219           \infty & \text{otherwise}.
220       \end{cases}
221
222    .. math::
223
224       \text{prox}_{\tau\, \iota_{\mathbf{a}}^{l,u}}(\mathbf{x})
225       :=
226       \begin{cases}
227           \mathbf{x} + \frac{l - \langle \mathbf{a}, \mathbf{x} \rangle}{\|\mathbf{a}\|^{2}} \mathbf{a} & \langle \mathbf{a}, \mathbf{x} \rangle < l, \\
228           \mathbf{x} + \frac{u - \langle \mathbf{a}, \mathbf{x} \rangle}{\|\mathbf{a}\|^{2}} \mathbf{a} & \langle \mathbf{a}, \mathbf{x} \rangle > u, \\
229           \mathbf{x} & \text{otherwise}.
230       \end{cases}
231    """
232


[docs]
233    @pxrt.enforce_precision(i=("lb", "ub"))
234    def __init__(self, a: pxa.LinFunc, lb: pxt.Real, ub: pxt.Real):
235        """
236        Parameters
237        ----------
238        A: ~pyxu.abc.operator.LinFunc
239            (N,) operator
240        lb: Real
241            Lower bound
242        ub: Real
243            Upper bound
244        """
245        assert lb < ub
246        super().__init__(dim=a.dim)
247
248        # Everything happens internally in normalized coordinates.
249        _norm = a.lipschitz  # \norm{a}{2}
250        self._a = a / _norm
251        self._l = lb / _norm
252        self._u = ub / _norm


253
254    @pxrt.enforce_precision(i="arr")
255    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
256        y = self._a.apply(arr)
257        in_set = ((self._l <= y) & (y <= self._u)).all(axis=-1, keepdims=True)
258        out = self._bool2indicator(in_set, arr.dtype)
259        return out
260
261    @pxrt.enforce_precision(i=("arr", "tau"))
262    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
263        xp = pxu.get_array_module(arr)
264
265        a = self._a.adjoint(xp.ones(1, dtype=arr.dtype))  # slab direction
266        y = self._a.apply(arr)
267        out = arr.copy()
268
269        l_corr = self._l - y
270        l_corr[l_corr <= 0] = 0
271        out += l_corr * a
272
273        u_corr = self._u - y
274        u_corr[u_corr >= 0] = 0
275        out += u_corr * a
276
277        return out

280class RangeSet(_IndicatorFunction):
281    r"""
282    Indicator function of a range set.
283
284    .. math::
285
286       \iota_{\mathbf{A}}^{R}(\mathbf{x})
287       :=
288       \begin{cases}
289           0 & \mathbf{x} \in \text{span}(\mathbf{A}) \\
290           \infty & \text{otherwise}.
291       \end{cases}
292
293    .. math::
294
295       \text{prox}_{\tau\, \iota_{\mathbf{A}}^{R}}(\mathbf{x})
296       :=
297       \mathbf{A} (\mathbf{A}^{T} \mathbf{A})^{-1} \mathbf{A}^{T} \mathbf{x}.
298    """

300    def __init__(self, A: pxa.LinOp):
301        """
302        Parameters
303        ----------
304        A: ~pyxu.abc.operator.LinOp
305            (M, N) operator
306        """
307        super().__init__(dim=A.codim)
308        self._A = A


309
310    @pxrt.enforce_precision(i="arr")
311    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
312        # I'm in range(A) if prox(x)==x.
313        xp = pxu.get_array_module(arr)
314        in_set = xp.isclose(self.prox(arr, tau=1), arr).all(axis=-1, keepdims=True)
315        out = self._bool2indicator(in_set, arr.dtype)
316        return out
317
318    @pxrt.enforce_precision(i=("arr", "tau"))
319    @pxu.vectorize(i="arr")  # see comment below
320    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
321        # [2023.01.03 Sepand]
322        #
323        # When more than one input is provided, `A.pinv(arr)` may sometimes return NaNs.
324        # The problem is pinpointed to the instruction below from CG():
325        #     alpha = rr / (p * Ap).sum(axis=-1, keepdims=True)
326        #
327        # Oddly the problem does not occur when `arr` is 1D.
328        # Could not figure out why the CG line breaks down at times with multi-inputs.
329        #
330        # Temporary(/Permanent?) workaround: use @vectorize() to evaluate prox calls one at a time.
331        y = self._A.pinv(arr, damp=0)
332        out = self._A.apply(y)
333        return out

336class AffineSet(_IndicatorFunction):
337    r"""
338    Indicator function of an affine set.
339
340    .. math::
341
342       \iota_{\mathbf{A}}^{\mathbf{b}}(\mathbf{x})
343       :=
344       \begin{cases}
345           0 & \mathbf{A} \mathbf{x} = \mathbf{b} \\
346           \infty & \text{otherwise}.
347       \end{cases}
348
349    .. math::
350
351       \text{prox}_{\tau\, \iota_{\mathbf{A}}^{\mathbf{b}}}(\mathbf{x})
352       :=
353       \mathbf{x} - \mathbf{A}^{T} (\mathbf{A}\mathbf{A}^{T})^{-1} (\mathbf{Ax - b})
354
355    Notes
356    -----
357    * Assumptions on :math:`\mathbf{A}` and :math:`\mathbf{b}`:
358
359      * :math:`\mathbf{b} \in \text{span}(\mathbf{A})`.
360      * :math:`\mathbf{A}` has full row-rank, i.e. :math:`\mathbf{A}` is square or fat.
361
362    * :py:class:`~pyxu.operator.func.indicator.AffineSet` instances are **not arraymodule-agnostic**:
363      they will only work with NDArrays belonging to the same array module as `A` and `b`.
364    """
365

366    @pxrt.enforce_precision(i="b")
367    def __init__(self, A: pxa.LinOp, b: pxt.NDArray):
368        """
369        Parameters
370        ----------
371        A: ~pyxu.abc.operator.LinOp
372            (M, N) operator
373        b: NDArray
374            (M,)
375        """
376        assert A.codim <= A.dim, f"`A` must have full row-rank, but A.shape = {A.shape}."
377        assert (b.ndim == 1) and (b.size == A.codim)
378        super().__init__(dim=A.dim)
379
380        # Some approximate solution to A x = b, i.e x0.
381        # x0 must be a good estimate given assumptions on (A, b).
382        self._x0 = A.pinv(b, damp=0)
383        self._b = A.apply(self._x0)  # fuzz `b` slightly to avoid numerical issues in .apply()
384        self._A = A


385
386    @pxrt.enforce_precision(i="arr")
387    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
388        xp = pxu.get_array_module(arr)
389        in_set = xp.isclose(self._A.apply(arr), self._b).all(axis=-1, keepdims=True)
390        out = self._bool2indicator(in_set, arr.dtype)
391        return out
392
393    @pxrt.enforce_precision(i=("arr", "tau"))
394    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
395        op = RangeSet(self._A.T)
396
397        out = arr.copy()
398        out -= op.prox(arr - self._x0, tau=1)  # tau arbitrary
399        return out

402class ConvexSetIntersection(_IndicatorFunction):
403    r"""
404    Indicator function of an intersection of convex domains
405    :math:`\mathcal{C}_{1} \cap \cdots \cap \mathcal{C}_{K}`.
406
407    .. math::
408
409       \iota(\mathbf{x})
410       :=
411       \iota_{1}(\mathbf{x}) + \cdots + \iota_{K}(\mathbf{x}).
412
413    :math:`\text{prox}_{\tau\, \iota}(\mathbf{x})` is computed using the PoCS algorithm.
414    (Dykstra's variant [PoCS_Dykstra]_.)
415
416    This function assumes :math:`\mathcal{C}_{1} \cap \cdots \cap \mathcal{C}_{K} \ne \emptyset`.
417    :py:meth:`~pyxu.operator.func.indicator.ConvexSetIntersection.prox` will loop indefinitely if
418    this condition is violated.
419
420    Examples
421    --------
422    .. code-block:: python3
423
424       import numpy as np
425       import pyxu.operator as pxo
426
427       op = pxo.ConvexSetIntersection(             # intersection of
428           pxo.LInfinityBall(),                    #   L-\infty ball centered at (0,0)
429           pxo.L2Ball().asloss(np.array([1, 0])),  #   L2 ball centered at (1,0)
430       )
431
432       x = np.array([
433           [0, 0],
434           [1, 0],
435           [2, 0],  # not in LInfinity ball -> \infty
436           [1, 1],
437           [1, -1],
438           [0, 1],  # not in L2 ball -> \infty
439        ])
440       op.apply(x)  # [0 0 inf 0 0 inf]
441    """
442

443    def __init__(self, *args: cabc.Sequence[pxa.ProxFunc]):
444        """
445        Parameters
446        ----------
447        args: :py:class:`list` ( :py:class:`~pyxu.abc.operator.ProxFunc` )
448            Sequence of indicator functions encoding convex domains.
449        """
450        # Create `op` to auto-compute best shape behind the scenes.
451        if len(args) == 1:
452            op = args[0]
453        else:
454            op = functools.reduce(operator.add, args)
455        super().__init__(dim=op.dim)
456
457        self._f = args
458
459        # Initialize PoCS solver.
460        # Useful to do so in __init__() to be able to examine solver stats() after prox() calls.
461        self._slvr = ConvexSetIntersection._PoCS(
462            f=self._f,
463            show_progress=False,
464            log_var="x",
465        )


466
467    @pxrt.enforce_precision(i="arr")
468    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
469        out = [None] * len(self._f)
470        for i, f in enumerate(self._f):
471            out[i] = f.apply(arr)
472        out = sum(out)
473        return out

475    @pxrt.enforce_precision(i=("arr", "tau"))
476    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
477        if len(self._f) == 1:
478            # One projection sufficient
479            op = self._f[0]
480            out = op.prox(arr, tau)  # tau can be arbitrary
481        else:
482            # Use Dykstra's PoCS algorithm until a point in the intersection is found
483            from pyxu.opt.stop import AbsError
484
485            self._slvr.fit(
486                x0=arr,
487                stop_crit=AbsError(
488                    var="x",
489                    eps=1,  # any point in intersection satisfies (eps < inf)
490                    f=lambda x: self.apply(x),
491                ),
492            )
493
494            data, _ = self._slvr.stats()
495            out = data["x"]
496        return out

502    class _PoCS(pxa.Solver):
503        # Dykstra's PoCS algorithm [PoCS_Dykstra]_
504        #
505        # This is a bare-bones Solver sub-class since only used internally.
506
507        def __init__(self, f: cabc.Sequence[pxa.ProxFunc], **kwargs):
508            super().__init__(**kwargs)
509            self._f = f
510
511        @pxrt.enforce_precision(i="x0")
512        def m_init(self, x0: pxt.NDArray):
513            mst = self._mstate  # shorthand
514            mst["x"] = x0.copy()
515
516            xp = pxu.get_array_module(x0)
517            K = len(self._f)
518            *sh, N = x0.shape
519            mst["residual"] = xp.zeros((K, *sh, N), dtype=x0.dtype)
520
521        def m_step(self):
522            mst = self._mstate  # shorthand
523            x, r = mst["x"], mst["residual"]
524
525            for i, f in enumerate(self._f):
526                y = f.prox(x - r[i], tau=1)  # tau is arbitrary
527                r[i] = y - x
528                x[:] = y
