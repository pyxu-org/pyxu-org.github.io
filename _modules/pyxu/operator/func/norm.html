 33class L1Norm(_ShiftLossMixin, pxa.ProxFunc):
 34    r"""
 35    :math:`\ell_{1}`-norm, :math:`\Vert\mathbf{x}\Vert_{1} := \sum_{i=1}^{N} |x_{i}|`.
 36    """
 37
 38    def __init__(self, dim: pxt.Integer):
 39        super().__init__(shape=(1, dim))
 40        self.lipschitz = np.sqrt(dim)
 41
 42    @pxrt.enforce_precision(i="arr")
 43    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 44        y = pxlg.norm(arr, ord=1, axis=-1, keepdims=True)
 45        return y
 46
 47    @pxrt.enforce_precision(i=("arr", "tau"))
 48    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
 49        xp = pxu.get_array_module(arr)
 50        y = xp.fmax(0, xp.fabs(arr) - tau)
 51        y *= xp.sign(arr)
 52        return y

 55class L2Norm(_ShiftLossMixin, pxa.ProxFunc):
 56    r"""
 57    :math:`\ell_{2}`-norm, :math:`\Vert\mathbf{x}\Vert_{2} := \sqrt{\sum_{i=1}^{N} |x_{i}|^{2}}`.
 58    """
 59
 60    def __init__(self, dim: pxt.Integer):
 61        super().__init__(shape=(1, dim))
 62        self.lipschitz = 1
 63        self.diff_lipschitz = np.inf
 64
 65    @pxrt.enforce_precision(i="arr")
 66    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 67        y = pxlg.norm(arr, ord=2, axis=-1, keepdims=True)
 68        return y
 69
 70    @pxrt.enforce_precision(i=("arr", "tau"))
 71    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
 72        xp = pxu.get_array_module(arr)
 73        scale = 1 - tau / xp.fmax(self.apply(arr), tau)
 74
 75        y = arr.copy()
 76        y *= scale.astype(dtype=arr.dtype)
 77        return y

 80class SquaredL2Norm(pxa.QuadraticFunc):
 81    r"""
 82    :math:`\ell^{2}_{2}`-norm, :math:`\Vert\mathbf{x}\Vert^{2}_{2} := \sum_{i=1}^{N} |x_{i}|^{2}`.
 83    """
 84
 85    def __init__(self, dim: pxt.Integer):
 86        super().__init__(shape=(1, dim))
 87        self.lipschitz = np.inf
 88        self.diff_lipschitz = 2
 89
 90    @pxrt.enforce_precision(i="arr")
 91    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 92        y = pxlg.norm(arr, axis=-1, keepdims=True)
 93        y **= 2
 94        return y
 95
 96    @pxrt.enforce_precision(i="arr")
 97    def grad(self, arr: pxt.NDArray) -> pxt.NDArray:
 98        return 2 * arr
 99
100    @pxrt.enforce_precision(i=("arr", "tau"))
101    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
102        y = arr.copy()
103        y /= 2 * tau + 1
104        return y

106    def _quad_spec(self):
107        from pyxu.operator.linop import HomothetyOp, NullFunc
108
109        Q = HomothetyOp(dim=self.dim, cst=2)
110        c = NullFunc(dim=self.dim)
111        t = 0
112        return (Q, c, t)

115class SquaredL1Norm(_ShiftLossMixin, pxa.ProxFunc):
116    r"""
117    :math:`\ell^{2}_{1}`-norm, :math:`\Vert\mathbf{x}\Vert^{2}_{1} := (\sum_{i=1}^{N} |x_{i}|)^{2}`.
118    """
119


[docs]
120    def __init__(self, dim: pxt.Integer, prox_algo: str = "sort"):
121        r"""
122        Parameters
123        ----------
124        dim: Integer
125        prox_algo: "root", "sort"
126            Algorithm used for computing the proximal operator:
127
128            * 'root' uses [FirstOrd]_ Lemma 6.70,
129            * 'sort' uses [OnKerLearn]_ Algorithm 2 (faster).
130
131        Notes
132        -----
133        Calling :py:meth:`~pyxu.operator.func.norm.SquaredL1Norm.prox` with DASK inputs when
134        `algo="sort"` is inefficient at scale.
135        Prefer `algo="root"` in this case.
136        """
137        super().__init__(shape=(1, dim))
138        self.lipschitz = np.inf
139
140        algo = prox_algo.strip().lower()
141        assert algo in ("root", "sort")
142        self._algo = algo


143
144    @pxrt.enforce_precision(i="arr")
145    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
146        y = pxlg.norm(arr, ord=1, axis=-1, keepdims=True)
147        y **= 2
148        return y

150    def _prox_root(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
151        xp = pxu.get_array_module(arr)
152        norm = xp.linalg.norm(arr, axis=-1)
153        if not (norm > 0):
154            out = arr
155        else:
156            # Part 1: Compute \mu_opt -----------------------------------------
157            def f(mu: pxt.Real) -> pxt.Real:
158                # Proxy function to compute \mu_opt
159                #
160                # Inplace implementation of
161                #     f(mu) = clip(|arr| * sqrt(tau / mu) - 2 * tau, 0, None).sum() - 1
162                x = xp.fabs(arr)
163                x *= xp.sqrt(tau / mu)
164                x -= 2 * tau
165                xp.clip(x, 0, None, out=x)
166                y = x.sum()
167                y -= 1
168                return y
169
170            mu_opt, res = sopt.brentq(
171                f=f,
172                a=1e-12,
173                b=xp.fabs(arr).max() ** 2 / (4 * tau),
174                full_output=True,
175                disp=False,
176            )
177            if not res.converged:
178                msg = "Computing mu_opt did not converge."
179                raise ValueError(msg)
180
181            # Part 2: Compute \lambda -----------------------------------------
182            # Inplace implementation of
183            #     lambda_ = clip(|arr| * sqrt(tau / mu_opt) - 2 * tau, 0, None)
184            lambda_ = xp.fabs(arr)
185            lambda_ *= xp.sqrt(tau / mu_opt)
186            lambda_ -= 2 * tau
187            xp.clip(lambda_, 0, None, out=lambda_)
188
189            # Part 3: Compute \prox -------------------------------------------
190            # Inplace implementation of
191            #     out = arr * lambda_ / (lambda_ + 2 * tau)
192            out = arr.copy()
193            out *= lambda_
194            lambda_ += 2 * tau
195            out /= lambda_
196        return out
197
198    def _prox_sort(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
199        xp = pxu.get_array_module(arr)
200
201        # Inplace implementation of
202        #     y = sort(|arr|, axis=-1)[::-1]
203        y = xp.fabs(arr)
204        y.sort(axis=-1)
205        y = y[::-1]
206
207        z = y.cumsum(axis=-1)
208        z *= tau / (0.5 + tau * xp.arange(1, arr.size + 1, dtype=z.dtype))
209        tau2 = z[max(xp.flatnonzero(y > z), default=0)]
210
211        out = L1Norm(dim=self.dim).prox(arr, tau2)
212        return out

214    @pxrt.enforce_precision(i=("arr", "tau"))
215    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
216        N = pxd.NDArrayInfo
217        if (N.from_obj(arr) == N.DASK) and (self._algo == "sort"):
218            msg = "\n".join(
219                [
220                    "Using prox_algo='sort' on DASK inputs is inefficient.",
221                    "Consider using prox_algo='root' instead.",
222                ]
223            )
224            warnings.warn(msg, pxw.PerformanceWarning)
225
226        vectorize = pxu.vectorize(
227            # DASK backend required since prox_[root|sort]() don't accept DASK inputs.
228            i="arr",
229            method="parallel",
230            codim=arr.shape[-1],
231        )
232        f = dict(
233            root=vectorize(self._prox_root),
234            sort=vectorize(self._prox_sort),
235        )[self._algo]
236
237        out = f(arr, tau)
238        return out

241class LInfinityNorm(_ShiftLossMixin, pxa.ProxFunc):
242    r"""
243    :math:`\ell_{\infty}`-norm, :math:`\Vert\mathbf{x}\Vert_{\infty} := \max_{i=1,\ldots,N} |x_{i}|`.
244    """
245
246    def __init__(self, dim: pxt.Integer):
247        super().__init__(shape=(1, dim))
248        self.lipschitz = 1
249
250    @pxrt.enforce_precision(i="arr")
251    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
252        y = pxlg.norm(arr, ord=np.inf, axis=-1, keepdims=True)
253        return y
254
255    @pxrt.enforce_precision(i=("arr", "tau"))
256    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
257        vectorize = pxu.vectorize(
258            i="arr",
259            method="parallel",
260            codim=arr.shape[-1],
261        )
262        f = vectorize(self._prox)
263
264        out = f(arr, tau)
265        return out
266
267    def _prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
268        xp = pxu.get_array_module(arr)
269        y = xp.zeros(arr.shape, dtype=arr.dtype)
270
271        def f(mu: pxt.Real) -> pxt.Real:
272            # Proxy function to compute \mu_opt
273            #
274            # Inplace implementation of
275            #     f(mu) = clip(xp.fabs(arr) - mu, 0, None).sum() - tau
276            x = xp.fabs(arr)
277            x -= mu
278            xp.clip(x, 0, None, out=x)
279            y = x.sum()
280            y -= tau
281            return y
282
283        mu_max = self.apply(arr)
284        if (mu_max > 0).all():
285            mu_opt = sopt.brentq(f, a=0, b=mu_max)
286
287            # Inplace implementation of
288            #     y = sgn(arr) * fmin(|arr|, mu_opt)
289            xp.fabs(arr, out=y)
290            xp.fmin(y, mu_opt, out=y)
291            y *= xp.sign(arr)
292
293        return y

296class L21Norm(_ShiftLossMixin, pxa.ProxFunc):
297    r"""
298    Mixed :math:`\ell_{2}-\ell_{1}` norm,
299    :math:`\Vert\mathbf{x}\Vert_{2, 1} := \sum_{i=1}^{N} \sqrt{\sum_{j=1}^{M} x_{i, j}^{2}}`,
300    for arrays of dimension :math:`\geq 2`.
301
302    Note
303    ----
304    The input array need not be 2-dimensional:
305    the :math:`\ell_{2}` norm is applied along a predefined subset of dimensions, and the
306    :math:`\ell_{1}` norm on the remaining ones.
307    """
308


[docs]
309    def __init__(
310        self,
311        arg_shape: pxt.NDArrayShape,
312        l2_axis: pxt.NDArrayAxis = (0,),
313    ):
314        r"""
315        Parameters
316        ----------
317        arg_shape: NDArrayShape
318            Shape of the input array.
319        l2_axis: NDArrayAxis
320            Axis (or axes) along which the :math:`\ell_{2}` norm is applied.
321        """
322        arg_shape = pxu.as_canonical_shape(arg_shape)
323        assert all(a > 0 for a in arg_shape)
324        N_dim = len(arg_shape)
325        assert N_dim >= 2
326
327        l2_axis = np.unique(pxu.as_canonical_shape(l2_axis))  # drop potential duplicates
328        assert np.all((-N_dim <= l2_axis) & (l2_axis < N_dim))  # all axes in valid range
329        l2_axis = (l2_axis + N_dim) % N_dim  # get rid of negative axes
330        l1_axis = np.setdiff1d(np.arange(N_dim), l2_axis)
331
332        super().__init__(shape=(1, np.prod(arg_shape)))
333        self.lipschitz = np.inf
334        self._arg_shape = arg_shape
335        self._l1_axis = l1_axis
336        self._l2_axis = l2_axis


337
338    @pxrt.enforce_precision(i="arr")
339    def apply(self, arr: pxt.NDArray):
340        sh = arr.shape[:-1]
341        arr = arr.reshape(sh + self._arg_shape)
342
343        l2_axis = tuple(len(sh) + self._l2_axis)
344        x = (arr**2).sum(axis=l2_axis, keepdims=True)
345        xp = pxu.get_array_module(arr)
346        xp.sqrt(x, out=x)
347
348        l1_axis = tuple(len(sh) + self._l1_axis)
349        out = x.sum(axis=l1_axis, keepdims=True)
350        return out.reshape(*sh, -1)
351
352    @pxrt.enforce_precision(i=("arr", "tau"))
353    def prox(self, arr: pxt.NDArray, tau: pxt.Real):
354        sh = arr.shape[:-1]
355        arr = arr.reshape(sh + self._arg_shape)
356
357        l2_axis = tuple(len(sh) + self._l2_axis)
358        n = (arr**2).sum(axis=l2_axis, keepdims=True)
359        xp = pxu.get_array_module(arr)
360        xp.sqrt(n, out=n)
361
362        out = arr.copy()
363        out *= 1 - tau / xp.fmax(n, tau)
364        return out.reshape(*sh, -1)

367class PositiveL1Norm(_ShiftLossMixin, pxa.ProxFunc):
368    r"""
369    :math:`\ell_{1}`-norm, with a positivity constraint.
370
371    .. math::
372
373       f(\mathbf{x})
374       :=
375       \lVert\mathbf{x}\rVert_{1} + \iota_{+}(\mathbf{x}),
376
377    .. math::
378
379       \textbf{prox}_{\tau f}(\mathbf{z})
380       :=
381       \max(\mathrm{soft}_\tau(\mathbf{z}), \mathbf{0})
382
383    See Also
384    --------
385    :py:class:`~pyxu.operator.func.indicator.PositiveOrthant`
386    """
387
388    def __init__(self, dim: pxt.Integer):
389        super().__init__(shape=(1, dim))
390        from pyxu.operator.func.indicator import PositiveOrthant
391
392        self._indicator = PositiveOrthant(dim=dim)
393        self._l1norm = L1Norm(dim=dim)
394        self.lipschitz = np.inf
395
396    @pxrt.enforce_precision(i="arr")
397    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
398        return self._indicator(arr) + self._l1norm(arr)
399
400    @pxrt.enforce_precision(i=["arr", "tau"])
401    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:
402        xp = pxu.get_array_module(arr)
403        return xp.fmax(0, arr - tau)
