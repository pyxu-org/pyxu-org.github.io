 25class FFTCorrelate(Stencil):
 26    r"""
 27    Multi-dimensional FFT-based correlation.
 28
 29    :py:class:`~pyxu.operator.FFTCorrelate` has the same interface as :py:class:`~pyxu.operator.Stencil`.
 30
 31    .. rubric:: Implementation Notes
 32
 33    * :py:class:`~pyxu.operator.FFTCorrelate` can scale to much larger kernels than :py:class:`~pyxu.operator.Stencil`.
 34    * Kernels must be small enough to fit in memory, i.e. unbounded kernels are not allowed.
 35    * Kernels should be supplied an NUMPY/CUPY arrays. DASK arrays will be evaluated if provided.
 36    * :py:class:`~pyxu.operator.FFTCorrelate` instances are **not arraymodule-agnostic**: they will only work with
 37      NDArrays belonging to the same array module as `kernel`, or DASK arrays where the chunk backend matches the
 38      init-supplied kernel backend.
 39    * A warning is emitted if inputs must be cast to the kernel dtype.
 40    * The input array is transformed by calling :py:class:`~pyxu.operator.FFT`.
 41    * When operating on DASK inputs, the kernel DFT is computed per chunk at the best size to handle the inputs.
 42      This was deemed preferable than pre-computing a huge kernel DFT once, then sending it to each worker process to
 43      compute its chunk.
 44
 45    See Also
 46    --------
 47    :py:class:`~pyxu.operator.Stencil`,
 48    :py:class:`~pyxu.operator.FFTConvolve`
 49    """

 51    def __init__(
 52        self,
 53        dim_shape: pxt.NDArrayShape,
 54        kernel: Stencil.KernelSpec,
 55        center: _Stencil.IndexSpec,
 56        mode: Pad.ModeSpec = "constant",
 57        enable_warnings: bool = True,
 58        **kwargs,
 59    ):
 60        r"""
 61        Parameters
 62        ----------
 63        dim_shape: NDArrayShape
 64            (M1,...,MD) input dimensions.
 65        kernel: ~pyxu.operator.Stencil.KernelSpec
 66            Kernel coefficients.  Two forms are accepted:
 67
 68            * NDArray of rank-:math:`D`: denotes a non-seperable kernel.
 69            * tuple[NDArray_1, ..., NDArray_D]: a sequence of 1D kernels such that dimension[k] is filtered by kernel
 70              `kernel[k]`, that is:
 71
 72              .. math::
 73
 74                 k = k_1 \otimes\cdots\otimes k_D,
 75
 76              or in Python: ``k = functools.reduce(numpy.multiply.outer, kernel)``.
 77
 78        center: ~pyxu.operator._Stencil.IndexSpec
 79            (i1,...,iD) index of the kernel's center.
 80
 81            `center` defines how a kernel is overlaid on inputs to produce outputs.
 82
 83        mode: str, :py:class:`list` ( str )
 84            Boundary conditions.  Multiple forms are accepted:
 85
 86            * str: unique mode shared amongst dimensions.
 87              Must be one of:
 88
 89              * 'constant' (zero-padding)
 90              * 'wrap'
 91              * 'reflect'
 92              * 'symmetric'
 93              * 'edge'
 94            * tuple[str, ...]: dimension[k] uses `mode[k]` as boundary condition.
 95
 96            (See :py:func:`numpy.pad` for details.)
 97        enable_warnings: bool
 98            If ``True``, emit a warning in case of precision mis-match issues.
 99        kwargs: dict
100            Extra kwargs forwarded to :py:class:`~pyxu.operator.FFT`.
101        """
102        super().__init__(
103            dim_shape=dim_shape,
104            kernel=kernel,
105            center=center,
106            mode=mode,
107            enable_warnings=enable_warnings,
108        )
109        self._fft_kwargs = kwargs  # Extra kwargs passed to FFT()

116    def _compute_pad_width(_kernel, _center, _mode) -> Pad.WidthSpec:
117        N = _kernel[0].ndim
118        pad_width = [None] * N
119        for i in range(N):
120            if len(_kernel) == 1:  # non-seperable filter
121                n = _kernel[0].shape[i]
122            else:  # seperable filter(s)
123                n = _kernel[i].size
124
125            # 1. Pad/Trim operators are shared amongst [apply,adjoint]():
126            #    lhs/rhs are thus padded equally.
127            # 2. Pad width must match kernel dimensions to retain border effects.
128            pad_width[i] = (n - 1, n - 1)
129        return tuple(pad_width)

131    @staticmethod
132    def _init_fw(_kernel, _center) -> list:
133        # Initialize kernels used in apply().
134        # The returned objects must have the following fields:
135        # * _kernel: ndarray[float] (D,)
136        # * _center: ndarray[int] (D,)
137
138        # Store kernels in convolution form.
139        _st_fw = [None] * len(_kernel)
140        _kernel, _center = Stencil._bw_equivalent(_kernel, _center)
141        for i, (k_fw, c_fw) in enumerate(zip(_kernel, _center)):
142            _st_fw[i] = KernelInfo(k_fw, c_fw)
143        return _st_fw

158    def _stencil_chain(self, x: pxt.NDArray, stencils: list) -> pxt.NDArray:
159        # Apply sequence of stencils to `x`.
160        #
161        # x: (..., M1,...,MD)
162        # z: (..., M1,...,MD)
163
164        # Contrary to Stencil._stencil_chain(), the `stencils` parameter is picklable directly.
165        def _chain(x, stencils, fft_kwargs):
166            xp = pxu.get_array_module(x)
167            xpf = FFT.fft_backend(xp)
168
169            # Compute constants -----------------------------------------------
170            if uni_kernel := (len(stencils) == 1):
171                M = np.r_[stencils[0]._kernel.shape]
172                dim_rank = stencils[0]._kernel.ndim
173            else:
174                M = np.array([st._kernel.size for st in stencils])
175                dim_rank = len(stencils)
176            Np = np.r_[x.shape[-dim_rank:]]
177            L = FFT.next_fast_len(Np + M - 1, xp=xp)
178            axes = tuple(range(-dim_rank, 0))
179
180            # Apply stencils in DFT domain ------------------------------------
181            fft = FFT(L, axes, **fft_kwargs)
182            Z = fft.capply(x)
183            if uni_kernel:
184                if len(M) == 1:  # 1D kernel
185                    K = xpf.fft(stencils[0]._kernel, n=L[0])
186                else:  # ND kernel
187                    K = fft.capply(stencils[0]._kernel)
188                Z *= K
189            else:
190                for ax, st in enumerate(stencils):
191                    K = xpf.fft(st._kernel, n=L[ax], axis=ax)
192                    Z *= K
193            z = fft.cpinv(Z, damp=0).real
194
195            # Extract ROI -----------------------------------------------------
196            if uni_kernel:
197                center = stencils[0]._center
198            else:
199                center = [st._center[i] for (i, st) in enumerate(stencils)]
200            extract = [slice(c, c + n) for (c, n) in zip(center, Np)]
201            return z[..., *extract]
202
203        ndi = pxd.NDArrayInfo.from_obj(x)
204        if ndi == pxd.NDArrayInfo.DASK:
205            # Compute (depth,boundary) values for [overlap,trim_internal]()
206            N_stack = x.ndim - self.dim_rank
207            depth = {ax: 0 for ax in range(x.ndim)}
208            for ax, (p_lhs, p_rhs) in enumerate(self._pad._pad_width):
209                assert p_lhs == p_rhs, "top-level Pad() should be symmetric."
210                depth[N_stack + ax] = p_lhs
211            boundary = 0
212
213            xp = ndi.module()
214            x_overlap = xp.overlap.overlap(  # Share padding between chunks
215                x,
216                depth=depth,
217                boundary=boundary,
218            )
219            z_overlap = x_overlap.map_blocks(  # Map _chain() to each chunk
220                func=_chain,
221                dtype=x.dtype,
222                chunks=x_overlap.chunks,
223                meta=x._meta,
224                # extra _chain() kwargs -------------------
225                stencils=stencils,
226                fft_kwargs=self._fft_kwargs,
227            )
228            z = xp.overlap.trim_internal(  # Trim inter-chunk excess
229                z_overlap,
230                axes=depth,
231                boundary=boundary,
232            )
233        else:
234            z = _chain(x, stencils, self._fft_kwargs)
235        return z
