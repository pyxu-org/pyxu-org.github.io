 17class FFT(pxa.LinOp):  # Inherits from LinOp instead of NormalOp since operator not square if `real=True`.
 18    r"""
 19    Multi-dimensional Discrete Fourier Transform (DFT) :math:`A: \mathbb{C}^{N_{1} \times \cdots
 20    \times N_{D}} \to \mathbb{C}^{N_{1} \times \cdots \times N_{D}}`:
 21
 22    .. math::
 23
 24       (A \, \mathbf{x})[\mathbf{k}]
 25       =
 26       \sum_{\mathbf{n}} \mathbf{x}[\mathbf{n}]
 27       \exp\left[-j 2 \pi \langle \frac{\mathbf{n}}{\mathbf{N}}, \mathbf{k} \rangle \right],
 28
 29    .. math::
 30
 31       (A^{*} \, \hat{\mathbf{x}})[\mathbf{n}]
 32       =
 33       \sum_{\mathbf{k}} \hat{\mathbf{x}}[\mathbf{k}]
 34       \exp\left[j 2 \pi \langle \frac{\mathbf{n}}{\mathbf{N}}, \mathbf{k} \rangle \right],
 35
 36    .. math::
 37
 38       (\mathbf{x}, \, \hat{\mathbf{x}}) \in \mathbb{C}^{N_{1} \times \cdots \times N_{D}},
 39       \quad
 40       (\mathbf{n}, \, \mathbf{k}) \in \{0, \ldots, N_{1}-1\} \times \cdots \times \{0, \ldots, N_{D}-1\}.
 41
 42    The DFT is taken over any number of axes by means of the Fast Fourier Transform algorithm (FFT).
 43
 44
 45    .. rubric:: Implementation Notes
 46
 47    * The CPU implementation uses `SciPy's FFT implementation <https://docs.scipy.org/doc/scipy/reference/fft.html>`_.
 48    * The GPU implementation uses cuFFT via `CuPy <https://docs.cupy.dev/en/latest/reference/scipy_fft.html>`_.
 49
 50
 51    Examples
 52    --------
 53
 54    * 1D DFT of a cosine pulse.
 55
 56      .. code-block:: python3
 57
 58         from pyxu.operator.linop import FFT
 59         import pyxu.util as pxu
 60
 61         N = 10
 62         op = FFT(
 63             arg_shape=(N,),
 64             real=True,
 65         )
 66
 67         x = np.cos(2 * np.pi / N * np.arange(N))
 68         y = pxu.view_as_complex(op.apply(x))
 69         # [0, N/2, 0, 0, 0, 0, 0, 0, 0, N/2]
 70
 71         z = op.adjoint(op.apply(x))
 72         # np.allclose(z, N * x) -> True
 73
 74    * 1D DFT of a complex exponential pulse.
 75
 76      .. code-block:: python3
 77
 78         from pyxu.operator.linop import FFT
 79         import pyxu.util as pxu
 80
 81         N = 10
 82         op = FFT(
 83             arg_shape=(N,),
 84             real=False,  # complex-valued inputs
 85         )
 86
 87         x = np.exp(1j * 2 * np.pi / N * np.arange(N))
 88         y = pxu.view_as_complex(
 89                 op.apply(
 90                     pxu.view_as_real(x)
 91                 )
 92             )
 93         # [0, N, 0, 0, 0, 0, 0, 0, 0, 0]
 94
 95         z = pxu.view_as_complex(
 96                 op.adjoint(
 97                     op.apply(
 98                         pxu.view_as_real(x)
 99                     )
100                 )
101             )
102         # np.allclose(z, N * x) -> True
103
104    * 2D DFT of an image
105
106      .. code-block:: python3
107
108         from pyxu.operator.linop import FFT
109         import pyxu.util as pxu
110
111         N_h, N_w = 10, 8
112         op = FFT(
113             arg_shape=(N_h, N_w),
114             real=True,
115         )
116
117         x = np.pad(  # an image
118             np.ones((N_h//2, N_w//2)),
119             pad_width=((0, N_h//2), (0, N_w//2)),
120         )
121         y = pxu.view_as_complex(  # sinc
122                 op.apply(x.reshape(-1))
123             ).reshape(N_h, N_w)
124         z = op.adjoint(
125                 op.apply(x.reshape(-1))
126             ).reshape(N_h, N_w)
127         # np.allclose(z, (N_h * N_w) * x) -> True
128
129    * 1D DFT of an image's rows
130
131      .. code-block:: python3
132
133         from pyxu.operator.linop import FFT
134         import pyxu.util as pxu
135
136         N_h, N_w = 10, 8
137         op = FFT(
138             arg_shape=(N_h, N_w),
139             axes=-1,
140             real=True,
141         )
142
143         x = np.pad(  # an image
144             np.ones((N_h//2, N_w//2)),
145             pad_width=((0, N_h//2), (0, N_w//2)),
146         )
147         y = pxu.view_as_complex(  # sinc
148                 op.apply(x.reshape(-1))
149             ).reshape(N_h, N_w)
150         z = op.adjoint(
151                 op.apply(x.reshape(-1))
152             ).reshape(N_h, N_w)
153         # np.allclose(z, N_w * x) -> True

156    def __init__(
157        self,
158        arg_shape: pxt.NDArrayShape,
159        axes: pxt.NDArrayAxis = None,
160        real: bool = False,
161        **kwargs,
162    ):
163        r"""
164        Parameters
165        ----------
166        arg_shape: NDArrayShape
167            (N_1, ..., N_D) shape of the input array :math:`\mathbf{x} \in \mathbb{R}^{N_{1} \times \cdots \times
168            N_{D}}` or :math:`\mathbb{C}^{N_{1} \times \cdots \times N_{D}}`.
169        axes: NDArrayAxis
170            Axis or axes along which the DFT is performed.
171            The default, axes=None, will transform all dimensions of the input array.
172        real: bool
173            If ``True``, assumes :py:func:`~pyxu.operator.linop.fft.fft.FFT.apply` takes (..., N.prod()) inputs in
174            :math:`\mathbb{R}^{N}`.
175
176            If ``False``, then :py:func:`~pyxu.operator.linop.fft.fft.FFT.apply` takes (..., 2N.prod()) inputs, i.e.
177            :math:`\mathbb{C}^{N}` vectors viewed as bijections with :math:`\mathbb{R}^{2N}`.
178        kwargs: dict
179            Extra kwargs passed to :py:func:`scipy.fft.fftn` or :py:func:`cupyx.scipy.fft.fftn`.
180
181            Supported parameters for :py:func:`scipy.fft.fftn` are:
182
183                * workers: int = 1
184
185            Supported parameters for :py:func:`cupyx.scipy.fft.fftn` are:
186
187                * NOT SUPPORTED FOR NOW
188
189            Default values are chosen if unspecified.
190        """
191        arg_shape = pxu.as_canonical_shape(arg_shape)
192        N_dim, N = len(arg_shape), np.prod(arg_shape)
193
194        if axes is None:
195            axes = tuple(range(N_dim))
196        axes = np.unique(pxu.as_canonical_shape(axes))  # drop potential duplicates
197        assert np.all((-N_dim <= axes) & (axes < N_dim))  # all axes in valid range
198        axes = (axes + N_dim) % N_dim  # get rid of negative axes
199
200        sh_op = [2 * N, 2 * N]
201        sh_op[1] //= 2 if real else 1
202        super().__init__(shape=sh_op)
203
204        self._arg_shape = tuple(arg_shape)
205        self._axes = tuple(axes)
206        self._real = bool(real)
207        self._kwargs = {
208            pxd.NDArrayInfo.NUMPY: dict(
209                workers=kwargs.get("workers", 1),
210            ),
211            pxd.NDArrayInfo.CUPY: dict(),
212        }
213
214        self.lipschitz = self.estimate_lipschitz()

216    def estimate_lipschitz(self, **kwargs) -> pxt.Real:
217        arg_shape = np.array(self._arg_shape, dtype=int)
218        axes = np.array(self._axes, dtype=int)
219
220        L = np.sqrt(arg_shape[axes].prod())
221        return L

223    def gram(self) -> pxt.OpT:
224        from pyxu.operator.linop import HomothetyOp
225
226        op_g = HomothetyOp(dim=self.dim, cst=self.lipschitz**2)
227        return op_g

229    def cogram(self) -> pxt.OpT:
230        if self._real:
231            # Enforcing real-valued adjoint() outputs implies a projection between A and A.T.
232            # There is no simple closed form to compute the co-gram in this case.
233            op_cg = super().cogram()
234        else:
235            from pyxu.operator.linop import HomothetyOp
236
237            op_cg = HomothetyOp(dim=self.codim, cst=self.lipschitz**2)
238        return op_cg

240    @pxrt.enforce_precision(i=("arr", "damp"))
241    def pinv(self, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
242        N = self.lipschitz**2
243        out = self.adjoint(arr)
244        out /= N + damp
245        return out

247    def dagger(self, damp: pxt.Real, **kwargs) -> pxt.OpT:
248        N = self.lipschitz**2
249        op = self.T / (N + damp)
250        return op

257    def _transform(self, arr: pxt.NDArray, mode: str) -> pxt.NDArray:
258        # Internal method for apply/adjoint.
259        #
260        # Parameters
261        # ----------
262        # arr: pxt.NDArray [float/complex]
263        #     (N_1, ..., N_D) array to transform.
264        # mode: str
265        #     Transform direction:
266        #
267        #     * 'fw': fftn(, norm="backward")
268        #     * 'bw': ifftn(, norm="forward")
269        #
270        # Returns
271        # -------
272        # out: pxt.NDArray [complex]
273        #     (N_1, ..., N_D) transformed array.
274        N = pxd.NDArrayInfo  # shorthand
275        ndi = N.from_obj(arr)
276
277        if ndi == N.NUMPY:
278            fft = pxu.import_module("scipy.fft")
279        elif ndi == N.CUPY:
280            fft = pxu.import_module("cupyx.scipy.fft")
281        else:
282            raise ValueError("Unknown NDArray category.")
283
284        func, norm = dict(  # ref: scipy.fft norm conventions
285            fw=(fft.fftn, "backward"),
286            bw=(fft.ifftn, "forward"),
287        )[mode]
288
289        # `self._kwargs()` contains parameters undersood by different FFT backends.
290        # If we must fallback to `scipy.fft`, need to drop all non-standard parameters.
291        sig = inspect.Signature.from_callable(func)
292        kwargs = {k: v for (k, v) in self._kwargs[ndi].items() if (k in sig.parameters)}
293
294        out = func(
295            x=arr,
296            axes=self._axes,
297            norm=norm,
298            **kwargs,
299        )
300        return out

302    @pxrt.enforce_precision(i="arr")
303    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
304        r"""
305        Parameters
306        ----------
307        arr: NDArray
308            * (...,  N.prod()) inputs :math:`\mathbf{x} \in \mathbb{R}^{N_{1} \times \cdots \times N_{D}}`
309              (``real=True``.)
310            * (..., 2N.prod()) inputs :math:`\mathbf{x} \in \mathbb{C}^{N_{1} \times \cdots \times N_{D}}` viewed as a
311              real array. (See :py:func:`~pyxu.util.complex.view_as_real`.)
312
313        Returns
314        -------
315        out: NDArray
316            (..., 2N.prod()) outputs :math:`\hat{\mathbf{x}} \in \mathbb{C}^{N_{1} \times \cdots \times N_{D}}` viewed
317            as a real array. (See :py:func:`~pyxu.util.complex.view_as_real`.)
318        """
319        if self._real:
320            r_width = pxrt.Width(arr.dtype)
321            c_dtype = r_width.complex.value
322        else:
323            arr = pxu.view_as_complex(arr)
324            c_dtype = arr.dtype
325
326        sh = arr.shape[:-1]
327        N_trans = np.prod(sh, dtype=int)
328        func = lambda _: self._transform(arr=_, mode="fw")
329        blks = _dask_zip(
330            func=(func,) * N_trans,
331            data=arr.reshape(N_trans, *self._arg_shape),
332            out_shape=(self._arg_shape,) * N_trans,
333            out_dtype=(c_dtype,) * N_trans,
334            parallel=True,
335        )
336
337        xp = pxu.get_array_module(arr)
338        out = xp.stack(blks, axis=0).reshape(*sh, -1)
339
340        return pxu.view_as_real(out)

342    @pxrt.enforce_precision(i="arr")
343    def adjoint(self, arr: pxt.NDArray) -> pxt.NDArray:
344        r"""
345        Parameters
346        ----------
347        arr: NDArray
348            (..., 2N.prod()) outputs :math:`\hat{\mathbf{x}} \in \mathbb{C}^{N_{1} \times \cdots \times N_{D}}` viewed
349            as a real array. (See :py:func:`~pyxu.util.complex.view_as_real`.)
350
351        Returns
352        -------
353        out: NDArray
354            * (...,  N.prod()) inputs :math:`\mathbf{x} \in \mathbb{R}^{N_{1} \times \cdots \times N_{D}}`
355              (``real=True``.)
356            * (..., 2N.prod()) inputs :math:`\mathbf{x} \in \mathbb{C}^{N_{1} \times \cdots \times N_{D}}` viewed as a
357              real array. (See :py:func:`~pyxu.util.complex.view_as_real`.)
358        """
359        arr = pxu.view_as_complex(arr)
360        c_dtype = arr.dtype
361
362        sh = arr.shape[:-1]
363        N_trans = np.prod(sh, dtype=int)
364        func = lambda _: self._transform(arr=_, mode="bw")
365        blks = _dask_zip(
366            func=(func,) * N_trans,
367            data=arr.reshape(N_trans, *self._arg_shape),
368            out_shape=(self._arg_shape,) * N_trans,
369            out_dtype=(c_dtype,) * N_trans,
370            parallel=True,
371        )
372
373        xp = pxu.get_array_module(arr)
374        out = xp.stack(blks, axis=0).reshape(*sh, -1)
375
376        if self._real:
377            return out.real
378        else:
379            return pxu.view_as_real(out)
