 24class IdentityOp(pxa.OrthProjOp):
 25    def __init__(self, dim: pxt.Integer):
 26        super().__init__(shape=(dim, dim))
 27        self.lipschitz = 1
 28
 29    @pxrt.enforce_precision(i="arr")
 30    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 31        return pxu.read_only(arr)
 32
 33    @pxrt.enforce_precision(i="arr")
 34    def adjoint(self, arr: pxt.NDArray) -> pxt.NDArray:
 35        return pxu.read_only(arr)
 36
 37    def svdvals(self, **kwargs) -> pxt.NDArray:
 38        return pxa.UnitOp.svdvals(self, **kwargs)
 39
 40    def asarray(self, **kwargs) -> pxt.NDArray:
 41        xp = kwargs.get("xp", pxd.NDArrayInfo.default().module())
 42        dtype = kwargs.get("dtype", pxrt.getPrecision().value)
 43        A = xp.eye(N=self.dim, dtype=dtype)
 44        return A
 45
 46    @pxrt.enforce_precision(i=("arr", "damp"))
 47    def pinv(self, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
 48        out = arr.copy()
 49        out /= 1 + damp
 50        return out
 51
 52    def dagger(self, damp: pxt.Real, **kwargs) -> pxt.OpT:
 53        cst = 1 / (1 + damp)
 54        op = HomothetyOp(cst=cst, dim=self.dim)
 55        return op
 56
 57    @pxrt.enforce_precision()
 58    def trace(self, **kwargs) -> pxt.Real:
 59        return self.dim

 62class NullOp(pxa.LinOp):
 63    """
 64    Null operator.
 65
 66    This operator maps any input vector on the null vector.
 67    """
 68
 69    def __init__(self, shape: pxt.OpShape):
 70        super().__init__(shape=shape)
 71        self.lipschitz = 0
 72
 73    @pxrt.enforce_precision(i="arr")
 74    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 75        xp = pxu.get_array_module(arr)
 76        return xp.broadcast_to(
 77            xp.array(0, arr.dtype),
 78            (*arr.shape[:-1], self.codim),
 79        )
 80
 81    @pxrt.enforce_precision(i="arr")
 82    def adjoint(self, arr: pxt.NDArray) -> pxt.NDArray:
 83        xp = pxu.get_array_module(arr)
 84        return xp.broadcast_to(
 85            xp.array(0, arr.dtype),
 86            (*arr.shape[:-1], self.dim),
 87        )
 88
 89    def svdvals(self, **kwargs) -> pxt.NDArray:
 90        gpu = kwargs.get("gpu", False)
 91        xp = pxd.NDArrayInfo.from_flag(gpu).module()
 92        width = pxrt.getPrecision()
 93
 94        D = xp.zeros(kwargs["k"], dtype=width.value)
 95        return D
 96
 97    def gram(self) -> pxt.OpT:
 98        op = NullOp(shape=(self.dim, self.dim))
 99        return op.asop(pxa.SelfAdjointOp).squeeze()
100
101    def cogram(self) -> pxt.OpT:
102        op = NullOp(shape=(self.codim, self.codim))
103        return op.asop(pxa.SelfAdjointOp).squeeze()
104
105    def asarray(self, **kwargs) -> pxt.NDArray:
106        xp = kwargs.get("xp", pxd.NDArrayInfo.default().module())
107        dtype = kwargs.get("dtype", pxrt.getPrecision().value)
108        A = xp.zeros(self.shape, dtype=dtype)
109        return A
110
111    @pxrt.enforce_precision()
112    def trace(self, **kwargs) -> pxt.Real:
113        return 0

127def HomothetyOp(dim: pxt.Integer, cst: pxt.Real) -> pxt.OpT:
128    """
129    Scaling operator.
130
131    Parameters
132    ----------
133    dim: Integer
134        Dimension of the domain.
135    cst: Real
136        Scaling factor.
137
138    Returns
139    -------
140    op: OpT
141        (dim, dim) scaling operator.
142
143    Note
144    ----
145    This operator is not defined in terms of :py:func:`~pyxu.operator.linop.base.DiagonalOp` since it is
146    array-backend-agnostic.
147    """
148    assert isinstance(cst, pxt.Real), f"cst: expected real, got {cst}."
149
150    if np.isclose(cst, 0):
151        op = NullOp(shape=(dim, dim))
152    elif np.isclose(cst, 1):
153        op = IdentityOp(dim=dim)
154    else:  # build PosDef or SelfAdjointOp
155
156        @pxrt.enforce_precision(i="arr")
157        def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
158            out = arr.copy()
159            out *= _._cst
160            return out
161
162        def op_svdvals(_, **kwargs) -> pxt.NDArray:
163            gpu = kwargs.get("gpu", False)
164            xp = pxd.NDArrayInfo.from_flag(gpu).module()
165            width = pxrt.getPrecision()
166
167            D = xp.full(
168                shape=kwargs["k"],
169                fill_value=abs(_._cst),
170                dtype=width.value,
171            )
172            return D
173
174        @pxrt.enforce_precision(i=("arr", "damp"))
175        def op_pinv(_, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
176            out = arr.copy()
177            scale = _._cst / (_._cst**2 + damp)
178            out *= scale
179            return out
180
181        def op_dagger(_, damp: pxt.Real, **kwargs) -> pxt.OpT:
182            scale = _._cst / (_._cst**2 + damp)
183            op = HomothetyOp(cst=scale, dim=_.dim)
184            return op
185
186        def op_gram(_):
187            return HomothetyOp(cst=_._cst**2, dim=_.dim)
188
189        @pxrt.enforce_precision()
190        def op_trace(_, **kwargs):
191            out = _._cst * _.codim
192            return out
193
194        op = pxsrc.from_source(
195            cls=pxa.PosDefOp if (cst > 0) else pxa.SelfAdjointOp,
196            shape=(dim, dim),
197            embed=dict(
198                _name="HomothetyOp",
199                _cst=cst,
200            ),
201            apply=op_apply,
202            svdvals=op_svdvals,
203            pinv=op_pinv,
204            gram=op_gram,
205            cogram=op_gram,
206            trace=op_trace,
207        )
208        op.dagger = types.MethodType(op_dagger, op)
209        op.lipschitz = abs(cst)
210    return op.squeeze()

213def DiagonalOp(
214    vec: pxt.NDArray,
215    enable_warnings: bool = True,
216) -> pxt.OpT:
217    r"""
218    Diagonal linear operator :math:`\mathbf{D}: \mathbf{x} \to \text{diag}(\mathbf{v}) \mathbf{x}`.
219
220    Note
221    ----
222    :py:func:`~pyxu.operator.linop.base.DiagonalOp` instances are **not arraymodule-agnostic**:
223    they will only work with NDArrays belonging to the same array module as `vec`.
224    Moreover, inner computations may cast input arrays when the precision of `vec` does not match the user-requested
225    precision.
226    If such a situation occurs, a warning is raised.
227
228    Parameters
229    ----------
230    vec: NDArray
231        (N,) diagonal scale factors.
232    enable_warnings: bool
233        If ``True``, emit a warning in case of precision mis-match issues.
234    """
235    assert len(vec) == np.prod(vec.shape), f"vec: {vec.shape} is not a DiagonalOp generator."
236    if (dim := vec.size) == 1:  # Module-agnostic
237        return HomothetyOp(cst=float(vec), dim=1)
238    else:
239        xp = pxu.get_array_module(vec)
240        if pxu.compute(xp.allclose(vec, 0)):
241            op = NullOp(shape=(dim, dim))
242        elif pxu.compute(xp.allclose(vec, 1)):
243            op = IdentityOp(dim=dim)
244        else:  # build PosDef or SelfAdjointOp
245
246            @pxrt.enforce_precision(i="arr")
247            def op_apply(_, arr):
248                if (_._vec.dtype != arr.dtype) and _._enable_warnings:
249                    msg = "Computation may not be performed at the requested precision."
250                    warnings.warn(msg, pxw.PrecisionWarning)
251                out = arr.copy()
252                out *= _._vec
253                return out
254
255            def op_asarray(_, **kwargs) -> pxt.NDArray:
256                xp = kwargs.get("xp", pxd.NDArrayInfo.default().module())
257                dtype = kwargs.get("dtype", pxrt.getPrecision().value)
258
259                v = pxu.compute(_._vec.astype(dtype=dtype, copy=False))
260                v = pxu.to_NUMPY(v)
261                A = xp.diag(v)
262                return A
263
264            def op_gram(_):
265                return DiagonalOp(
266                    vec=_._vec**2,
267                    enable_warnings=_._enable_warnings,
268                )
269
270            def op_svdvals(_, **kwargs):
271                gpu = kwargs.get("gpu", False)
272                xp = pxd.NDArrayInfo.from_flag(gpu).module()
273                width = pxrt.getPrecision()
274
275                k = kwargs["k"]
276                which = kwargs.get("which", "LM")
277
278                D = xp.abs(pxu.compute(_._vec))
279                D = D[xp.argsort(D)]
280                D = D.astype(width.value, copy=False)
281                return D[:k] if (which == "SM") else D[-k:]
282
283            @pxrt.enforce_precision(i=("arr", "damp"))
284            def op_pinv(_, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
285                with warnings.catch_warnings():
286                    warnings.simplefilter("ignore")
287                    scale = _._vec / (_._vec**2 + damp)
288                    scale[xp.isnan(scale)] = 0
289                out = arr.copy()
290                out *= scale
291                return out
292
293            def op_dagger(_, damp: pxt.Real, **kwargs) -> pxt.OpT:
294                with warnings.catch_warnings():
295                    warnings.simplefilter("ignore")
296                    scale = _._vec / (_._vec**2 + damp)
297                    scale[xp.isnan(scale)] = 0
298                return DiagonalOp(
299                    vec=scale,
300                    enable_warnings=_._enable_warnings,
301                )
302
303            @pxrt.enforce_precision()
304            def op_trace(_, **kwargs):
305                return float(_._vec.sum())
306
307            def op_estimate_lipschitz(_, **kwargs):
308                # Calling LinOp's generic method=svd solver may fail since it relies on LinearOperator.
309                # We insead use the fact that _lipschitz is computed exactly at construction time.
310                return _._lipschitz
311
312            op = pxsrc.from_source(
313                cls=pxa.PosDefOp if pxu.compute(xp.all(vec > 0)) else pxa.SelfAdjointOp,
314                shape=(dim, dim),
315                embed=dict(
316                    _name="DiagonalOp",
317                    _vec=vec,
318                    _enable_warnings=bool(enable_warnings),
319                ),
320                apply=op_apply,
321                estimate_lipschitz=op_estimate_lipschitz,
322                asarray=op_asarray,
323                gram=op_gram,
324                cogram=op_gram,
325                svdvals=op_svdvals,
326                pinv=op_pinv,
327                trace=op_trace,
328            )
329            op.dagger = types.MethodType(op_dagger, op)
330            op.lipschitz = float(abs(vec).max())
331        return op.squeeze()

334def _ExplicitLinOp(
335    cls: pxt.OpC,
336    mat: typ.Union[pxt.NDArray, pxt.SparseArray],
337    enable_warnings: bool = True,
338) -> pxt.OpT:
339    r"""
340    Build a linear operator from its matrix representation.
341
342    Given a matrix :math:`\mathbf{A} \in \mathbb{R}^{N \times M}`, the *explicit linear operator* associated to
343    :math:`\mathbf{A}` is defined as
344
345    .. math::
346
347       f_\mathbf{A}(\mathbf{x})
348       =
349       \mathbf{A}\mathbf{x},
350       \qquad
351       \forall \mathbf{x} \in \mathbb{R}^{M},
352
353    with adjoint given by:
354
355    .. math::
356
357       f^{\ast}_{\mathbf{A}}(\mathbf{z})
358       =
359       \mathbf{A}^{T}\mathbf{z},
360       \qquad
361       \forall \mathbf{z} \in \mathbb{R}^{N}.
362
363    Parameters
364    ----------
365    cls: OpC
366        LinOp sub-class to instantiate.
367    mat: NDArray, SparseArray
368        (M, N) matrix generator.
369        The input array can be *dense* or *sparse*.
370        Accepted sparse arrays are:
371
372        * CPU: COO/CSC/CSR/BSR/GCXS
373        * GPU: COO/CSC/CSR
374    enable_warnings: bool
375        If ``True``, emit a warning in case of precision mis-match issues.
376
377    Notes
378    -----
379    * :py:class:`~pyxu.operator.linop.base._ExplicitLinOp` instances are **not arraymodule-agnostic**:
380      they will only work with NDArrays belonging to the same (dense) array module as `mat`.
381      Moreover, inner computations may cast input arrays when the precision of `mat` does not match the user-requested
382      precision.
383      If such a situation occurs, a warning is raised.
384
385    * The matrix provided to :py:func:`~pyxu.operator.linop.base._ExplicitLinOp` is used as-is and can be accessed via
386      ``.mat``.
387    """
388
389    def _standard_form(A):
390        fail_dense = False
391        try:
392            pxd.NDArrayInfo.from_obj(A)
393        except Exception:
394            fail_dense = True
395
396        fail_sparse = False
397        try:
398            pxd.SparseArrayInfo.from_obj(A)
399        except Exception:
400            fail_sparse = True
401
402        if fail_dense and fail_sparse:
403            raise ValueError("mat: format could not be inferred.")
404        else:
405            return A
406
407    def _matmat(A, b, warn: bool) -> pxt.NDArray:
408        # A: (M, N) dense/sparse
409        # b: (..., N) dense
410        # out: (..., M) dense
411        if (A.dtype != b.dtype) and warn:
412            msg = "Computation may not be performed at the requested precision."
413            warnings.warn(msg, pxw.PrecisionWarning)
414
415        M, N = A.shape
416        sh_out = (*b.shape[:-1], M)
417        b = b.reshape((-1, N)).T  # (N, (...).prod)
418        out = A.dot(b)  # (M, (...).prod)
419        return out.T.reshape(sh_out)
420
421    @pxrt.enforce_precision(i="arr")
422    def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
423        return _matmat(_.mat, arr, warn=_._enable_warnings)
424
425    @pxrt.enforce_precision(i="arr")
426    def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
427        return _matmat(_.mat.T, arr, warn=_._enable_warnings)
428
429    def op_estimate_lipscthitz(_, **kwargs) -> pxt.Real:
430        N = pxd.NDArrayInfo
431        S = pxd.SparseArrayInfo
432
433        try:  # Sparse arrays
434            sdi = S.from_obj(_.mat)
435            if sdi in (S.SCIPY_SPARSE, S.PYDATA_SPARSE):
436                ndi = N.NUMPY
437            else:  # S.CUPY_SPARSE
438                ndi = N.CUPY
439        except Exception:  # Dense arrays
440            ndi = N.from_obj(_.mat)
441
442        kwargs.update(xp=ndi.module())
443        klass = _.__class__
444        return klass.estimate_lipschitz(_, **kwargs)
445
446    def op_asarray(_, **kwargs) -> pxt.NDArray:
447        N = pxd.NDArrayInfo
448        S = pxd.SparseArrayInfo
449        xp = kwargs.get("xp", pxd.NDArrayInfo.default().module())
450        dtype = kwargs.get("dtype", pxrt.getPrecision().value)
451
452        try:  # Sparse arrays
453            info = S.from_obj(_.mat)
454            if info in (S.SCIPY_SPARSE, S.CUPY_SPARSE):
455                f = lambda _: _.toarray()
456            elif info == S.PYDATA_SPARSE:
457                f = lambda _: _.todense()
458            A = f(_.mat.astype(dtype))  # `copy` field not ubiquitous
459        except Exception:  # Dense arrays
460            info = N.from_obj(_.mat)
461            A = pxu.compute(_.mat.astype(dtype, copy=False))
462
463            if info == N.DASK:
464                # Chunks may have been mapped to sparse arrays.
465                # Call .asarray() again for a 2nd pass.
466                _op = _ExplicitLinOp(_.__class__, mat=A)
467                A = _op.asarray(xp=N.NUMPY.module(), dtype=A.dtype)
468        finally:
469            A = pxu.to_NUMPY(A)
470
471        return xp.array(A, dtype=dtype)
472
473    @pxrt.enforce_precision()
474    def op_trace(_, **kwargs) -> pxt.Real:
475        if _.dim != _.codim:
476            raise NotImplementedError
477        else:
478            try:
479                tr = _.mat.trace()
480            except Exception:
481                # .trace() missing for [PYDATA,CUPY]_SPARSE API.
482                S = pxd.SparseArrayInfo
483                info = S.from_obj(_.mat)
484                if info == S.PYDATA_SPARSE:
485                    # use `sparse.diagonal().sum()`, but array must be COO.
486                    try:
487                        A = _.mat.tocoo()  # GCXS inputs
488                    except Exception:
489                        A = _.mat  # COO inputs
490                    finally:
491                        tr = info.module().diagonal(A).sum()
492                elif info == S.CUPY_SPARSE:
493                    tr = _.mat.diagonal().sum()
494                else:
495                    raise ValueError(f"Unknown sparse format {_.mat}.")
496            return float(tr)
497
498    op = pxsrc.from_source(
499        cls=cls,
500        shape=mat.shape,
501        embed=dict(
502            _name="_ExplicitLinOp",
503            mat=_standard_form(mat),
504            _enable_warnings=bool(enable_warnings),
505        ),
506        apply=op_apply,
507        adjoint=op_adjoint,
508        asarray=op_asarray,
509        trace=op_trace,
510        estimate_lipschitz=op_estimate_lipscthitz,
511    )
512    return op
