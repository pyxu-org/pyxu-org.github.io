 16def kron(A: pxt.OpT, B: pxt.OpT) -> pxt.OpT:
 17    r"""
 18    Kronecker product :math:`A \otimes B` between two linear operators.
 19
 20    The Kronecker product :math:`A \otimes B` is defined as
 21
 22    .. math::
 23
 24       A \otimes B
 25       =
 26       \left[
 27           \begin{array}{ccc}
 28               A_{11} B     & \cdots & A_{1N_{A}} B \\
 29               \vdots       & \ddots & \vdots   \\
 30               A_{M_{A}1} B & \cdots & A_{M_{A}N_{A}} B \\
 31           \end{array}
 32       \right],
 33
 34    where :math:`A : \mathbb{R}^{N_{A}} \to \mathbb{R}^{M_{A}}`, and :math:`B : \mathbb{R}^{N_{B}} \to
 35    \mathbb{R}^{M_{B}}`.
 36
 37    Parameters
 38    ----------
 39    A: OpT
 40        (mA, nA) linear operator
 41    B: OpT
 42        (mB, nB) linear operator
 43
 44    Returns
 45    -------
 46    op: OpT
 47        (mA*mB, nA*nB) linear operator.
 48
 49    Notes
 50    -----
 51    This implementation is **matrix-free** by leveraging properties of the Kronecker product, i.e.  :math:`A` and
 52    :math:`B` need not be known explicitly.
 53    In particular :math:`(A \otimes B) x` and :math:`(A \otimes B)^{*} x` are computed implicitly via the relation:
 54
 55    .. math::
 56
 57       \text{vec}(\mathbf{A}\mathbf{B}\mathbf{C})
 58       =
 59       (\mathbf{C}^{T} \otimes \mathbf{A}) \text{vec}(\mathbf{B}),
 60
 61    where :math:`\mathbf{A}`, :math:`\mathbf{B}`, and :math:`\mathbf{C}` are matrices.
 62    """
 63
 64    def _infer_op_shape(shA: pxt.OpShape, shB: pxt.OpShape) -> pxt.OpShape:
 65        sh = (shA[0] * shB[0], shA[1] * shB[1])
 66        return sh
 67
 68    def _infer_op_klass(A: pxt.OpT, B: pxt.OpT) -> pxt.OpC:
 69        # linear \kron linear -> linear
 70        # square (if output square)
 71        # normal \kron normal -> normal
 72        # unit \kron unit -> unit
 73        # self-adj \kron self-adj -> self-adj
 74        # pos-def \kron pos-def -> pos-def
 75        # idemp \kron idemp -> idemp
 76        # func \kron func -> func
 77        properties = set(A.properties() & B.properties())
 78        sh = _infer_op_shape(A.shape, B.shape)
 79        if sh[0] == sh[1]:
 80            properties.add(pxo.Property.LINEAR_SQUARE)
 81        if pxo.Property.FUNCTIONAL in properties:
 82            klass = pxo.LinFunc
 83        else:
 84            klass = pxo.Operator._infer_operator_type(properties)
 85        return klass
 86
 87    @pxrt.enforce_precision(i="arr")
 88    def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
 89        # If `x` is a vector, then:
 90        #     (A \kron B)(x) = vec(B * mat(x) * A.T)
 91        sh_prefix = arr.shape[:-1]
 92        sh_dim = len(sh_prefix)
 93
 94        x = arr.reshape((*sh_prefix, _._A.dim, _._B.dim))  # (..., A.dim, B.dim)
 95        y = _._B.apply(x)  # (..., A.dim, B.codim)
 96        z = y.transpose((*range(sh_dim), -1, -2))  # (..., B.codim, A.dim)
 97        t = _._A.apply(z)  # (..., B.codim, A.codim)
 98        u = t.transpose((*range(sh_dim), -1, -2))  # (..., A.codim, B.codim)
 99
100        out = u.reshape((*sh_prefix, -1))  # (..., A.codim * B.codim)
101        return out
102
103    @pxrt.enforce_precision(i="arr")
104    def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
105        # If `x` is a vector, then:
106        #     (A \kron B).H(x) = vec(B.H * mat(x) * A.conj)
107        sh_prefix = arr.shape[:-1]
108        sh_dim = len(sh_prefix)
109
110        x = arr.reshape((*sh_prefix, _._A.codim, _._B.codim))  # (..., A.codim, B.codim)
111        y = _._B.adjoint(x)  # (..., A.codim, B.dim)
112        z = y.transpose((*range(sh_dim), -1, -2))  # (..., B.dim, A.codim)
113        t = _._A.adjoint(z)  # (..., B.dim, A.dim)
114        u = t.transpose((*range(sh_dim), -1, -2))  # (..., A.dim, B.dim)
115
116        out = u.reshape((*sh_prefix, -1))  # (..., A.dim * B.dim)
117        return out

119    def op_estimate_lipschitz(_, **kwargs) -> pxt.Real:
120        no_eval = "__rule" in kwargs
121        if no_eval:
122            L_A = _._A.lipschitz
123            L_B = _._B.lipschitz
124            L = L_A * L_B
125        else:
126            L = _.__class__.estimate_lipschitz(_, **kwargs)
127        return L

129    def op_asarray(_, **kwargs) -> pxt.NDArray:
130        # (A \kron B).asarray() = A.asarray() \kron B.asarray()
131        A = _._A.asarray(**kwargs)
132        B = _._B.asarray(**kwargs)
133        xp = kwargs.get("xp", pxd.NDArrayInfo.default().module())
134        C = xp.tensordot(A, B, axes=0).transpose((0, 2, 1, 3)).reshape(_.shape)
135        return C
136
137    def op_gram(_) -> pxt.OpT:
138        # (A \kron B).gram() = A.gram() \kron B.gram()
139        A = _._A.gram()
140        B = _._B.gram()
141        op = kron(A, B)
142        return op

144    def op_cogram(_) -> pxt.OpT:
145        # (A \kron B).cogram() = A.cogram() \kron B.cogram()
146        A = _._A.cogram()
147        B = _._B.cogram()
148        op = kron(A, B)
149        return op

151    def op_svdvals(_, **kwargs) -> pxt.NDArray:
152        # (A \kron B).svdvals(k, which)
153        # = outer(
154        #     A.svdvals(k, which),
155        #     B.svdvals(k, which)
156        #   ).[top|bottom](k)
157        k = kwargs.get("k", 1)
158        which = kwargs.get("which", "LM")
159        if which.upper() == "SM":
160            # `scipy.sparse.linalg.svds()` and `scipy.linalg.svd()` will only return up to
161            # min(shape) singular values.
162            # As such (A.svdvals(), B.svdvals()) output is insufficient to infer SM-singular values
163            # of (A \kron B).
164            D_C = _.__class__.svdvals(_, **kwargs)
165        else:
166            D_A = _._A.svdvals(**kwargs)
167            D_B = _._B.svdvals(**kwargs)
168            xp = pxu.get_array_module(D_A)
169            pad_length = np.fmax(k - len(D_A) * len(D_B), 0)
170            D_C = xp.concatenate(
171                [
172                    xp.zeros(pad_length, dtype=D_A.dtype),
173                    xp.sort(xp.outer(D_A, D_B), axis=None),
174                ]
175            )[-k:]
176        return D_C
177
178    @pxrt.enforce_precision(i=("arr", "damp"))
179    def op_pinv(_, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
180        if np.isclose(damp, 0):
181            # (A \kron B).dagger() = A.dagger() \kron B.dagger()
182            op_d = kron(_._A.dagger(damp, **kwargs), _._B.dagger(damp, **kwargs))
183            out = op_d.apply(arr)
184        else:
185            # default algorithm
186            out = _.__class__.pinv(_, arr, damp, **kwargs)
187        return out
188
189    @pxrt.enforce_precision()
190    def op_trace(_, **kwargs) -> pxt.Real:
191        # tr(A \kron B) = tr(A) * tr(B)
192        # [if both square, else default algorithm]
193        P = pxo.Property.LINEAR_SQUARE
194        if not _.has(P):
195            raise NotImplementedError
196
197        if _._A.has(P) and _._B.has(P):
198            tr = _._A.trace(**kwargs) * _._B.trace(**kwargs)
199        else:
200            tr = _.__class__.trace(_, **kwargs)
201        return tr
202
203    _A = A.squeeze()
204    _B = B.squeeze()
205    assert (klass := _infer_op_klass(_A, _B)).has(pxo.Property.LINEAR)
206    is_scalar = lambda _: _.shape == (1, 1)
207    if is_scalar(_A) and is_scalar(_B):
208        from pyxu.operator.linop.base import HomothetyOp
209
210        return HomothetyOp(cst=(_A.asarray() * _B.asarray()).item(), dim=1)
211    elif is_scalar(_A) and (not is_scalar(_B)):
212        return _A.asarray().item() * _B
213    elif (not is_scalar(_A)) and is_scalar(B):
214        return _A * _B.asarray().item()
215    else:
216        op = pxsrc.from_source(
217            cls=klass,
218            shape=_infer_op_shape(_A.shape, _B.shape),
219            embed=dict(
220                _name="kron",
221                _A=_A,
222                _B=_B,
223            ),
224            apply=op_apply,
225            adjoint=op_adjoint,
226            asarray=op_asarray,
227            gram=op_gram,
228            cogram=op_cogram,
229            svdvals=op_svdvals,
230            pinv=op_pinv,
231            trace=op_trace,
232            estimate_lipschitz=op_estimate_lipschitz,
233            _expr=lambda _: (_._name, _._A, _._B),
234        )
235        op.lipschitz = op.estimate_lipschitz(__rule=True)
236    return op

239def khatri_rao(A: pxt.OpT, B: pxt.OpT) -> pxt.OpT:
240    r"""
241    Column-wise Khatri-Rao product :math:`A \circ B` between two linear operators.
242
243    The Khatri-Rao product :math:`A \circ B` is defined as
244
245    .. math::
246
247       A \circ B
248       =
249       \left[
250           \begin{array}{ccc}
251           \mathbf{a}_{1} \otimes \mathbf{b}_{1} & \cdots & \mathbf{a}_{N} \otimes \mathbf{b}_{N}
252           \end{array}
253       \right],
254
255    where :math:`A : \mathbb{R}^{N} \to \mathbb{R}^{M_{A}}`, :math:`B : \mathbb{R}^{N} \to \mathbb{R}^{M_{B}}`, and
256    :math:`\mathbf{a}_{k}` (repectively :math:`\mathbf{b}_{k}`) denotes the :math:`k`-th column of :math:`A`
257    (respectively :math:`B`).
258
259    Parameters
260    ----------
261    A: OpT
262        (mA, n) linear operator
263    B: OpT
264        (mB, n) linear operator
265
266    Returns
267    -------
268    op: OpT
269        (mA*mB, n) linear operator.
270
271    Notes
272    -----
273    This implementation is **matrix-free** by leveraging properties of the Khatri-Rao product, i.e.  :math:`A` and
274    :math:`B` need not be known explicitly.
275    In particular :math:`(A \circ B) x` and :math:`(A \circ B)^{*} x` are computed implicitly via the relation:
276
277    .. math::
278
279       \text{vec}(\mathbf{A}\text{diag}(\mathbf{b})\mathbf{C})
280       =
281       (\mathbf{C}^{T} \circ \mathbf{A}) \mathbf{b},
282
283    where :math:`\mathbf{A}`, :math:`\mathbf{C}` are matrices, and :math:`\mathbf{b}` is a vector.
284
285    Note however that a matrix-free implementation of the Khatri-Rao product does not permit the same optimizations as a
286    matrix-based implementation.
287    Thus the Khatri-Rao product as implemented here is only marginally more efficient than applying
288    :py:func:`~pyxu.operator.linop.kron.kron` and pruning its output.
289    """
290
291    def _infer_op_shape(shA: pxt.OpShape, shB: pxt.OpShape) -> pxt.OpShape:
292        if shA[1] != shB[1]:
293            raise ValueError(f"Khatri-Rao product of {shA} and {shB} operators forbidden.")
294        sh = (shA[0] * shB[0], shA[1])
295        return sh
296
297    def _infer_op_klass(A: pxt.OpT, B: pxt.OpT) -> pxt.OpC:
298        # linear \kr linear -> linear
299        # square (if output square)
300        sh = _infer_op_shape(A.shape, B.shape)
301        if sh[0] == 1:
302            klass = pxo.LinFunc
303        else:
304            properties = set(pxo.LinOp.properties())
305            if sh[0] == sh[1]:
306                properties.add(pxo.Property.LINEAR_SQUARE)
307            klass = pxo.Operator._infer_operator_type(properties)
308        return klass
309
310    @pxrt.enforce_precision(i="arr")
311    def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
312        # If `x` is a vector, then:
313        #     (A \kr B)(x) = vec(B * diag(x) * A.T)
314        sh_prefix = arr.shape[:-1]
315        sh_dim = len(sh_prefix)
316        xp = pxu.get_array_module(arr)
317        I = xp.eye(N=_.dim, dtype=arr.dtype)  # noqa: E741
318
319        x = arr.reshape((*sh_prefix, 1, _.dim))  # (..., 1, dim)
320        y = _._B.apply(x * I)  # (..., dim, B.codim)
321        z = y.transpose((*range(sh_dim), -1, -2))  # (..., B.codim, dim)
322        t = _._A.apply(z)  # (..., B.codim, A.codim)
323        u = t.transpose((*range(sh_dim), -1, -2))  # (..., A.codim, B.codim)
324
325        out = u.reshape((*sh_prefix, -1))  # (..., A.codim * B.codim)
326        return out
327
328    @pxrt.enforce_precision(i="arr")
329    def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
330        # If `x` is a vector, then:
331        #     (A \kr B).H(x) = diag(B.H * mat(x) * A.conj)
332        sh_prefix = arr.shape[:-1]
333        sh_dim = len(sh_prefix)
334        xp = pxu.get_array_module(arr)
335        I = xp.eye(N=_.dim, dtype=arr.dtype)  # noqa: E741
336
337        x = arr.reshape((*sh_prefix, _._A.codim, _._B.codim))  # (..., A.codim, B.codim)
338        y = _._B.adjoint(x)  # (..., A.codim, B.dim)
339        z = y.transpose((*range(sh_dim), -1, -2))  # (..., dim, A.codim)
340        t = pxu.copy_if_unsafe(_._A.adjoint(z))  # (..., dim, dim)
341        t *= I
342
343        out = t.sum(axis=-1)  # (..., dim)
344        return out
345
346    def op_asarray(_, **kwargs) -> pxt.NDArray:
347        # (A \kr B).asarray()[:,i] = A.asarray()[:,i] \kron B.asarray()[:,i]
348        A = _._A.asarray(**kwargs).T.reshape((_.dim, _._A.codim, 1))
349        B = _._B.asarray(**kwargs).T.reshape((_.dim, 1, _._B.codim))
350        C = (A * B).reshape((_.dim, -1)).T
351        return C
352
353    @pxrt.enforce_precision()
354    def op_lipschitz(_, **kwargs) -> pxt.Real:
355        if kwargs.get("tight", False):
356            _._lipschitz = _.__class__.lipschitz(_, **kwargs)
357        else:
358            op = kron(_._A, _._B)
359            _._lipschitz = op.lipschitz(**kwargs)
360        return _._lipschitz
361
362    _A = A.squeeze()
363    _B = B.squeeze()
364    assert (klass := _infer_op_klass(_A, _B)).has(pxo.Property.LINEAR)
365
366    op = pxsrc.from_source(
367        cls=klass,
368        shape=_infer_op_shape(_A.shape, _B.shape),
369        embed=dict(
370            _name="khatri_rao",
371            _A=_A,
372            _B=_B,
373        ),
374        apply=op_apply,
375        adjoint=op_adjoint,
376        asarray=op_asarray,
377        _expr=lambda _: (_._name, _._A, _._B),
378    )
379
380    # kr(A,B) = kron(A,B) + sub-sampling -> upper-bound provided by kron(A,B).lipschitz
381    op.lipschitz = kron(_A, _B).lipschitz
382    return op
