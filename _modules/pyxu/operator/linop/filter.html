  74def MovingAverage(
  75    arg_shape: pxt.NDArrayShape,
  76    size: typ.Union[typ.Tuple, pxt.Integer],
  77    center: typ.Optional[IndexSpec] = None,
  78    mode: ModeSpec = "constant",
  79    gpu: bool = False,
  80    dtype: typ.Optional[pxt.DType] = None,
  81):
  82    r"""
  83    Multidimensional moving average or uniform filter.
  84
  85    Notes
  86    -----
  87    This operator performs a convolution between the input :math:`D`-dimensional NDArray :math:`\mathbf{x} \in
  88    \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` and a uniform :math:`D`-dimensional filter :math:`\mathbf{h} \in
  89    \mathbb{R}^{\text{size} \times \cdots \times \text{size}}` that computes the `size`-point local mean values using
  90    separable kernels for improved performance.
  91
  92    .. math::
  93
  94       y_{i} = \frac{1}{|\mathcal{N}_{i}|}\sum_{j \in \mathcal{N}_{i}} x_{j}
  95
  96    Where :math:`\mathcal{N}_{i}` is the set of elements neighbouring the :math:`i`-th element of the input array, and
  97    :math:`\mathcal{N}_{i}` denotes its cardinality, i.e. the total number of neighbors.
  98
  99
 100    Parameters
 101    ----------
 102    arg_shape: NDArrayShape
 103        Shape of the input array.
 104    size: int, tuple
 105        Size of the moving average kernel.
 106
 107        If a single integer value is provided, then the moving average filter will have as many dimensions as the input
 108        array.
 109        If a tuple is provided, it should contain as many elements as `arg_shape`.
 110        For example, the ``size=(1, 3)`` will convolve the input image with the filter ``[[1, 1, 1]] / 3``.
 111
 112    center: IndexSpec
 113        (i_1, ..., i_D) index of the kernel's center.
 114
 115        `center` defines how a kernel is overlaid on inputs to produce outputs.
 116        For odd `size`, it defaults to the central element (``center=size//2``).
 117        For even `size` the desired center indices must be provided.
 118
 119    mode: str, list[str]
 120        Boundary conditions.
 121        Multiple forms are accepted:
 122
 123        * str: unique mode shared amongst dimensions.
 124          Must be one of:
 125
 126          * 'constant' (default): zero-padding
 127          * 'wrap'
 128          * 'reflect'
 129          * 'symmetric'
 130          * 'edge'
 131        * tuple[str, ...]: the `d`-th dimension uses `mode[d]` as boundary condition.
 132
 133        (See :py:func:`numpy.pad` for details.)
 134    gpu: bool
 135        Input NDArray type (`True` for GPU, `False` for CPU). Defaults to `False`.
 136    dtype: DType
 137        Working precision of the linear operator.
 138
 139    Returns
 140    -------
 141    op: OpT
 142        MovingAverage
 143
 144    Example
 145    -------
 146
 147    .. plot::
 148
 149       import matplotlib.pyplot as plt
 150       from pyxu.operator import MovingAverage
 151
 152       arg_shape = (11, 11)
 153       image = np.zeros(arg_shape)
 154       image[5, 5] = 1.
 155
 156       ma = MovingAverage(arg_shape, size=5)
 157       out = ma(image.ravel())
 158       plt.figure(figsize=(10, 5))
 159       plt.subplot(121)
 160       plt.imshow(image)
 161       plt.colorbar()
 162       plt.subplot(122)
 163       plt.imshow(out.reshape(*arg_shape))
 164       plt.colorbar()
 165
 166    See Also
 167    --------
 168    :py:class:`~pyxu.operator.linop.filter.Gaussian`
 169    """
 170    size = _to_canonical_form(size, arg_shape)
 171    if center is None:
 172        assert all([s % 2 == 1 for s in size]), (
 173            "Can only infer center for odd `size`s. For even `size`s, please " "provide the desired `center`s."
 174        )
 175        center = [s // 2 for s in size]
 176
 177    ndim, dtype, xp = _sanitize_inputs(arg_shape, dtype, gpu)
 178
 179    kernel = [xp.ones(s, dtype=dtype) for s in size]  # use separable filters
 180    scale = 1 / np.prod(size)
 181
 182    op = scale * pxls.Stencil(arg_shape=arg_shape, kernel=kernel, center=center, mode=mode)
 183    op._name = "MovingAverage"
 184    return op

 187def Gaussian(
 188    arg_shape: pxt.NDArrayShape,
 189    sigma: typ.Union[typ.Tuple[pxt.Real], pxt.Real] = 1.0,
 190    truncate: typ.Union[typ.Tuple[pxt.Real], pxt.Real] = 3.0,
 191    order: typ.Union[typ.Tuple[pxt.Integer], pxt.Integer] = 0,
 192    mode: ModeSpec = "constant",
 193    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 194    gpu: bool = False,
 195    dtype: typ.Optional[pxt.DType] = None,
 196):
 197    r"""
 198    Multidimensional Gaussian filter.
 199
 200    Notes
 201    -----
 202    This operator performs a convolution between the input :math:`D`-dimensional NDArray :math:`\mathbf{x} \in
 203    \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` and a Gaussian :math:`D`-dimensional filter :math:`\mathbf{h} \in
 204    \mathbb{R}^{\text{size} \times \cdots \times \text{size}}` using separable kernels for improved performance.
 205
 206    .. math::
 207
 208       y_{i} = \sum_{j \in \mathcal{N}_{i}} a_{j} x_{j} \exp(\frac{d_{ij}^{2}}{\sigma^{2}})
 209
 210    Where :math:`\mathcal{N}_{i}` is the set of elements neighbouring the :math:`i`-th element of the input array, and
 211    :math:`a_{j} = \sum_{j \in \mathcal{N}_{i}} a_{j} \exp(\frac{d_{ij}^{2}}{\sigma^{2}})` normalizes the kernel to sum
 212    to one.
 213
 214
 215    Parameters
 216    ----------
 217    arg_shape: NDArrayShape
 218        Shape of the input array.
 219    sigma: float, tuple
 220        Standard deviation of the Gaussian kernel.
 221
 222        If a scalar value is provided, then the Gaussian filter will have as many dimensions as the input array.
 223        If a tuple is provided, it should contain as many elements as `arg_shape`.
 224        Use ``0`` to prevent filtering in a given dimension.
 225        For example, the ``sigma=(0, 3)`` will convolve the input image in its last dimension.
 226    truncate: float, tuple
 227        Truncate the filter at this many standard deviations.
 228        Defaults to 3.0.
 229    order: int, tuple
 230        Gaussian derivative order.
 231
 232        Use ``0`` for the standard Gaussian kernel.
 233    mode: str, list[str]
 234        Boundary conditions.
 235        Multiple forms are accepted:
 236
 237        * str: unique mode shared amongst dimensions.
 238          Must be one of:
 239
 240          * 'constant' (default): zero-padding
 241          * 'wrap'
 242          * 'reflect'
 243          * 'symmetric'
 244          * 'edge'
 245        * tuple[str, ...]: the `d`-th dimension uses `mode[d]` as boundary condition.
 246
 247        (See :py:func:`numpy.pad` for details.)
 248    sampling: Real, list[Real]
 249        Sampling step (i.e. distance between two consecutive elements of an array).
 250        Defaults to 1.
 251    gpu: bool
 252        Input NDArray type (`True` for GPU, `False` for CPU).
 253        Defaults to `False`.
 254    dtype: DType
 255        Working precision of the linear operator.
 256
 257    Returns
 258    -------
 259    op: OpT
 260        Gaussian
 261
 262    Example
 263    -------
 264
 265    .. plot::
 266
 267       import matplotlib.pyplot as plt
 268       from pyxu.operator import Gaussian
 269
 270       arg_shape = (11, 11)
 271       image = np.zeros(arg_shape)
 272       image[5, 5] = 1.
 273
 274       gaussian = Gaussian(arg_shape, sigma=3)
 275       out = gaussian(image.ravel())
 276       plt.figure(figsize=(10, 5))
 277       plt.subplot(121)
 278       plt.imshow(image)
 279       plt.colorbar()
 280       plt.subplot(122)
 281       plt.imshow(out.reshape(*arg_shape))
 282       plt.colorbar()
 283
 284    See Also
 285    --------
 286    :py:class:`~pyxu.operator.linop.filter.MovingAverage`,
 287    :py:class:`~pyxu.operator.linop.filter.DifferenceOfGaussians`
 288    """
 289
 290    ndim, dtype, xp = _sanitize_inputs(arg_shape, dtype, gpu)
 291    sigma = _to_canonical_form(sigma, arg_shape)
 292    truncate = _to_canonical_form(truncate, arg_shape)
 293    order = _to_canonical_form(order, arg_shape)
 294    sampling = _to_canonical_form(sampling, arg_shape)
 295
 296    kernel = [
 297        xp.array([1], dtype=dtype),
 298    ] * len(arg_shape)
 299    center = [0 for _ in range(len(arg_shape))]
 300
 301    for i, (sigma_, truncate_, order_, sampling_) in enumerate(zip(sigma, truncate, order, sampling)):
 302        if sigma_:
 303            sigma_pix = sigma_ / sampling_
 304            radius = int(truncate_ * float(sigma_pix) + 0.5)
 305            kernel[i] = xp.asarray(np.flip(scif._gaussian_kernel1d(sigma_pix, order_, radius)), dtype=dtype)
 306            kernel[i] /= sampling_**order_
 307            center[i] = radius
 308
 309    op = pxls.Stencil(arg_shape=arg_shape, kernel=kernel, center=center, mode=mode)
 310    op._name = "Gaussian"
 311    return op

 314def DifferenceOfGaussians(
 315    arg_shape: pxt.NDArrayShape,
 316    low_sigma=1.0,
 317    high_sigma=None,
 318    low_truncate=3.0,
 319    high_truncate=3.0,
 320    mode: ModeSpec = "constant",
 321    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 322    gpu: bool = False,
 323    dtype: typ.Optional[pxt.DType] = None,
 324):
 325    r"""
 326    Multidimensional Difference of Gaussians filter.
 327
 328    Notes
 329    -----
 330
 331    This operator uses the Difference of Gaussians (DoG) method to a :math:`D`-dimensional NDArray :math:`\mathbf{x} \in
 332    \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` using separable kernels for improved performance.
 333    The DoG method blurs the input image with two Gaussian kernels with different sigma, and subtracts the more-blurred
 334    signal from the less-blurred image.
 335    This creates an output signal containing only the information from the original signal at the spatial scale
 336    indicated by the two sigmas.
 337
 338    Parameters
 339    ----------
 340    arg_shape: NDArrayShape
 341        Shape of the input array.
 342    low_sigma: float, tuple
 343        Standard deviation of the Gaussian kernel with smaller sigmas across all axes.
 344
 345        If a scalar value is provided, then the Gaussian filter will have as many dimensions as the input array.
 346        If a tuple is provided, it should contain as many elements as `arg_shape`.
 347        Use ``0`` to prevent filtering in a given dimension.
 348        For example, the ``low_sigma=(0, 3)`` will convolve the input image in its last dimension.
 349    high_sigma: float, tuple, None
 350        Standard deviation of the Gaussian kernel with larger sigmas across all axes.
 351        If ``None`` is given (default), sigmas for all axes are calculated as ``1.6 * low_sigma``.
 352    low_truncate: float, tuple
 353        Truncate the filter at this many standard deviations.
 354        Defaults to 3.0.
 355    high_truncate: float, tuple
 356        Truncate the filter at this many standard deviations.
 357        Defaults to 3.0.
 358    mode: str, list[str]
 359        Boundary conditions.
 360        Multiple forms are accepted:
 361
 362        * str: unique mode shared amongst dimensions.
 363          Must be one of:
 364
 365          * 'constant' (default): zero-padding
 366          * 'wrap'
 367          * 'reflect'
 368          * 'symmetric'
 369          * 'edge'
 370        * tuple[str, ...]: the `d`-th dimension uses `mode[d]` as boundary condition.
 371
 372        (See :py:func:`numpy.pad` for details.)
 373    sampling: Real, list[Real]
 374        Sampling step (i.e. distance between two consecutive elements of an array).
 375        Defaults to 1.
 376    gpu: bool
 377        Input NDArray type (`True` for GPU, `False` for CPU). Defaults to `False`.
 378    dtype: DType
 379        Working precision of the linear operator.
 380
 381    Returns
 382    -------
 383    op: OpT
 384        DifferenceOfGaussians
 385
 386    Example
 387    -------
 388
 389    .. plot::
 390
 391       import matplotlib.pyplot as plt
 392       from pyxu.operator import DoG
 393
 394       arg_shape = (11, 11)
 395       image = np.zeros(arg_shape)
 396       image[5, 5] = 1.
 397
 398       dog = DoG(arg_shape, low_sigma=3)
 399       out = dog(image.ravel())
 400       plt.figure(figsize=(10, 5))
 401       plt.subplot(121)
 402       plt.imshow(image)
 403       plt.colorbar()
 404       plt.subplot(122)
 405       plt.imshow(out.reshape(*arg_shape))
 406       plt.colorbar()
 407
 408    See Also
 409    --------
 410    :py:class:`~pyxu.operator.linop.filter.Gaussian`,
 411    :py:class:`~pyxu.operator.linop.filter.Sobel`,
 412    :py:class:`~pyxu.operator.linop.filter.Prewitt`,
 413    :py:class:`~pyxu.operator.linop.filter.Scharr`,
 414    :py:class:`~pyxu.operator.linop.filter.StructureTensor`
 415    """
 416
 417    low_sigma = _to_canonical_form(low_sigma, arg_shape)
 418    if high_sigma is None:
 419        high_sigma = tuple(s * 1.6 for s in low_sigma)
 420
 421    high_sigma = _to_canonical_form(high_sigma, arg_shape)
 422    low_truncate = _to_canonical_form(low_truncate, arg_shape)
 423    high_truncate = _to_canonical_form(high_truncate, arg_shape)
 424
 425    kwargs = {
 426        "arg_shape": arg_shape,
 427        "order": 0,
 428        "mode": mode,
 429        "gpu": gpu,
 430        "dtype": dtype,
 431        "sampling": sampling,
 432    }
 433    op_low = Gaussian(sigma=low_sigma, truncate=low_truncate, **kwargs)
 434    op_high = Gaussian(sigma=high_sigma, truncate=high_truncate, **kwargs)
 435    op = op_low - op_high
 436    op._name = "DifferenceOfGaussians"
 437    return op

 443def Laplace(
 444    arg_shape: pxt.NDArrayShape,
 445    mode: ModeSpec = "constant",
 446    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 447    gpu: bool = False,
 448    dtype: typ.Optional[pxt.DType] = None,
 449):
 450    r"""
 451    Multidimensional Laplace filter based on second derivatives approximated via finite differences.
 452
 453    Notes
 454    -----
 455
 456    This operator uses the applies the Laplace kernel :math:`[1 -2 1]` to a :math:`D`-dimensional NDArray
 457    :math:`\mathbf{x} \in \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` using separable kernels for improved
 458    performance.
 459    The Laplace filter is commonly used to find high-frequency components in the signal, such as for example, the edges
 460    in an image.
 461
 462    Parameters
 463    ----------
 464    arg_shape: NDArrayShape
 465        Shape of the input array.
 466    mode: str, list[str]
 467        Boundary conditions.
 468        Multiple forms are accepted:
 469
 470        * str: unique mode shared amongst dimensions.
 471          Must be one of:
 472
 473          * 'constant' (default): zero-padding
 474          * 'wrap'
 475          * 'reflect'
 476          * 'symmetric'
 477          * 'edge'
 478        * tuple[str, ...]: the `d`-th dimension uses `mode[d]` as boundary condition.
 479
 480        (See :py:func:`numpy.pad` for details.)
 481    sampling: Real, list[Real]
 482            Sampling step (i.e. distance between two consecutive elements of an array).
 483            Defaults to 1.
 484    gpu: bool
 485        Input NDArray type (`True` for GPU, `False` for CPU). Defaults to `False`.
 486    dtype: DType
 487        Working precision of the linear operator.
 488
 489    Returns
 490    -------
 491    op: OpT
 492        DifferenceOfGaussians
 493
 494    Example
 495    -------
 496
 497    .. plot::
 498
 499       import matplotlib.pyplot as plt
 500       from pyxu.operator import Laplace
 501
 502       arg_shape = (11, 11)
 503       image = np.zeros(arg_shape)
 504       image[5, 5] = 1.
 505
 506       laplace = Laplace(arg_shape)
 507       out = laplace(image.ravel())
 508       plt.figure(figsize=(10, 5))
 509       plt.subplot(121)
 510       plt.imshow(image)
 511       plt.colorbar()
 512       plt.subplot(122)
 513       plt.imshow(out.reshape(*arg_shape))
 514       plt.colorbar()
 515
 516    See Also
 517    --------
 518    :py:class:`~pyxu.operator.linop.filter.Sobel`,
 519    :py:class:`~pyxu.operator.linop.filter.Prewitt`,
 520    :py:class:`~pyxu.operator.linop.filter.Scharr`,
 521    """
 522
 523    ndim, dtype, xp = _sanitize_inputs(arg_shape, dtype, gpu)
 524    sampling = _to_canonical_form(sampling, arg_shape)
 525    centers = [[1 if i == dim else 0 for i in range(ndim)] for dim in range(ndim)]
 526    kernels = [
 527        xp.array([1.0, -2.0, 1.0]).reshape([-1 if i == dim else 1 for i in range(ndim)]) / sampling[dim]
 528        for dim in range(ndim)
 529    ]
 530    ops = [pxls.Stencil(arg_shape=arg_shape, kernel=k, center=c, mode=mode) for (k, c) in zip(kernels, centers)]
 531    op = functools.reduce(lambda x, y: x + y, ops)
 532    op._name = "Laplace"
 533    return op

 585            Defaults to 1.
 586    gpu: bool
 587        Input NDArray type (`True` for GPU, `False` for CPU). Defaults to `False`.

 589        Working precision of the linear operator.

 594        Sobel
 595

 618    See Also

 636def Prewitt(
 637    arg_shape: pxt.NDArrayShape,
 638    axis: typ.Optional[typ.Tuple] = None,
 639    mode: ModeSpec = "constant",
 640    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 641    gpu: bool = False,
 642    dtype: typ.Optional[pxt.DType] = None,
 643):
 644    r"""
 645    Multidimensional Prewitt filter.
 646
 647    Notes
 648    -----
 649
 650    This operator uses the applies the multi-dimensional Prewitt filter to a :math:`D`-dimensional NDArray
 651    :math:`\mathbf{x} \in \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` using separable kernels for improved
 652    performance.  The Prewitt filter applies the following edge filter in the dimensions of interest:
 653    ``[1, 0, -1]``, and the smoothing filter on the rest of dimensions: ``[1, 1, 1] / 3``.
 654    The Prewitt filter is commonly used to find high-frequency components in the signal, such as for example, the edges
 655    in an image.
 656
 657    Parameters
 658    ----------
 659    arg_shape: NDArrayShape
 660        Shape of the input array.
 661    axis: int, tuple
 662        Compute the edge filter along this axis. If not provided, the edge magnitude is computed. This is defined as:
 663

 833def _EdgeFilter(
 834    arg_shape: pxt.NDArrayShape,
 835    smooth_kernel: KernelSpec,
 836    filter_name: str,
 837    axis: typ.Optional[typ.Tuple] = None,
 838    mode: ModeSpec = "constant",
 839    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 840    gpu: bool = False,
 841    dtype: typ.Optional[pxt.DType] = None,
 842):
 843    from pyxu.operator.map.ufunc import sqrt, square
 844
 845    ndim, dtype, xp = _sanitize_inputs(arg_shape, dtype, gpu)
 846    sampling = _to_canonical_form(sampling, arg_shape)
 847
 848    axes = _get_axes(axis, ndim)
 849
 850    return_magnitude = len(axes) > 1
 851
 852    op_list = []
 853    for edge_dim in axes:
 854        kernel = [xp.array(1)] * len(arg_shape)
 855        center = np.ones(len(arg_shape), dtype=int)
 856        # We define the kernel reversed compared to Scipy or Skimage because we use correlation instead of convolution
 857        kernel[edge_dim] = xp.array([-1, 0, 1], dtype=dtype) / sampling[edge_dim]
 858        smooth_axes = list(set(range(ndim)) - {edge_dim})
 859        for smooth_dim in smooth_axes:
 860            kernel[smooth_dim] = xp.asarray(smooth_kernel, dtype=dtype) / sampling[smooth_dim]
 861
 862        if return_magnitude:
 863            op_list.append(square(pxls.Stencil(arg_shape=arg_shape, kernel=kernel, center=center, mode=mode)))
 864        else:
 865            op_list.append(pxls.Stencil(arg_shape=arg_shape, kernel=kernel, center=center, mode=mode))
 866
 867    op = functools.reduce(lambda x, y: x + y, op_list)
 868    if return_magnitude:
 869        op = (1 / np.sqrt(ndim)) * sqrt(op)
 870
 871    op._name = filter_name
 872    return op

 875class StructureTensor(pxo.DiffMap):
 876    r"""
 877    Structure tensor operator.
 878
 879    Notes
 880    -----
 881    The Structure Tensor, also known as the second-order moment tensor or the inertia tensor, is a matrix derived from
 882    the gradient of a function. It describes the distribution of the gradient (i.e., its prominent directions) in a
 883    specified neighbourhood around a point, and the degree to which those directions are coherent.
 884    The structure tensor of a :math:`D`-dimensional signal
 885    :math:`\mathbf{f} \in \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` can be written as:
 886
 887    .. math::
 888
 889        \mathbf{S}_\sigma \mathbf{f} = \mathbf{g}_{\sigma} * \nabla\mathbf{f} (\nabla\mathbf{f})^{\top} = \mathbf{g}_{\sigma} *
 890        \begin{bmatrix}
 891        \left( \dfrac{ \partial\mathbf{f} }{ \partial x_{0} } \right)^2 &  \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{0}\,\partial x_{1} } & \cdots & \dfrac{ \partial\mathbf{f} }{ \partial x_{0} } \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1} } \\
 892        \dfrac{ \partial\mathbf{f} }{ \partial x_{1} } \dfrac{ \partial\mathbf{f} }{ \partial x_{0} } & \left( \dfrac{ \partial\mathbf{f} }{ \partial x_{1} }\right)^2 & \cdots & \dfrac{ \partial\mathbf{f} }{ \partial x_{1} } \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1} } \\
 893        \vdots & \vdots & \ddots & \vdots \\
 894        \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1} } \dfrac{ \partial\mathbf{f} }{ \partial x_{0} } & \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1} } \dfrac{ \partial\mathbf{f} }{ \partial x_{1} } & \cdots & \left( \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1}} \right)^2
 895        \end{bmatrix},
 896
 897    where :math:`\mathbf{g}_{\sigma} \in \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` is a discrete Gaussian filter
 898    with standard variation :math:`\sigma` with which a convolution is performed elementwise.
 899
 900    However, due to the symmetry of the structure tensor, only the upper triangular part is computed in practice:
 901
 902    .. math::
 903
 904        \mathbf{H}_{\mathbf{v}_1, \ldots ,\mathbf{v}_m} \mathbf{f} = \mathbf{g}_{\sigma} * \begin{bmatrix}
 905        \left( \dfrac{ \partial\mathbf{f} }{ \partial x_{0} } \right)^2 \\
 906        \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{0}\,\partial x_{1} } \\
 907        \vdots \\
 908        \left( \dfrac{ \partial\mathbf{f} }{ \partial x_{D-1}} \right)^2
 909        \end{bmatrix} \mathbf{f} \in \mathbb{R}^{\frac{D (D-1)}{2} \times N_0 \times \cdots \times N_{D-1}}
 910
 911    Remark
 912    ------
 913    In case of using the finite differences (`diff_type="fd"`), the finite difference scheme defaults to `central`
 914    (see :py:class:`~pyxu.operator.linop.diff.PartialDerivative`).
 915
 916    Example
 917    -------
 918
 919    .. plot::
 920
 921       import numpy as np
 922       import matplotlib.pyplot as plt
 923       from pyxu.operator.linop.diff import StructureTensor
 924       from pyxu.util.misc import peaks
 925
 926       # Define input image
 927       n = 1000
 928       x = np.linspace(-3, 3, n)
 929       xx, yy = np.meshgrid(x, x)
 930       image = peaks(xx, yy)
 931       nsamples = 2
 932       arg_shape = image.shape  # (1000, 1000)
 933       images = np.tile(image, (nsamples, 1, 1)).reshape(nsamples, -1)
 934       print(images.shape)  # (2, 1000000)
 935       # Instantiate structure tensor operator
 936       structure_tensor = StructureTensor(arg_shape=arg_shape)
 937
 938       outputs = structure_tensor(images)
 939       print(outputs.shape)  # (2, 3000000)
 940       # Plot
 941       outputs = structure_tensor.unravel(outputs)
 942       print(outputs.shape)  # (2, 3, 1000, 1000)
 943       plt.figure()
 944       plt.imshow(images[0].reshape(arg_shape))
 945       plt.colorbar()
 946       plt.title("Image")
 947       plt.axis("off")
 948
 949       plt.figure()
 950       plt.imshow(outputs[0][0].reshape(arg_shape))
 951       plt.colorbar()
 952       plt.title(r"$\hat{S}_{xx}$")
 953       plt.axis("off")
 954
 955       plt.figure()
 956       plt.imshow(outputs[0][1].reshape(arg_shape))
 957       plt.colorbar()
 958       plt.title(r"$\hat{S}_{xy}$")
 959       plt.axis("off")
 960
 961       plt.figure()
 962       plt.imshow(outputs[0][2].reshape(arg_shape))
 963       plt.colorbar()
 964       plt.title(r"$\hat{S}_{yy}$")
 965       plt.axis("off")
 966
 967    See Also
 968    --------
 969    :py:class:`~pyxu.operator.linop.diff.PartialDerivative`, :py:class:`~pyxu.operator.linop.diff.Gradient`,
 970    :py:class:`~pyxu.operator.linop.diff.Hessian`.
 971    """
 972
 973    def __init__(
 974        self,
 975        arg_shape: pxt.NDArrayShape,
 976        diff_method="fd",
 977        smooth_sigma: typ.Union[pxt.Real, tuple[pxt.Real, ...]] = 1.0,
 978        smooth_truncate: typ.Union[pxt.Real, tuple[pxt.Real, ...]] = 3.0,
 979        mode: ModeSpec = "constant",
 980        sampling: typ.Union[pxt.Real, tuple[pxt.Real, ...]] = 1,
 981        gpu: bool = False,
 982        dtype: typ.Optional[pxt.DType] = None,
 983        parallel: bool = False,
 984        **diff_kwargs,
 985    ):
 986        self.arg_shape = arg_shape
 987        size = int(np.prod(arg_shape))
 988        ndim = len(arg_shape)
 989        ntriu = (ndim * (ndim + 1)) // 2
 990        super().__init__(shape=(ntriu * size, size))
 991        self.directions = tuple(
 992            list(_) for _ in itertools.combinations_with_replacement(np.arange(len(arg_shape)).astype(int), 2)
 993        )
 994
 995        if diff_method == "fd":
 996            diff_kwargs.update({"scheme": diff_kwargs.pop("scheme", "central")})
 997        self.grad = pxld.Gradient(
 998            arg_shape=arg_shape,
 999            directions=None,
1000            mode=mode,
1001            gpu=gpu,
1002            dtype=dtype,
1003            sampling=sampling,
1004            parallel=parallel,
1005            **diff_kwargs,
1006        )
1007
1008        if smooth_sigma:
1009            self.smooth = Gaussian(
1010                arg_shape=arg_shape,
1011                sigma=smooth_sigma,
1012                truncate=smooth_truncate,
1013                order=0,
1014                mode=mode,
1015                sampling=sampling,
1016                gpu=gpu,
1017                dtype=dtype,
1018            )
1019        else:
1020            self.smooth = pxlb.IdentityOp(dim=np.prod(arg_shape).item())
1021
1022    def unravel(self, arr):
1023        return arr.reshape(*arr.shape[:-1], -1, *self.arg_shape)
1024
1025    def ravel(self, arr):
1026        return arr.reshape(*arr.shape[: -1 - len(self.arg_shape)], -1)
1027
1028    def apply(self, arr):
1029        xp = pxu.get_array_module(arr)
1030        sh = arr.shape[:-1]
1031        grad = self.grad.unravel(self.grad(arr))
1032
1033        def slice_(ax):
1034            return (slice(None),) * len(sh) + (slice(ax, ax + 1),)
1035
1036        return xp.concatenate(
1037            [
1038                self.grad.unravel(self.smooth((grad[slice_(i)] * grad[slice_(j)]).reshape(*sh, -1)))
1039                for i, j in self.directions
1040            ],
1041            axis=len(sh),
1042        ).reshape(*sh, -1)
