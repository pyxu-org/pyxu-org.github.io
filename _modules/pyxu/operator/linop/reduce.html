 16def Sum(
 17    arg_shape: pxt.NDArrayShape,
 18    axis: pxt.NDArrayAxis = None,
 19) -> pxt.OpT:
 20    r"""
 21    Multi-dimensional sum reduction.
 22
 23    This operator re-arranges the input array to a multidimensional array of shape ``arg_shape``, then reduces it via
 24    summation across one or more ``axis``.
 25
 26    For example, assuming the input array :math:`\mathbf{x} \in \mathbb{R}^{N_1 \times N_2 \times N_3}` and ``axis=-1``,
 27    then
 28
 29    .. math::
 30
 31       \mathbf{y}_{i,j} = \sum_{k}{\mathbf{x}_{i,j,k}}.
 32
 33    Parameters
 34    ----------
 35    arg_shape: NDArrayShape
 36        Shape of the data to be reduced.
 37    axis: NDArrayAxis
 38        Axis or axes along which a sum is performed.
 39        The default, axis=None, will sum all the elements of the input array.
 40        If axis is negative it counts from the last to the first axis.
 41
 42    Notes
 43    -----
 44    The matrix operator of a 1D reduction applied to :math:`\mathbf{x} \in \mathbb{R}^{N}` is given by
 45
 46    .. math::
 47
 48       \mathbf{A}(x) = \mathbf{1}^{T} \mathbf{x},
 49
 50    where :math:`\sigma_{\max}(\mathbf{A}) = \sqrt{N}`.
 51    An ND reduction is a chain of 1D reductions in orthogonal dimensions.
 52    Hence the Lipschitz constant of an ND reduction is the product of Lipschitz constants of all 1D reductions involved,
 53    i.e.:
 54
 55    .. math::
 56
 57       L = \sqrt{\prod_{i_{k}} N_{i_{k}}},
 58
 59    where :math:`\{i_{k}\}_{k}` denotes the axes being summed over.
 60    """
 61
 62    def as_array(obj) -> np.ndarray:
 63        if isinstance(obj, cabc.Sequence):
 64            pass
 65        else:
 66            obj = [obj]
 67        return np.array(obj, dtype=int)
 68
 69    arg_shape = as_array(arg_shape)
 70    assert np.all(arg_shape > 0)
 71    N_dim = len(arg_shape)
 72
 73    if axis is None:
 74        axis = np.arange(N_dim)
 75    axis = np.unique(as_array(axis))  # drop potential duplicates
 76    assert np.all((-N_dim <= axis) & (axis < N_dim))  # all axes in valid range
 77    axis = (axis + N_dim) % N_dim  # get rid of negative axes
 78
 79    sum_shape = arg_shape.copy()  # array shape after reduction
 80    sum_shape[axis] = 1
 81
 82    @pxrt.enforce_precision(i="arr")
 83    def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
 84        sh = arr.shape[:-1]
 85        arr = arr.reshape(sh + _._arg_shape)
 86
 87        axis = tuple(ax + len(sh) for ax in _._axis)
 88        out = arr.sum(axis=axis).reshape(*sh, -1)
 89
 90        return out
 91
 92    @pxrt.enforce_precision(i="arr")
 93    def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
 94        sh = arr.shape[:-1]
 95        arr = arr.reshape(sh + _._sum_shape)
 96
 97        xp = pxu.get_array_module(arr)
 98        out = xp.broadcast_to(arr, sh + _._arg_shape)
 99
100        out = out.reshape(*sh, -1)
101        return out
102
103    def op_estimate_lipschitz(_, **kwargs) -> pxt.Real:
104        N = np.prod(_._arg_shape) / np.prod(_._sum_shape)
105        L = np.sqrt(N)
106        return L
107
108    dim = arg_shape.prod()
109    codim = dim // arg_shape[axis].prod()
110
111    op = pxsrc.from_source(
112        cls=pxa.LinOp if codim > 1 else pxa.LinFunc,
113        shape=(codim, dim),
114        embed=dict(
115            _name="Sum",
116            _axis=tuple(axis),
117            _arg_shape=tuple(arg_shape),
118            _sum_shape=tuple(sum_shape),
119        ),
120        apply=op_apply,
121        adjoint=op_adjoint,
122        estimate_lipschitz=op_estimate_lipschitz,
123    )
124    op.lipschitz = op.estimate_lipschitz()
125    return op
