 157def _FiniteDifference(
 158    order: typ.Union[pxt.Integer, cabc.Sequence[pxt.Integer, ...]],
 159    arg_shape: pxt.NDArrayShape,
 160    scheme: typ.Union[str, cabc.Sequence[str, ...]] = "forward",
 161    axes: pxt.NDArrayAxis = None,

 163    sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 164) -> tuple[cabc.Sequence[pxt.NDArray], pxt.NDArray]:
 165    r"""
 166    Finite difference base operator along a single dimension.
 167
 168    This class is used by :py:class:`~pyxu.operator.linop.diff.PartialDerivative`,
 169    :py:class:`~pyxu.operator.linop.diff.Gradient` and :py:class:`~pyxu.operator.linop.diff.Hessian`.
 170    See :py:class:`~pyxu.operator.linop.diff.PartialDerivative.finite_difference` for documentation.
 171
 172    See Also
 173    --------
 174    :py:class:`~pyxu.operator.linop.diff._GaussianDerivative`,
 175    :py:class:`~pyxu.operator.linop.diff.PartialDerivative`,
 176    :py:class:`~pyxu.operator.linop.diff.Gradient`,
 177    :py:class:`~pyxu.operator.linop.diff.Hessian`.
 178
 179    Parameters
 180    ----------
 181    order: Integer, list[Integer]
 182        Derivative order. If a single integer value is provided, then `axes` should be provided to
 183        indicate which dimension should be differentiated.
 184        If a tuple is provided, it should contain as many elements as `arg_shape`.
 185    arg_shape: pxt.NDArrayShape
 186        Shape of the input array.
 187    scheme: str, list[str]
 188        Type of finite differences: ["forward", "backward", "central"].

 201    sampling: pxt.Real, list[Real]

 203        Defaults to 1.
 204    """
 205    diff_kwargs = {"scheme": scheme, "accuracy": accuracy}
 206    order, sampling, scheme, accuracy, axes = _sanitize_init_kwargs(
 207        order=order,
 208        diff_method="fd",
 209        diff_kwargs=diff_kwargs,
 210        arg_shape=arg_shape,
 211        axes=axes,
 212        sampling=sampling,
 213    )

 215    def _compute_ids(order: pxt.Integer, scheme: str, accuracy: pxt.Integer) -> list:
 216        """
 217        Computes the Finite difference indices according to the order, type and accuracy.
 218        """
 219        if scheme == "central":
 220            n_coefs = 2 * ((order + 1) // 2) - 1 + accuracy
 221            ids = np.arange(-(n_coefs // 2), n_coefs // 2 + 1, dtype=int)
 222        else:
 223            n_coefs = order + accuracy
 224            if scheme == "forward":
 225                ids = np.arange(0, n_coefs, dtype=int)
 226            elif scheme == "backward":
 227                ids = np.arange(-n_coefs + 1, 1, dtype=int)
 228            else:
 229                raise ValueError(
 230                    f"Incorrect value for variable 'type'. 'type' should be ['forward', 'backward', "
 231                    f"'central'], but got {scheme}."
 232                )
 233        return ids.tolist()
 234
 235    def _compute_coefficients(stencil_ids: cabc.Sequence, order: pxt.Integer, sampling: pxt.Real) -> pxt.NDArray:
 236        """
 237        Computes the finite difference coefficients based on the order and indices.
 238        """
 239        # vander doesn't allow precision specification
 240        stencil_mat = np.vander(
 241            np.array(stencil_ids),
 242            increasing=True,
 243        ).T.astype(pxrt.getPrecision().value)
 244        vec = np.zeros(len(stencil_ids), dtype=pxrt.getPrecision().value)
 245        vec[order] = math.factorial(order)
 246        coefs = np.linalg.solve(stencil_mat, vec)
 247        coefs /= sampling**order
 248        return coefs
 249
 250    # FILL COEFFICIENTS
 251    def _fill_coefs(i: pxt.Integer) -> tuple[list[pxt.NDArray], pxt.Integer]:
 252        r"""
 253        Defines kernel elements.
 254        """
 255        stencil_ids = _compute_ids(order=order[i], scheme=scheme[i], accuracy=accuracy[i])
 256        stencil_coefs = _compute_coefficients(stencil_ids=stencil_ids, order=order[i], sampling=sampling[i])
 257        center = stencil_ids.index(0)
 258        return stencil_ids, stencil_coefs, center
 259
 260    kernel, center = _create_kernel(arg_shape, axes, _fill_coefs)
 261    return kernel, center

 265    order: typ.Union[pxt.Integer, cabc.Sequence[pxt.Integer, ...]],
 266    arg_shape: pxt.NDArrayShape,
 267    sigma: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]],

 273    Gaussian derivative base operator along a single dimension.
 274
 275    This class is used by :py:class:`~pyxu.operator.linop.diff.PartialDerivative`,
 276    :py:class:`~pyxu.operator.linop.diff.Gradient` and :py:class:`~pyxu.operator.linop.diff.Hessian`.

 280    --------
 281    :py:class:`~pyxu.operator.linop.diff._BaseDifferential`,
 282    :py:class:`~pyxu.operator.linop.diff._FiniteDifference`,
 283    :py:class:`~pyxu.operator.linop.diff.PartialDerivative`,
 284    :py:class:`~pyxu.operator.linop.diff.Gradient`,
 285    :py:class:`~pyxu.operator.linop.diff.Hessian`.
 286
 287    Parameters
 288    ----------
 289    order: Integer, list[Integer]
 290        Derivative order.
 291        If a single integer value is provided, then `axes` should be provided to indicate which
 292        dimension should be used for differentiation.
 293        If a tuple is provided, it should contain as many elements as number of dimensions in
 294        `axes`.
 295    arg_shape: pxt.NDArrayShape
 296        Shape of the input array.
 297    sigma: Real, list[Real]
 298        Standard deviation of the Gaussian kernel.
 299    axes: pxt.NDArrayAxis

 305        Truncate the filter at this many standard deviations (at each side from the origin).

 308        Sampling step (i.e., the distance between two consecutive elements of an array).
 309        Defaults to 1.

 312    order, sampling, sigma, truncate, axes = _sanitize_init_kwargs(
 313        order=order,
 314        diff_method="gd",
 315        diff_kwargs=diff_kwargs,
 316        arg_shape=arg_shape,
 317        axes=axes,
 318        sampling=sampling,

 321    def _fill_coefs(i: pxt.Integer) -> tuple[cabc.Sequence[pxt.Integer], pxt.NDArray, pxt.Integer]:
 322        r"""
 323        Defines kernel elements.
 324        """
 325        # make the radius of the filter equal to `truncate` standard deviations
 326        sigma_pix = sigma[i] / sampling[i]  # Sigma rescaled to pixel units
 327        radius = int(truncate[i] * float(sigma_pix) + 0.5)
 328        stencil_coefs = _gaussian_kernel1d(sigma=sigma_pix, order=order[i], sampling=sampling[i], radius=radius)
 329        stencil_ids = [i for i in range(-radius, radius + 1)]
 330        return stencil_ids, stencil_coefs, radius
 331
 332    def _gaussian_kernel1d(
 333        sigma: pxt.Real,
 334        order: pxt.Integer,
 335        sampling: pxt.Real,
 336        radius: pxt.Integer,
 337    ) -> pxt.NDArray:
 338        """
 339        Computes a 1-D Gaussian convolution kernel.
 340        Wraps scipy.ndimage.filters._gaussian_kernel1d
 341        It flips the output because the original kernel is meant for convolution instead of correlation.
 342        """
 343        coefs = np.flip(scif._gaussian_kernel1d(sigma, order, radius))
 344        coefs /= sampling**order
 345        return coefs
 346
 347    kernel, center = _create_kernel(arg_shape, axes, _fill_coefs)
 348    return kernel, center

 351def _PartialDerivative(
 352    kernel: KernelSpec,
 353    center: KernelSpec,
 354    arg_shape: pxt.NDArrayShape,
 355    mode: ModeSpec = "constant",
 356    gpu: bool = False,
 357    dtype: typ.Optional[pxt.DType] = None,
 358    meta: typ.Optional[typ.NamedTuple] = None,
 359) -> pxt.OpT:
 360    r"""
 361    Helper base class for partial derivative operator based on Numba stencils (see
 362    https://numba.pydata.org/numba-doc/latest/user/stencil.html).
 363
 364    See Also
 365    --------
 366    :py:class:`~pyxu.operator.linop.diff.PartialDerivative`,
 367    :py:class:`~pyxu.operator.linop.stencil.stencil.Stencil`,
 368    :py:func:`~pyxu.math.stencil.make_nd_stencil`,
 369    :py:class:`~pyxu.operator.linop.diff._FiniteDifference`,
 370    :py:class:`~pyxu.operator.linop.diff._GaussianDerivative`.
 371
 372    Parameters
 373    ----------
 374    kernel: KernelSpec
 375        Stencil coefficients.
 376        Two forms are accepted:
 377
 378            * NDArray of rank-:math:`D`: denotes a non-seperable stencil.
 379            * tuple[NDArray_1, ..., NDArray_D]: a sequence of 1D stencils such that is filtered by
 380              the stencil `kernel[d]` along the :math:`d`-th dimension.
 381    center: IndexSpec
 382        (i_1, ..., i_D) index of the stencil's center.
 383
 384        `center` defines how a kernel is overlaid on inputs to produce outputs.
 385
 386        .. math::
 387
 388           y[i_{1},\ldots,i_{D}]
 389           =
 390           \sum_{q_{1},\ldots,q_{D}=0}^{k_{1},\ldots,k_{D}}
 391           x[i_{1} - c_{1} + q_{1},\ldots,i_{D} - c_{D} + q_{D}]
 392           \,\cdot\,
 393           k[q_{1},\ldots,q_{D}]
 394    arg_shape: pxt.NDArrayShape
 395        Shape of the input array.
 396    mode: str, list[str]
 397        Boundary conditions.
 398        Multiple forms are accepted:
 399
 400        * str: unique mode shared amongst dimensions.
 401          Must be one of:
 402
 403          * 'constant' (default): zero-padding
 404          * 'wrap'
 405          * 'reflect'
 406          * 'symmetric'
 407          * 'edge'
 408        * tuple[str, ...]: the `d`-th dimension uses ``mode[d]`` as boundary condition.
 409
 410        (See :py:func:`numpy.pad` for details.)
 411    gpu: bool
 412        Input NDArray type (`True` for GPU, `False` for CPU). Defaults to `False`.
 413    dtype: DType
 414        Working precision of the linear operator.
 415    meta: typ.NamedTuple
 416        Partial derivative metadata describing:
 417
 418        * Diff_method
 419        * Sampling
 420        * Scheme (for finite differences)
 421        * Accuracy (for finite differences)
 422        * Sigma (for Gaussian derivative)
 423        * Truncate (for Gaussian derivative)
 424    """
 425    if dtype is None:
 426        dtype = pxrt.getPrecision().value
 427
 428    if gpu:
 429        assert pxd.CUPY_ENABLED
 430        import cupy as xp
 431    else:
 432        import numpy as xp
 433
 434    if isinstance(kernel, cabc.MutableSequence):
 435        for i in range(len(kernel)):
 436            kernel[i] = xp.array(kernel[i], dtype=dtype)
 437    else:
 438        kernel = xp.array(kernel, dtype=dtype)
 439
 440    op = pxls.Stencil(arg_shape=arg_shape, kernel=kernel, center=center, mode=mode)
 441    setattr(op, "meta", meta)
 442
 443    return op

 446class PartialDerivative:
 447    r"""
 448    Partial derivative operator based on `Numba stencils
 449    <https://numba.pydata.org/numba-doc/latest/user/stencil.html>`_.
 450
 451    Notes
 452    -----
 453    * This operator approximates the partial derivative of a :math:`D`-dimensional signal :math:`\mathbf{f} \in
 454      \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}`
 455
 456      .. math::
 457
 458         \frac{\partial^{n} \mathbf{f}}{\partial x_0^{n_0} \cdots \partial x_{D-1}^{n_{D-1}}} \in
 459         \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}
 460
 461      where :math:`\frac{\partial^{n_i}}{\partial x_i^{n_i}}` is the :math:`n_i`-th order partial derivative along
 462      dimension :math:`i` and :math:`n = \prod_{i=0}^{D-1} n_{i}` is the total derivative order.
 463
 464      Partial derivatives can be implemented with `finite differences
 465      <https://en.wikipedia.org/wiki/Finite_difference>`_ via the
 466      :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.finite_difference` constructor, or with the `Gaussian
 467      derivative <https://www.crisluengo.net/archives/22/>`_ via the
 468      :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.gaussian_derivative` constructor.
 469
 470    * When using the :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.finite_difference` constructor, the adjoint
 471      of the resulting linear operator will vary depending on the type of finite differences:
 472
 473      * For ``forward`` type, the adjoint corresponds to:
 474
 475        :math:`(\frac{\partial^{\text{fwd}}}{\partial x})^{\ast} = -\frac{\partial^{\text{bwd}}}{\partial x}`
 476
 477      * For ``backward`` type, the adjoint corresponds to:
 478
 479        :math:`(\frac{\partial^{\text{bwd}}}{\partial x})^{\ast} = -\frac{\partial^{\text{fwd}}}{\partial x}`
 480
 481      * For ``central`` type, and for the :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.gaussian_derivative`
 482        constructor, the adjoint corresponds to:
 483
 484        :math:`(\frac{\partial}{\partial x})^{\ast} = -\frac{\partial}{\partial x}`
 485
 486
 487    .. warning::
 488
 489       When dealing with high-order partial derivatives, the stencils required to compute them can become large,
 490       resulting in computationally expensive evaluations.
 491       In such scenarios, it can be more efficient to construct the partial derivative through a composition of
 492       lower-order partial derivatives.
 493
 494    See Also
 495    --------
 496    :py:class:`~pyxu.operator.linop.diff.Gradient`,
 497    :py:class:`~pyxu.operator.linop.diff.Laplacian`,
 498    :py:class:`~pyxu.operator.linop.diff.Hessian`.
 499    """

 505        scheme: typ.Union[str, cabc.Sequence[str, ...]] = "forward",
 506        accuracy: typ.Union[pxt.Integer, cabc.Sequence[pxt.Integer, ...]] = 1,
 507        mode: ModeSpec = "constant",
 508        gpu: bool = False,

 510        sampling: typ.Union[pxt.Real, cabc.Sequence[pxt.Real, ...]] = 1,
 511    ) -> pxt.OpT:
 512        r"""
 513        Compute partial derivatives for multi-dimensional signals using finite differences.
 514
 515        Parameters
 516        ----------
 517        arg_shape: NDArrayShape
 518            Shape of the input array.
 519        order: list[Integer]
 520            Derivative order for each dimension.
 521            The total order of the partial derivative is the sum of the elements of the tuple.
 522            Use zeros to indicate dimensions in which the derivative should not be computed.
 523        scheme: str, list[str]
 524            Type of finite differences: ['forward, 'backward, 'central'].
 525            Defaults to 'forward'.
 526            If a string is provided, the same `scheme` is assumed for all dimensions.
 527            If a tuple is provided, it should have as many elements as `order`.
 528        accuracy: Integer, list[Integer]
 529            Determines the number of points used to approximate the derivative with finite differences (see `Notes`).
 530            Defaults to 1.
 531            If an int is provided, the same `accuracy` is assumed for all dimensions.
 532            If a tuple is provided, it should have as many elements as `arg_shape`.
 533        mode: str, list[str]
 534            Boundary conditions.
 535            Multiple forms are accepted:
 536
 537            * str: unique mode shared amongst dimensions.
 538              Must be one of:
 539
 540              * 'constant' (default): zero-padding
 541              * 'wrap'
 542              * 'reflect'
 543              * 'symmetric'
 544              * 'edge'
 545            * tuple[str, ...]: the `d`-th dimension uses ``mode[d]`` as boundary condition.
 546
 547            (See :py:func:`numpy.pad` for details.)
 548        gpu: bool
 549            Input NDArray type (`True` for GPU, `False` for CPU).
 550            Defaults to `False`.
 551        dtype: DType
 552            Working precision of the linear operator.
 553        sampling: Real, list[Real]
 554            Sampling step (i.e. distance between two consecutive elements of an array).
 555            Defaults to 1.
 556
 557        Returns
 558        -------
 559        op: OpT
 560            Partial derivative
 561
 562        Notes
 563        -----
 564        We explain here finite differences for one-dimensional signals; this operator performs finite differences for
 565        multi-dimensional signals along dimensions specified by `order`.
 566
 567        This operator approximates derivatives with `finite differences
 568        <https://en.wikipedia.org/wiki/Finite_difference>`_.
 569        It is inspired by the `Finite Difference Coefficients Calculator
 570        <https://web.media.mit.edu/~crtaylor/calculator.html>`_ to construct finite-difference approximations for the
 571        desired *(i)* derivative order, *(ii)* approximation accuracy, and *(iii)* finite difference type.
 572        Three basic types of finite differences are supported, which lead to the following first-order (``order = 1``)
 573        operators with ``accuracy = 1`` and sampling step ``sampling = h`` for one-dimensional signals:
 574
 575        - **Forward difference**:
 576          Approximates the continuous operator :math:`D_{F}f(x) = \frac{f(x+h) - f(x)}{h}` with the discrete operator
 577
 578          .. math::
 579
 580             \mathbf{D}_{F} f [n] = \frac{f[n+1] - f[n]}{h},
 581
 582          whose kernel is :math:`d = \frac{1}{h}[-1, 1]` and center is (0, ).
 583
 584        - **Backward difference**:
 585          Approximates the continuous operator :math:`D_{B}f(x) = \frac{f(x) - f(x-h)}{h}` with the discrete operator
 586
 587          .. math::
 588
 589             \mathbf{D}_{B} f [n] = \frac{f[n] - f[n-1]}{h},
 590
 591          whose kernel is :math:`d = \frac{1}{h}[-1, 1]` and center is (1, ).
 592
 593        - **Central difference**:
 594          Approximates the continuous operator :math:`D_{C}f(x) = \frac{f(x+h) - f(x-h)}{2h}` with the discrete operator
 595
 596          .. math::
 597
 598             \mathbf{D}_{C} f [n] = \frac{f[n+1] - f[n-1]}{2h},
 599
 600          whose kernel is :math:`d = \frac{1}{h}[-\frac12, 0, \frac12]` and center is (1, ).
 601
 602        .. warning::
 603
 604           For forward and backward differences, higher-order operators correspond to the composition of first-order
 605           operators.
 606           This is not the case for central differences: the second-order continuous operator is given by
 607           :math:`D^2_{C}f(x) = \frac{f(x+h) - 2 f(x) + f(x-h)}{h}`, hence :math:`D^2_{C} \neq D_{C} \circ D_{C}`.
 608           The corresponding discrete operator is given by :math:`\mathbf{D}^2_{C} f [n] = \frac{f[n+1] - 2 f[n] +
 609           f[n-1]}{h}`, whose kernel is :math:`d = \frac{1}{h}[1, -2, 1]` and center is (1, ).
 610           We refer to `this paper
 611           <https://www.ams.org/journals/mcom/1988-51-184/S0025-5718-1988-0935077-0/S0025-5718-1988-0935077-0.pdf>`_ for
 612           more details.
 613
 614        For a given derivative order :math:`N\in\mathbb{Z}^{+}` and accuracy :math:`a\in\mathbb{Z}^{+}`, the size
 615        :math:`N_s` of the stencil kernel :math:`d` used for finite differences is given by:
 616
 617        - For central differences: :math:`N_s = 2 \lfloor\frac{N + 1}{2}\rfloor - 1 + a`
 618
 619        - For forward and backward differences: :math:`N_s = N + a`
 620
 621        For :math:`N_s` given support indices :math:`\{s_1, \ldots , s_{N_s} \} \subset \mathbb{Z}` and a derivative
 622        order :math:`N<N_s`, the stencil kernel :math:`d = [d_1, \ldots, d_{N_s}]` of the finite-difference
 623        approximation of the derivative is obtained by solving the following system of linear equations (see the `Finite
 624        Difference Coefficients Calculator <https://web.media.mit.edu/~crtaylor/calculator.html>`_ documentation):
 625
 626        .. admonition:: Remark
 627
 628           The number of coefficients of the finite-difference kernel is chosen to guarantee the requested accuracy, but
 629           might be larger than requested accuracy.
 630           For example, if choosing `scheme='central'` with `accuracy=1`, it will create a kernel corresponding to
 631           `accuracy=2`, as it is the minimum accuracy possible for such scheme (see the `finite difference coefficient
 632           table <https://en.wikipedia.org/wiki/Finite_difference_coefficient>`_).
 633
 634           .. math::
 635
 636              \left(\begin{array}{ccc}
 637              s_{1}^{0} & \cdots & s_{N_s}^{0} \\
 638              \vdots & \ddots & \vdots \\
 639              s_{1}^{N_s-1} & \cdots & s_{N_s}^{N_s-1}
 640              \end{array}\right)\left(\begin{array}{c}
 641              d_{1} \\
 642              \vdots \\
 643              d_{N_s}
 644              \end{array}\right)= \frac{1}{h^{N}}\left(\begin{array}{c}
 645              \delta_{0, N} \\
 646              \vdots \\
 647              \delta_{i, N} \\
 648              \vdots \\
 649              \delta_{N_s-1, N}
 650              \end{array}\right),
 651
 652           where :math:`\delta_{i, j}` is the Kronecker delta.
 653
 654        This class inherits its methods from :py:class:`~pyxu.operator.linop.stencil.stencil.Stencil`.
 655

 679           # Compute derivatives

 754        Compute partial derivatives for multi-dimensional signals using gaussian derivatives.

 756        Parameters
 757        ----------
 758        arg_shape: NDArrayShape

 760        order: list[Integer]

 798        Returns

 800        op: OpT
 801            Partial derivative.

 803        Notes
 804        -----
 805        We explain here Gaussian derivatives for one-dimensional signals; this operator performs partial Gaussian
 806        derivatives for multi-dimensional signals along dimensions specified by ``order``.
 807
 808        A Gaussian derivative is an approximation of a derivative that consists in convolving the input function with a
 809        Gaussian function :math:`g` before applying a derivative.
 810        In the continuous domain, the :math:`N`-th order Gaussian derivative :math:`D^N_G` amounts to a convolution with
 811        the :math:`N`-th order derivative of :math:`g`:
 812
 813        .. math::
 814
 815           D^N_G f (x)
 816           =
 817           \frac{\mathrm{d}^N (f * g) }{\mathrm{d} x^N} (x)
 818           =
 819           f(x) * \frac{\mathrm{d}^N g}{\mathrm{d} x^N} (x).
 820
 821        For discrete signals :math:`f[n]`, this operator is approximated by
 822
 823        .. math::
 824
 825           \mathbf{D}^N_G f [n]
 826           =
 827           f[n] *\frac{\mathrm{d}^N g}{\mathrm{d} x^N} \left(\frac{n}{h}\right),
 828
 829        where :math:`h` is the spacing between samples and the operator :math:`*` is now a discrete convolution.
 830
 831        .. warning::
 832
 833           The operator :math:`\mathbf{D}_{G} \circ \mathbf{D}_{G}` is not directly related to
 834           :math:`\mathbf{D}_{G}^{2}`: Gaussian smoothing is performed twice in the case of the former, whereas it is
 835           performed only once in the case of the latter.
 836
 837        Note that in contrast with finite differences (see
 838        :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.finite_difference`), Gaussian derivatives compute exact
 839        derivatives in the continuous domain, since Gaussians can be differentiated analytically.
 840        This derivative is then sampled in order to perform a discrete convolution.
 841
 842        This class inherits its methods from :py:class:`~pyxu.operator.linop.stencil.stencil.Stencil`.

1605    r"""
1606    Hessian operator.
1607
1608    Notes
1609    -----
1610
1611    The Hessian matrix or Hessian of a :math:`D`-dimensional signal :math:`\mathbf{f} \in \mathbb{R}^{N_0 \times \cdots
1612    \times N_{D-1}}` is the square matrix of second-order partial derivatives:
1613
1614    .. math::
1615
1616       \mathbf{H} \mathbf{f} = \begin{bmatrix}
1617       \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{0}^{2} } &  \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{0}\,\partial x_{1} } & \cdots & \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{0} \, \partial x_{D-1} } \\
1618       \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{1} \, \partial x_{0} } & \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{1}^{2} } & \cdots & \dfrac{ \partial^{2}\mathbf{f} }{\partial x_{1} \,\partial x_{D-1}} \\
1619       \vdots & \vdots & \ddots & \vdots \\
1620       \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{D-1} \, \partial x_{0} } & \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{D-1} \, \partial x_{1} } & \cdots & \dfrac{ \partial^{2}\mathbf{f} }{ \partial x_{D-1}^{2}}
1621       \end{bmatrix}
1622
1623    The partial derivatives can be approximated by `finite differences
1624    <https://en.wikipedia.org/wiki/Finite_difference>`_ via the
1625    :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.finite_difference` constructor or by the `Gaussian derivative
1626    <https://www.crisluengo.net/archives/22/>`_ via
1627    :py:meth:`~pyxu.operator.linop.diff.PartialDerivative.gaussian_derivative` constructor.
1628    The parametrization of the partial derivatives can be done via the keyword arguments `\*\*diff_kwargs`, which will
1629    default to the same values as the :py:class:`~pyxu.operator.linop.diff.PartialDerivative` constructor.
1630
1631    The Hessian being symmetric, only the upper triangular part at most needs to be computed.
1632    Due to the (possibly) large size of the full Hessian, 4 different options are handled:
1633
1634    * [mode 0] ``directions`` is an integer, e.g.: ``directions=0``
1635
1636      .. math::
1637
1638         \partial^{2}\mathbf{f}/\partial x_{0}^{2}.
1639
1640    * [mode 1] ``directions`` is a tuple of length 2, e.g.: ``directions=(0,1)``
1641
1642      .. math::
1643
1644         \partial^{2}\mathbf{f}/\partial x_{0}\partial x_{1}.
1645
1646    * [mode 2] ``directions`` is a tuple of tuples, e.g.: ``directions=((0,0), (0,1))``
1647
1648      .. math::
1649
1650         \left(\frac{ \partial^{2}\mathbf{f} }{ \partial x_{0}^{2} }, \frac{ \partial^{2}\mathbf{f} }{ \partial x_{0}\partial x_{1} }\right).
1651
1652    * [mode 3] ``directions = 'all'`` (default), computes the Hessian for all directions (only the
1653      upper triangular part of the Hessian matrix), in row order, i.e.:
1654
1655      .. math::
1656
1657         \left(\frac{ \partial^{2}\mathbf{f} }{ \partial x_{0}^{2} }, \frac{ \partial^{2}\mathbf{f} }
1658         { \partial x_{0}\partial x_{1} }, \, \ldots , \, \frac{ \partial^{2}\mathbf{f} }{ \partial x_{D-1}^{2} }\right).
1659
1660    The shape of the output :py:class:`~pyxu.abc.operator.LinOp` depends on the number of computed directions; by
1661    default (all directions), we have :math:`\mathbf{H} \mathbf{f} \in \mathbb{R}^{\frac{D(D-1)}{2} \times N_0 \times
1662    \cdots \times N_{D-1}}`.
1663
1664    The Hessian operator's :py:meth:`~pyxu.operator.linop.diff.Hessian.unravel` method allows reshaping the vectorized
1665    output Hessian to `[n_dirs, N0, ..., ND]` (see the example below).
1666

1668    ----------
1669    arg_shape: NDArrayShape
1670        Shape of the input array.
1671    directions: Integer, (Integer, Integer), ((Integer, Integer), ..., (Integer, Integer)), 'all'
1672        Hessian directions.
1673        Defaults to `all`, which computes the Hessian for all directions. (See ``Notes``.)
1674    diff_method: "gd", "fd"
1675        Method used to approximate the derivative.
1676        Must be one of:
1677
1678        * 'fd' (default): finite differences
1679        * 'gd': Gaussian derivative
1680    mode: str, list[str]
1681        Boundary conditions.
1682        Multiple forms are accepted:
1683
1684        * str: unique mode shared amongst dimensions.
1685          Must be one of:
1686

1938def DirectionalDerivative(
1939    arg_shape: pxt.NDArrayShape,
1940    order: pxt.Integer,
1941    directions: typ.Union[pxt.NDArray, typ.Sequence[pxt.NDArray]],
1942    diff_method: str = "fd",
1943    mode: ModeSpec = "constant",
1944    parallel: bool = False,
1945    **diff_kwargs,
1946) -> pxt.OpT:
1947    r"""
1948    Directional derivative operator.
1949
1950    Notes
1951    -----
1952    The **first-order** ``DirectionalDerivative`` of a :math:`D`-dimensional signal :math:`\mathbf{f} \in
1953    \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}` applies a derivative along the direction specified by a constant
1954    unitary vector :math:`\mathbf{v} \in \mathbb{R}^D`:
1955
1956    .. math::
1957
1958       \boldsymbol{\nabla}_\mathbf{v} \mathbf{f} = \sum_{i=0}^{D-1} v_i \frac{\partial \mathbf{f}}{\partial x_i} \in
1959       \mathbb{R}^{N_0 \times \cdots \times N_{D-1}}
1960
1961    or along spatially-varying directions :math:`\mathbf{v} = [\mathbf{v}_0, \ldots , \mathbf{v}_{D-1}]^\top \in
1962    \mathbb{R}^{D \times N_0 \times \cdots \times N_{D-1} }` where each direction :math:`\mathbf{v}_{\cdot, i_0, \ldots
1963    , i_{D-1}} \in \mathbb{R}^D` for any :math:`0 \leq i_d \leq N_d-1` with :math:`0 \leq d \leq D-1` is a unitary
1964    vector:
1965
1966    .. math::
1967
1968       \boldsymbol{\nabla}_\mathbf{v} \mathbf{f} = \sum_{i=0}^{D-1} \mathbf{v}_i \odot
1969       \frac{\partial \mathbf{f}}{\partial x_i} \in \mathbb{R}^{N_0 \times \cdots \times N_{D-1}},
1970
1971    where :math:`\odot` denotes the Hadamard (elementwise) product.
1972
1973    Note that choosing :math:`\mathbf{v}= \mathbf{e}_d \in \mathbb{R}^D` (the :math:`d`-th canonical basis vector)
1974    amounts to the first-order :py:func:`~pyxu.operator.linop.diff.PartialDerivative` operator applied along axis
1975    :math:`d`.
1976
1977    The **second-order** ``DirectionalDerivative`` :math:`\boldsymbol{\nabla}^2_{\mathbf{v}_{1, 2}}` is obtained by
1978    composing the first-order directional derivatives :math:`\boldsymbol{\nabla}_{\mathbf{v}_{1}}` and
1979    :math:`\boldsymbol{\nabla}_{\mathbf{v}_{2}}`:
1980
1981    .. math::
1982
1983       \boldsymbol{\nabla}_{\mathbf{v}_{1}} (\boldsymbol{\nabla}_{\mathbf{v}_{2}} \mathbf{f}) =
1984       \boldsymbol{\nabla}_{\mathbf{v}_{1}} (\sum_{i=0}^{D-1} {v_{2}}_{i} \frac{\partial \mathbf{f}}{\partial x_i}) =
1985       \sum_{j=0}^{D-1} {v_{1}}_{j} \frac{\partial}{\partial x_j} (\sum_{i=0}^{D-1} {v_{2}}_{i} \frac{\partial \mathbf{f}}{\partial x_i}) =
1986       \sum_{i, j=0}^{D-1} {v_{1}}_{j} {v_{2}}_{i} \frac{\partial}{\partial x_j} \frac{\partial}{\partial x_i}\mathbf{f} =
1987       \mathbf{v}_{1}^{\top}\mathbf{H}\mathbf{f}\mathbf{v}_{2},
1988
1989    where :math:`\mathbf{H}` is the discrete Hessian operator, implemented via
1990    :py:class:`~pyxu.operator.linop.diff.Hessian`.
1991
1992    Higher-order ``DirectionalDerivative`` :math:`\boldsymbol{\nabla}^{N}_\mathbf{v}` can be obtained by composing the
1993    first-order directional derivative :math:`\boldsymbol{\nabla}_\mathbf{v}` :math:`N` times.
1994
1995    The directional derivative operator's method :py:meth:`~pyxu.operator.linop.diff.DirectionalDerivative.unravel`
1996    allows reshaping the vectorized output directional derivative to ``[..., N0, ..., ND]`` (see the example below).
1997
1998    .. warning::
1999
2000       - :py:func:`~pyxu.operator.linop.diff.DirectionalDerivative` instances are **not array module-agnostic**: they
2001         will only work with NDArrays belonging to the same array module as ``directions``. Inner
2002         computations may recast input arrays when the precision of ``directions`` does not match the user-requested
2003         precision.
2004       - ``directions`` are always normalized to be unit vectors.
2005
2006    Parameters
2007    ----------
2008    arg_shape: NDArrayShape
2009        Shape of the input array.
2010    order: Integer
2011        Which directional derivative (restricted to 1: First or 2: Second, see ``Notes``).
2012    directions: NDArray, list
2013        For ``order=1``, it can be a single direction (array of size :math:`(D,)`, where :math:`D` is the number of
2014        axes) or spatially-varying directions:
2015
2016        * array of size :math:`(D, N_0 \times \ldots \times N_{D-1})` for ``order=1``, i.e., one direction per element
2017          in the gradient, and,
2018
2019        * array of size :math:`(D * (D + 1) / 2, N_0 \times \ldots \times N_{D-1})` for ``order=2``, i.e., one direction
2020          per element in the Hessian.
2021
2022        For ``order=2``, it can be a tuple/list of two single directions or spatially-varying dimensions of the same
2023        shape, which will compute :math:`\mathbf{v}_{1}^{\top}\mathbf{H}\mathbf{f}\mathbf{v}_{2}` or a single direction,
2024        as for ``order=1``, which will compute :math:`\mathbf{v}_{1}^{\top}\mathbf{H}\mathbf{f}\mathbf{v}_{1}`. Note
2025        that for ``order=2``, even though directions are spatially-varying, no differentiation is performed for this
2026        parameter.
2027    diff_method: "gd", "fd"
2028        Method used to approximate the derivative. Must be one of:
2029
2030        * 'fd' (default): finite differences
2031        * 'gd': Gaussian derivative
2032    mode: str, list[str]
2033        Boundary conditions.
2034        Multiple forms are accepted:
2035
2036        * str: unique mode shared amongst dimensions.
2037          Must be one of:
2038
2039          * 'constant' (default): zero-padding
2040          * 'wrap'
2041          * 'reflect'
2042          * 'symmetric'
2043          * 'edge'
2044        * tuple[str, ...]: the `d`-th dimension uses ``mode[d]`` as boundary condition.

2113    else:

2117                "NDArrays of the same shape"

2123            directions2 = directions

2126    assert directions.shape[0] == ndim, "The length of `directions` should match `len(arg_shape)`"

2128    xp = pxu.get_array_module(directions)
2129    gpu = xp == pxd.NDArrayInfo.CUPY.module()
2130
2131    dtype = directions.dtype

2134        arg_shape=arg_shape,
2135        diff_method=diff_method,
2136        mode=mode,
2137        gpu=gpu,
2138        dtype=dtype,
2139        parallel=parallel,
2140        **diff_kwargs,
2141    )

2143    norm_dirs = (directions / xp.linalg.norm(directions, axis=0, keepdims=True)).astype(dtype)

2153        # Multiply off-diag components x2 and use only triangular upper part of the outer product
2154        if ndim > 1:

2158            off_diag_inds = dummy_mat[np.triu_indices(ndim, k=1)].ravel()
2159            norm_dirs[off_diag_inds] *= 2
2160            inds = dummy_mat[np.triu_indices(ndim, k=0)].ravel()
2161            norm_dirs = norm_dirs[inds]
2162        else:
2163            norm_dirs = norm_dirs.ravel()

2166        diag = xp.tile(norm_dirs, arg_shape + (1,)).transpose().ravel()
2167
2168    else:
2169        diag = norm_dirs.ravel()
