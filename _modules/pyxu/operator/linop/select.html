 18class SubSample(pxa.LinOp):
 19    r"""
 20    Multi-dimensional sub-sampling operator.
 21
 22    This operator extracts a subset of the input matching the provided subset specifier.
 23    Its Lipschitz constant is 1.
 24
 25    Examples
 26    --------
 27    .. code-block:: python3
 28
 29       ### Extract even samples of a 1D signal.
 30       import pyxu.operator as pxo
 31       x = np.arange(10)
 32       S = pxo.SubSample(
 33             x.shape,
 34             slice(0, None, 2),
 35       )
 36       y = S(x)  # array([0, 2, 4, 6, 8])
 37
 38
 39    .. code-block:: python3
 40
 41       ### Extract columns[1, 3, -1] from a 2D matrix
 42       import pyxu.operator as pxo
 43       x = np.arange(3 * 40).reshape(3, 40)  # the input
 44       S = pxo.SubSample(
 45             x.shape,
 46             slice(None),  # take all rows
 47             [1, 3, -1],   # and these columns
 48       )
 49       y = S(x.reshape(-1)).reshape(3, 3)  # array([[  1.,   3.,  39.],
 50                                           #        [ 41.,  43.,  79.],
 51                                           #        [ 81.,  83., 119.]])
 52
 53    .. code-block:: python3
 54
 55       ### Extract all red rows of an (D,H,W) RGB image matching a boolean mask.
 56       import pyxu.operator as pxo
 57       x = np.arange(3 * 5 * 4).reshape(3, 5, 4)
 58       mask = np.r_[True, False, False, True, False]
 59       S = pxo.SubSample(
 60             x.shape,
 61             0,            # red channel
 62             mask,         # row selector
 63             slice(None),  # all columns; this field can be omitted.
 64       )
 65       y = S(x.reshape(-1)).reshape(1, mask.sum(), 4)  # array([[[ 0.,  1.,  2.,  3.],
 66                                                       #         [12., 13., 14., 15.]]])
 67    """
 68    IndexSpec = typ.Union[
 69        pxt.Integer,
 70        cabc.Sequence[pxt.Integer],
 71        slice,
 72        pxt.NDArray,  # ints or boolean mask (per dimension)
 73    ]
 74
 75    TrimSpec = typ.Union[
 76        pxt.Integer,
 77        cabc.Sequence[pxt.Integer],
 78        cabc.Sequence[tuple[pxt.Integer, pxt.Integer]],
 79    ]

 81    def __init__(
 82        self,
 83        arg_shape: pxt.NDArrayShape,
 84        *indices: IndexSpec,
 85    ):
 86        """
 87        Parameters
 88        ----------
 89        arg_shape: NDArrayShape
 90            Shape of the data to be sub-sampled.
 91        indices: ~pyxu.operator.linop.select.SubSample.IndexSpec
 92            Sub-sample specifier per dimension. (See examples.)
 93
 94            Valid specifiers are:
 95
 96            * integers
 97            * lists (or arrays) of indices
 98            * slices
 99            * 1D boolean masks
100
101            Unspecified trailing dimensions are not sub-sampled.
102        """
103        self._arg_shape = pxu.as_canonical_shape(arg_shape)
104
105        assert 1 <= len(indices) <= len(self._arg_shape)
106        self._idx = [slice(None)] * len(self._arg_shape)
107        for i, idx in enumerate(indices):
108            if isinstance(idx, pxt.Integer):
109                idx = slice(idx, idx + 1)
110            self._idx[i] = idx
111        self._idx = tuple(self._idx)
112
113        output = np.broadcast_to(0, self._arg_shape)[self._idx]
114        self._sub_shape = np.atleast_1d(output).shape
115
116        super().__init__(shape=(np.prod(self._sub_shape), np.prod(self._arg_shape)))
117        self.lipschitz = 1

119    @pxrt.enforce_precision(i="arr")
120    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
121        """
122        Sub-sample the data.
123
124        Parameters
125        ----------
126        arr: NDArray
127            (..., arg_shape.prod()) data
128
129        Returns
130        -------
131        out: NDArray
132            (..., sub_shape.prod()) sub-sampled data points.
133        """
134        sh = arr.shape[:-1]
135        arr = arr.reshape(*sh, *self._arg_shape)
136
137        selector = (*[slice(None) for dim in sh], *self._idx)
138        out = arr[selector].reshape(*sh, -1)
139
140        out = pxu.read_only(out)
141        return out

143    @pxrt.enforce_precision(i="arr")
144    def adjoint(self, arr: pxt.NDArray) -> pxt.NDArray:
145        """
146        Up-sample the data.
147
148        Parameters
149        ----------
150        arr: NDArray
151            (..., sub_shape.prod()) data points.
152
153        Returns
154        -------
155        out: NDArray
156            (..., arg_shape.prod()) up-sampled data points. (Zero-filled.)
157        """
158        sh = arr.shape[:-1]
159        arr = arr.reshape(*sh, *self._sub_shape)
160
161        xp = pxu.get_array_module(arr)
162        out = xp.zeros((*sh, *self._arg_shape), dtype=arr.dtype)
163        selector = (*[slice(None) for dim in sh], *self._idx)
164        out[selector] = arr
165
166        out = out.reshape(*sh, -1)
167        return out

205def Trim(
206    arg_shape: pxt.NDArrayShape,
207    trim_width: SubSample.TrimSpec,
208) -> pxt.OpT:
209    """
210    Multi-dimensional trimming operator.
211
212    This operator trims the input array in each dimension according to specified widths.
213
214    Parameters
215    ----------
216    arg_shape: NDArrayShape
217        Shape of the input array.
218    trim_width: ~pyxu.operator.linop.select.SubSample.TrimSpec
219        Number of values trimmed from the edges of each axis.
220        Multiple forms are accepted:
221
222        * ``int``: trim each dimension's head/tail by `trim_width`.
223        * ``tuple[int, ...]``: trim dimension[k]'s head/tail by `trim_width[k]`.
224        * ``tuple[tuple[int, int], ...]``: trim dimension[k]'s head/tail by `trim_width[k][0]` /
225          `trim_width[k][1]` respectively.
226
227    Returns
228    -------
229    op: OpT
230    """
231    arg_shape = tuple(arg_shape)
232    N_dim = len(arg_shape)
233
234    # transform `trim_width` to canonical form tuple[tuple[int, int]]
235    is_seq = lambda _: isinstance(_, cabc.Sequence)
236    if not is_seq(trim_width):  # int-form
237        trim_width = ((trim_width, trim_width),) * N_dim
238    assert len(trim_width) == N_dim, "arg_shape/trim_width are length-mismatched."
239    if not is_seq(trim_width[0]):  # tuple[int, ...] form
240        trim_width = tuple((w, w) for w in trim_width)
241    else:  # tuple[tuple[int, int], ...] form
242        pass
243
244    # translate `trim_width` to `indices` needed for SubSample
245    indices = []
246    for (w_head, w_tail), dim_size in zip(trim_width, arg_shape):
247        s = slice(w_head, dim_size - w_tail)
248        indices.append(s)
249
250    op = SubSample(arg_shape, *indices)
251    return op
