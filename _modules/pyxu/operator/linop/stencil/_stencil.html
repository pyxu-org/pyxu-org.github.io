 15def _signature(params, returns) -> str:
 16    # Translate a signature of the form
 17    #     [in_1_spec, ..., in_N_spec] -> out_spec
 18    # to Numba's string representation.
 19    #
 20    # Parameters
 21    # ----------
 22    # params: list(spec)
 23    # returns: spec | None
 24    #
 25    # Returns
 26    # -------
 27    # sig: str
 28    #
 29    # Notes
 30    # -----
 31    # A parameter spec is characterized by the triplet
 32    #     (dtype[single/double], ndim[int], c_contiguous[bool])
 33    def fmt(spec) -> str:
 34        dtype, ndim, c_contiguous = spec
 35
 36        _dtype_spec = {
 37            pxrt.Width.SINGLE: "float32",
 38            pxrt.Width.DOUBLE: "float64",
 39        }[pxrt.Width(dtype)]
 40
 41        dim_spec = [":"] * ndim
 42        if c_contiguous and (ndim > 0):
 43            dim_spec[-1] = "::1"
 44        dim_spec = "[" + ",".join(dim_spec) + "]"
 45
 46        _repr = _dtype_spec
 47        if ndim > 0:
 48            _repr += dim_spec
 49        return _repr
 50
 51    sig = "".join(
 52        [
 53            "void" if (returns is None) else fmt(returns),
 54            "(",
 55            ", ".join(map(fmt, params)),
 56            ")",
 57        ]
 58    )
 59    return sig

 62class _Stencil:
 63    """
 64    Multi-dimensional JIT-compiled stencil.
 65
 66    This low-level class creates a gu-vectorized stencil applicable on multiple inputs simultaneously.
 67
 68    Create instances via factory method :py:meth:`~pyxu.operator.linop.stencil._stencil._Stencil.init`.
 69
 70    Example
 71    -------
 72    Correlate a stack of images `A` with a (3, 3) kernel such that:
 73
 74    .. math::
 75
 76       B[n, m] = A[n-1, m] + A[n, m-1] + A[n, m+1] + A[n+1, m]
 77
 78    .. code-block:: python3
 79
 80       import numpy as np
 81       from pyxu.operator.linop.stencil._stencil import _Stencil
 82
 83       # create the stencil
 84       kernel = np.array([[0, 1, 0],
 85                          [1, 0, 1],
 86                          [0, 1, 0]], dtype=np.float64)
 87       center = (1, 1)
 88       stencil = _Stencil.init(kernel, center)
 89
 90       # apply it to the data
 91       rng = np.random.default_rng()
 92       A = rng.normal(size=(2, 3, 4, 30, 30))  # 24 images of size (30, 30)
 93       B = np.zeros_like(A)
 94       stencil.apply(A, B)  # (2, 3, 4, 30, 30)
 95    """

 99    @staticmethod
100    def init(
101        kernel: pxt.NDArray,
102        center: IndexSpec,
103    ):
104        """
105        Parameters
106        ----------
107        kernel: NDArray
108            (k_1, ..., k_D) kernel coefficients.
109
110            Only float32/64 kernels are supported.
111        center: ~pyxu.operator.linop.stencil._stencil._Stencil.IndexSpec
112            (D,) index of the kernel's center.
113
114        Returns
115        -------
116        st: ~pyxu.operator.linop.stencil._stencil._Stencil
117            Rank-D stencil.
118        """
119        dtype = kernel.dtype
120        if dtype not in {_.value for _ in pxrt.Width}:
121            raise ValueError(f"Unsupported kernel precision {dtype}.")
122
123        center = np.array(center, dtype=int)
124        assert center.size == kernel.ndim
125        assert np.all(0 <= center) and np.all(center < kernel.shape)
126
127        N = pxd.NDArrayInfo
128        ndi = N.from_obj(kernel)
129        if ndi == N.NUMPY:
130            klass = _Stencil_NP
131        elif ndi == N.CUPY:
132            klass = _Stencil_CP
133        else:
134            raise NotImplementedError
135
136        st = klass(kernel, center)
137        return st

139    def apply(
140        self,
141        arr: pxt.NDArray,
142        out: pxt.NDArray,
143        **kwargs,
144    ) -> pxt.NDArray:
145        r"""
146        Evaluate stencil on multiple inputs.
147
148        Parameters
149        ----------
150        arr: NDArray
151            (..., N_1, ..., N_D) data to process.
152        out: NDArray
153            (..., N_1, ..., N_D) array to which outputs are written.
154        kwargs: dict
155            Extra kwargs to configure `f_jit()`, the Dispatcher instance created by Numba.
156
157            Only relevant for GPU stencils, with values:
158
159            * blockspergrid: int
160            * threadsperblock: int
161
162            Default values are chosen if unspecified.
163
164        Returns
165        -------
166        out: NDArray
167            (..., N_1, ..., N_D) outputs.
168
169        Notes
170        -----
171        * `arr` and `out` must have the same type/dtype as the kernel used during instantiation.
172        * Index regions in `out` where the stencil is not fully supported are set to 0.
173        * :py:meth:`~pyxu.operator.linop.stencil._stencil._Stencil.apply` may raise
174          ``CUDA_ERROR_LAUNCH_OUT_OF_RESOURCES`` when the number of GPU registers required exceeds resource limits.
175          There are 2 solutions to this problem:
176
177            (1) Pass the `max_registers` kwarg to f_jit()'s decorator; or
178            (2) `Limit the number of threads per block <https://stackoverflow.com/a/68659008>`_.
179
180          (1) must be set at compile time; it is thus left unbounded.
181          (2) is accessible through .apply(\*\*kwargs).
182        """
183        assert arr.dtype == out.dtype == self._kernel.dtype
184        assert arr.shape == out.shape
185        # assert arr.flags.c_contiguous and out.flags.c_contiguous
186        # [2022.12.25, Sepand]
187        #   Preferable to explicitly enforce C-contiguity for better performance, but then
188        #   LinOp.to_sciop() is partially broken.
189
190        K_dim = len(self._kernel.shape)
191        arg_shape = arr.shape[-K_dim:]
192
193        stencil = self._configure_dispatcher(arr.size, **kwargs)
194        stencil(
195            arr.reshape(-1, *arg_shape),
196            out.reshape(-1, *arg_shape),
197        )
198        return out

232class _Stencil_NP(_Stencil):
233    def _gen_code(self) -> str:
234        _template = string.Template(
235            r"""
236import numba
237
238@numba.stencil(
239    func_or_mode="constant",
240    cval=0,
241)
242def f_stencil(a):
243    # rank-D kernel [specified as rank-(D+1) to work across stacked dimension].
244    return ${stencil_spec}
245
246@numba.jit(
247    "${signature}",
248    nopython=True,
249    nogil=True,
250    forceobj=False,
251    parallel=True,
252    error_model="numpy",
253    cache=False,  # not applicable to dynamically-defined functions (https://github.com/numba/numba/issues/3501)
254    fastmath=True,
255    boundscheck=False,
256)
257def f_jit(arr, out):
258    # arr: (N_stack, N_1, ..., N_D)
259    # out: (N_stack, N_1, ..., N_D)
260    f_stencil(arr, out=out)
261"""
262        )
263        code = _template.substitute(
264            signature=self.__sig_spec(),
265            stencil_spec=self.__stencil_spec(),
266        )
267        return code

278    def __stencil_spec(self) -> str:
279        f_fmt = {  # coef float-formatter
280            pxrt.Width.SINGLE: "1.8e",
281            pxrt.Width.DOUBLE: "1.16e",
282        }[pxrt.Width(self._kernel.dtype)]
283
284        entry = []
285        _range = list(map(range, self._kernel.shape))
286        for idx in itertools.product(*_range):
287            idx_c = [i - c for (i, c) in zip(idx, self._center)]
288            idx_c = ",".join(map(str, idx_c))
289
290            cst = self._kernel[idx]
291            if np.isclose(cst, 0):
292                # no useless look-ups at runtime
293                e = None
294            elif np.isclose(cst, 1):
295                # no multiplication required
296                e = f"a[0,{idx_c}]"
297            else:
298                # general case
299                e = f"({cst:{f_fmt}} * a[0,{idx_c}])"
300
301            if e is not None:
302                entry.append(e)
303
304        spec = " + ".join(entry)
305        return spec

308class _Stencil_CP(_Stencil):
309    def _gen_code(self) -> str:
310        _template = string.Template(
311            r"""
312import numba.cuda
313
314@numba.cuda.jit(
315    device=True,
316    inline=True,
317    fastmath=True,
318    opt=True,
319    cache=False,
320)
321def unravel_offset(
322    offset,  # int
323    shape,  # (D+1,)
324):
325    # Compile-time-equivalent of np.unravel_index(offset, shape, order='C').
326    ${unravel_spec}
327    return idx
328
329
330@numba.cuda.jit(
331    device=True,
332    inline=True,
333    fastmath=True,
334    opt=True,
335    cache=False,
336)
337def full_overlap(
338    idx,  # (D+1,)
339    dimensions,  # (D+1,)
340):
341    # Computes intersection of:
342    # * idx[0] < dimensions[0]
343    # * 0 <= idx[1:] - kernel_center
344    # * idx[1:] - kernel_center <= dimensions[1:] - kernel_width
345    if not (idx[0] < dimensions[0]):
346        # went beyond stack-dim limit
347        return False
348
349    kernel_width = ${kernel_width}
350    kernel_center = ${kernel_center}
351    for i, w, c, n in zip(
352        idx[1:],
353        kernel_width,
354        kernel_center,
355        dimensions[1:],
356    ):
357        if not (0 <= i - c <= n - w):
358            # kernel partially overlaps ROI around `idx`
359            return False
360
361    # kernel fully overlaps ROI around `idx`
362    return True
363
364
365@numba.cuda.jit(
366    func_or_sig="${signature}",
367    device=False,
368    inline=False,
369    fastmath=True,
370    opt=True,
371    cache=False,  # not applicable to dynamically-defined functions (https://github.com/numba/numba/issues/3501)
372    # max_registers=None,  # see .apply() notes
373)
374def f_jit(arr, out):
375    # arr: (N_stack, N_1, ..., N_D)
376    # out: (N_stack, N_1, ..., N_D)
377    offset = numba.cuda.grid(1)
378    if offset < arr.size:
379        idx = unravel_offset(offset, arr.shape)
380        if full_overlap(idx, arr.shape):
381            out[idx] = ${stencil_spec}
382        else:
383            out[idx] = 0
384"""
385        )
386        code = _template.substitute(
387            kernel_center=str(tuple(self._center)),
388            kernel_width=str(self._kernel.shape),
389            signature=self.__sig_spec(),
390            stencil_spec=self.__stencil_spec(),
391            unravel_spec=self.__unravel_spec(),
392        )
393        return code

395    def _configure_dispatcher(self, pb_size: int, **kwargs) -> cabc.Callable:
396        # Set (`threadsperblock`, `blockspergrid`)
397        assert set(kwargs.keys()) <= {
398            "threadsperblock",
399            "blockspergrid",
400        }
401
402        attr = self._kernel.device.attributes
403        tpb = kwargs.get("threadsperblock", attr["MaxThreadsPerBlock"])
404        bpg = kwargs.get("blockspergrid", (pb_size // tpb) + 1)
405        return self._dispatch[bpg, tpb]

412    def __stencil_spec(self) -> str:
413        f_fmt = {  # coef float-formatter
414            pxrt.Width.SINGLE: "1.8e",
415            pxrt.Width.DOUBLE: "1.16e",
416        }[pxrt.Width(self._kernel.dtype)]
417
418        entry = []
419        _range = list(map(range, self._kernel.shape))
420        for idx in itertools.product(*_range):
421            # create string of form "idx[1]+i1,...,idx[K]+iK"
422            idx_c = [i - c for (i, c) in zip(idx, self._center)]
423            idx_c = [f"idx[{i1}]{i2:+d}" for (i1, i2) in enumerate(idx_c, start=1)]
424            idx_c = ",".join(idx_c)
425
426            cst = self._kernel[idx]
427            if np.isclose(cst, 0):
428                # no useless look-ups at runtime
429                e = None
430            elif np.isclose(cst, 1):
431                # no multiplication required
432                e = f"arr[idx[0],{idx_c}]"
433            else:
434                # general case
435                e = f"({cst:{f_fmt}} * arr[idx[0],{idx_c}])"
436
437            if e is not None:
438                entry.append(e)
439
440        spec = " + ".join(entry)
441        return spec

443    def __unravel_spec(self) -> str:
444        N = self._kernel.ndim + 1  # 1 stack-dim
445        entry = []
446
447        # left = offset
448        e = "left = offset"
449        entry.append(e)
450
451        # blk = prod(shape)
452        e = "blk = " + " * ".join([f"shape[{n}]" for n in range(N)])
453        entry.append(e)
454
455        for n in range(N):
456            # blk //= shape[n]
457            e = f"blk //= shape[{n}]"
458            entry.append(e)
459
460            # i{n} = left // blk
461            e = f"i{n} = left // blk"
462            entry.append(e)
463
464            # left -= i{n} * blk
465            e = f"left -= i{n} * blk"
466            entry.append(e)
467
468        # idx = (i0, ..., i{N})
469        e = "idx = (" + ", ".join([f"i{n}" for n in range(N)]) + ")"
470        entry.append(e)
471
472        # indent each entry by 4, then concatenate
473        for i in range(1, len(entry)):  # 1st line skipped
474            entry[i] = "    " + entry[i]
475        spec = "\n".join(entry)
476        return spec
