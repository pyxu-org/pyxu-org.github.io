   1r"""
   2Universal functions.
   3
   4Ufuncs have class-oriented and function-oriented interfaces.
   5Only the functional interface is exposed due to its common use.
   6
   7Example
   8-------
   9.. code-block:: python3
  10
  11   from pyxu.operator.map import Sin, sin
  12   from pyxu.abc import LinOp
  13
  14   N = 10
  15   x = np.random.randn(N)
  16   A = LinOp.from_array(np.random.randn(N, N))
  17
  18   op1 = _Sin(x.size) * A  # class interface
  19   op2 = sin(A)           # function interface
  20
  21   np.allclose(op1.apply(x), np.sin(A.apply(x)))  # True
  22   np.allclose(op2.apply(x), np.sin(A.apply(x)))  # True
  23"""

  82def sin(op: pxt.OpT) -> pxt.OpT:
  83    r"""
  84    Trigonometric sine, element-wise.
  85
  86    Notes
  87    -----
  88    * :math:`f(x) = \sin(x)`
  89    * :math:`f'(x) = \cos(x)`
  90    * :math:`\vert f(x) - f(y) \vert \le L \vert x - y \vert`, with Lipschitz constant :math:`L = 1`.
  91
  92      (Reason: :math:`\vert f'(x) \vert` is bounded by :math:`L` at :math:`x = k \pi, \, k \in \mathbb{Z}`.)
  93    * :math:`\vert f'(x) - f'(y) \vert \le \partial L \vert x - y \vert`, with diff-Lipschitz
  94      constant :math:`\partial L = 1`.
  95
  96      (Reason: :math:`\vert f''(x) \vert` is bounded by :math:`\partial L` at :math:`x = (2k + 1)
  97      \frac{\pi}{2}, \, k \in \mathbb{Z}`.)
  98    """
  99    return _Sin(op.dim) * op

 271    def jacobian(self, arr: pxt.NDArray) -> pxt.OpT:
 272        v = arr**2
 273        v += 1
 274        return pxlb.DiagonalOp(1 / v)

 373        self.diff_lipschitz = 4 / (3 * np.sqrt(3))

 375        # = max_{x \in R} |-2 tanh(x) [1 - tanh(x)^2|
 376        # = 4 / (3 \sqrt(3))  [at x = ln(2 +- \sqrt(3))]

 385        v = self.apply(arr)
 386        v**2
 387        v *= -1
 388        v += 1
 389        return pxlb.DiagonalOp(v)

 532class _Exp(pxa.DiffMap):
 533    def __init__(self, dim: pxt.Integer, base: pxt.Real = None):
 534        super().__init__(shape=(dim, dim))
 535        self.lipschitz = np.inf
 536        self.diff_lipschitz = np.inf
 537        self._base = base
 538
 539    @pxrt.enforce_precision(i="arr")
 540    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 541        out = arr.copy()
 542        if self._base is not None:
 543            out *= np.log(float(self._base))
 544
 545        xp = pxu.get_array_module(arr)
 546        xp.exp(out, out=out)
 547        return out
