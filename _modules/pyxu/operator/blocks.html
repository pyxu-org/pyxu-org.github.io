  30def stack(
  31    ops: cabc.Sequence[pxt.OpT],
  32    axis: pxt.Integer,
  33    **kwargs,
  34) -> pxt.OpT:
  35    r"""
  36    Construct a stacked operator.
  37
  38    A stacked-operator :math:`V: \mathbb{R}^{d} \to \mathbb{R}^{c}` is an operator containing (vertically or
  39    horizontally) blocks of smaller operators :math:`\{O_{1}, \ldots, O_{N}\}`.
  40
  41    This is a convenience function around :py:func:`~pyxu.operator.blocks.hstack` and
  42    :py:func:`~pyxu.operator.blocks.vstack`.
  43
  44    Parameters
  45    ----------
  46    ops: :py:class:`~collections.abc.Sequence` ( :py:attr:`~pyxu.info.ptype.OpT` )
  47        [op1(c1, d1), ..., opN(cN, dN)] operators to join.
  48    axis: Integer
  49        The axis along which operators will be joined, i.e.
  50
  51        * 0: stack vertically (row-wise)
  52        * 1: stack horizontally (column-wise)
  53    kwargs
  54        Extra keyword-arguments forwarded to :py:class:`~pyxu.operator.blocks._COOBlock`.
  55
  56    Returns
  57    -------
  58    op: OpT
  59        Stacked operator.
  60
  61    See Also
  62    --------
  63    :py:func:`~pyxu.operator.blocks.hstack`,
  64    :py:func:`~pyxu.operator.blocks.vstack`.
  65    """
  66    axis = int(axis)
  67    assert axis in {0, 1}, f"axis: out-of-bounds axis '{axis}'."
  68
  69    f = {0: vstack, 1: hstack}[axis]
  70    op = f(ops, **kwargs)
  71    return op

  74def vstack(
  75    ops: cabc.Sequence[pxt.OpT],
  76    **kwargs,
  77) -> pxt.OpT:
  78    r"""
  79    Construct a vertically-stacked operator.
  80
  81    A vstacked-operator :math:`V: \mathbb{R}^{d} \to \mathbb{R}^{c_{1} + \cdots + c_{N}}` is an operator containing
  82    (vertically) blocks of smaller operators :math:`\{O_{1}: \mathbb{R}^{d} \to \mathbb{R}^{c_{1}}, \ldots, O_{N}:
  83    \mathbb{R}^{d} \to \mathbb{R}^{c_{N}}\}`, i.e.
  84
  85    .. math::
  86
  87       V
  88       =
  89       \left[
  90           \begin{array}{c}
  91               O_{1}  \\
  92               \vdots \\
  93               O_{N}  \\
  94           \end{array}
  95       \right]
  96
  97    Parameters
  98    ----------
  99    ops: :py:class:`~collections.abc.Sequence` ( :py:attr:`~pyxu.info.ptype.OpT` )
 100        [op1(c1, d), ..., opN(cN, d)] operators to concatenate.
 101    kwargs
 102        Extra keyword-arguments forwarded to :py:class:`~pyxu.operator.blocks._COOBlock`.
 103
 104    Returns
 105    -------
 106    op: OpT
 107        Vertically-stacked (c1+...+cN, d) operator.
 108
 109    Notes
 110    -----
 111    * All sub-operator domains must have compatible shapes, i.e. all integer-valued ``dim`` s must be identical.
 112
 113    See Also
 114    --------
 115    :py:func:`~pyxu.operator.blocks.hstack`,
 116    :py:func:`~pyxu.operator.blocks.stack`.
 117    """
 118
 119    def op_expr(_) -> tuple:
 120        N_row = _._grid_shape[0]
 121        ops = [_._block[(r, 0)] for r in range(N_row)]
 122        return ("vstack", *ops)
 123
 124    N_data = len(ops)
 125    op = _COOBlock(
 126        ops=(
 127            ops,  # data
 128            (
 129                tuple(range(N_data)),  # i
 130                [0] * N_data,  # j
 131            ),
 132        ),
 133        grid_shape=(N_data, 1),
 134        parallel=kwargs.get("parallel", False),
 135    ).op()
 136    op._expr = types.MethodType(op_expr, op)
 137    return op

 140def hstack(
 141    ops: cabc.Sequence[pxt.OpT],
 142    **kwargs,
 143) -> pxt.OpT:
 144    r"""
 145    Construct a horizontally-stacked operator.
 146
 147    A hstacked-operator :math:`H: \mathbb{R}^{d_{1} + \cdots + d_{N}} \to \mathbb{R}^{c}` is an operator containing
 148    (horizontally) blocks of smaller operators :math:`\{O_{1}: \mathbb{R}^{d_{1}} \to \mathbb{R}^{c}, \ldots, O_{N}:
 149    \mathbb{R}^{d_{N}} \to \mathbb{R}^{c}\}`, i.e.
 150
 151    .. math::
 152
 153       H
 154       =
 155       \left[
 156           \begin{array}{ccc}
 157               O_{1} & \cdots & O_{N}
 158           \end{array}
 159       \right]
 160
 161    Parameters
 162    ----------
 163    ops: :py:class:`~collections.abc.Sequence` ( :py:attr:`~pyxu.info.ptype.OpT` )
 164        [op1(c, d1), ..., opN(c, dN)] operators to concatenate.
 165    kwargs
 166        Extra keyword-arguments forwarded to :py:class:`~pyxu.operator.blocks._COOBlock`.
 167
 168    Returns
 169    -------
 170    op: OpT
 171        Horizontally-stacked (c, d1+....+dN) operator.
 172
 173    Notes
 174    -----
 175    * All sub-operator domains must have compatible shapes, i.e. all ``codim`` s must be identical.
 176
 177    See Also
 178    --------
 179    :py:func:`~pyxu.operator.blocks.vstack`,
 180    :py:func:`~pyxu.operator.blocks.stack`.
 181    """
 182
 183    def op_expr(_) -> tuple:
 184        N_col = _._grid_shape[1]
 185        ops = [_._block[(0, c)] for c in range(N_col)]
 186        return ("hstack", *ops)
 187
 188    N_data = len(ops)
 189    op = _COOBlock(
 190        ops=(
 191            ops,  # data
 192            (
 193                [0] * N_data,  # i
 194                tuple(range(N_data)),  # j
 195            ),
 196        ),
 197        grid_shape=(1, N_data),
 198        parallel=kwargs.get("parallel", False),
 199    ).op()
 200    op._expr = types.MethodType(op_expr, op)
 201    return op

 204def block_diag(
 205    ops: cabc.Sequence[pxt.OpT],
 206    **kwargs,
 207) -> pxt.OpT:
 208    r"""
 209    Construct a block-diagonal operator.
 210
 211    A block-diagonal operator :math:`D: \mathbb{R}^{d_{1} + \cdots + d_{N}} \to \mathbb{R}^{c_{1} + \cdots + c_{N}}` is
 212    an operator containing (diagonally) blocks of smaller operators :math:`\{O_{1}: \mathbb{R}^{d_{1}} \to
 213    \mathbb{R}^{c_{1}}, \ldots, O_{N}: \mathbb{R}^{d_{N}} \to \mathbb{R}^{c_{N}}\}`, i.e.
 214
 215    .. math::
 216
 217       D
 218       =
 219       \left[
 220           \begin{array}{ccc}
 221               O_{1} &        &       \\
 222                     & \ddots &       \\
 223                     &        & O_{N} \\
 224           \end{array}
 225       \right]
 226
 227    Parameters
 228    ----------
 229    ops: :py:class:`~collections.abc.Sequence` ( :py:attr:`~pyxu.info.ptype.OpT` )
 230        [op1(c1, d1), ..., opN(cN, dN)] operators to concatenate.
 231    kwargs
 232        Extra keyword-arguments forwarded to :py:class:`~pyxu.operator.blocks._COOBlock`.
 233
 234    Returns
 235    -------
 236    op: OpT
 237        Block-diagonal (c1+...+cN, d1+...+dN) operator.
 238
 239    See Also
 240    --------
 241    :py:func:`~pyxu.operator.blocks.block`,
 242    :py:func:`~pyxu.operator.blocks.coo_block`.
 243    """
 244
 245    def op_svdvals(_, **kwargs) -> pxt.NDArray:
 246        # op.svdvals(**kwargs) = [top|bottom-k]([op1.svdvals(**kwargs), ..., opN.svdvals(**kwargs)])
 247        if not _.has(pxo.Property.LINEAR):
 248            raise NotImplementedError
 249
 250        k = kwargs.get("k", 1)
 251        which = kwargs.get("which", "LM")
 252        if which.upper() == "SM":
 253            D = _.__class__.svdvals(_, **kwargs)
 254        else:
 255            parts = [op.svdvals(**kwargs) for op in _._block.values()]
 256            xp = pxu.get_array_module(parts[0])
 257            D = xp.sort(xp.concatenate(parts, axis=0), axis=None)[-k:]
 258        return D
 259
 260    @pxrt.enforce_precision(i=("arr", "damp"))
 261    def op_pinv(_, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
 262        # op.pinv(y, damp) = concatenate([op1.pinv(y1, damp), ..., opN.pinv(yN, damp)], axis=-1)
 263        if not _.has(pxo.Property.LINEAR):
 264            raise NotImplementedError
 265
 266        parts = dict()
 267        for idx, op in _._block.items():
 268            offset = _._block_offset[idx][0]
 269            p = op.pinv(arr[..., offset : offset + op.codim], damp, **kwargs)
 270            parts[idx] = p
 271
 272        xp = pxu.get_array_module(arr)
 273        parts = [parts[k] for k in sorted(parts.keys())]
 274        out = xp.concatenate(parts, axis=-1)
 275        return out
 276
 277    @pxrt.enforce_precision()
 278    def op_trace(_, **kwargs) -> pxt.Real:
 279        # op.trace(**kwargs) = if [op1,...,opN] square
 280        #                          sum([op1.trace(**kwargs), ..., opN.trace(**kwargs)])
 281        #                      else
 282        #                          SquareOp.trace(**kwargs)
 283        if not _.has(pxo.Property.LINEAR_SQUARE):
 284            raise NotImplementedError
 285
 286        if all(op.has(pxo.Property.LINEAR_SQUARE) for op in _._block.values()):
 287            parts = [op.trace(**kwargs) for op in _._block.values()]
 288            tr = sum(parts)
 289        else:  # default fallback
 290            tr = pxo.SquareOp.trace(_, **kwargs)
 291        return tr
 292
 293    def op_expr(_) -> tuple:
 294        ops = [_._block[k] for k in sorted(_._block.keys())]
 295        return ("block_diag", *ops)
 296
 297    N_data = len(ops)
 298    op = _COOBlock(
 299        ops=(
 300            ops,  # data
 301            (
 302                tuple(range(N_data)),  # i
 303                tuple(range(N_data)),  # j
 304            ),
 305        ),
 306        grid_shape=(N_data, N_data),
 307        parallel=kwargs.get("parallel", False),
 308    ).op()
 309    op.svdvals = types.MethodType(op_svdvals, op)
 310    op.pinv = types.MethodType(op_pinv, op)
 311    op.trace = types.MethodType(op_trace, op)
 312    op._expr = types.MethodType(op_expr, op)
 313    return op

 316def block(
 317    ops: cabc.Sequence[cabc.Sequence[pxt.OpT]],
 318    order: pxt.Integer,
 319    **kwargs,
 320) -> pxt.OpT:
 321    r"""
 322    Construct a (dense) block-defined operator.
 323
 324    A block-defined operator is an operator containing blocks of smaller operators.
 325    Blocks are stacked horizontally/vertically in a user-specified order to obtain the final shape.
 326
 327    Parameters
 328    ----------
 329    ops: :py:class:`~collections.abc.Sequence` ( :py:class:`~collections.abc.Sequence` ( :py:attr:`~pyxu.info.ptype.OpT` ) )
 330        2D nested sequence of (ck, dk)-shaped operators.
 331    order: 0, 1
 332        Order in which the nested operators are specified/concatenated:
 333
 334        * 0: concatenate inner-most blocks via :py:func:`~pyxu.operator.blocks.vstack`, then :py:func:`~pyxu.operator.blocks.hstack`.
 335        * 1: concatenate inner-most blocks via :py:func:`~pyxu.operator.blocks.hstack`, then :py:func:`~pyxu.operator.blocks.vstack`.
 336    kwargs
 337        Extra keyword-arguments forwarded to :py:class:`~pyxu.operator.blocks._COOBlock`.
 338
 339    Returns
 340    -------
 341    op: OpT
 342        Block-defined operator. (See below for examples.)
 343
 344    Notes
 345    -----
 346    * Each row/column may contain a different number of operators.
 347
 348    Examples
 349    --------
 350
 351    .. code::
 352
 353       >>> block(
 354       ...    [
 355       ...     [A],        # ABEEGGG
 356       ...     [B, C, D],  # ACEEHHH
 357       ...     [E, F],     # ADFFHHH
 358       ...     [G, H],
 359       ...    ],
 360       ...    order=0,
 361       ... )
 362
 363       >>> block(
 364       ...    [
 365       ...     [A, B, C, D],  # ABBCD
 366       ...     [E],           # EEEEE
 367       ...     [F, G],        # FFGGG
 368       ...    ],
 369       ...    order=1,
 370       ... )
 371
 372    See Also
 373    --------
 374    :py:func:`~pyxu.operator.blocks.block_diag`,
 375    :py:func:`~pyxu.operator.blocks.coo_block`.
 376    """
 377    order = int(order)
 378    assert order in {0, 1}, f"order: out-of-bounds order '{order}'."
 379
 380    inner = {0: vstack, 1: hstack}[order]
 381    outer = {0: hstack, 1: vstack}[order]
 382
 383    op = outer([inner(row, **kwargs) for row in ops], **kwargs)
 384    return op

 387def coo_block(
 388    ops: tuple[
 389        cabc.Sequence[pxt.OpT],
 390        tuple[
 391            cabc.Sequence[pxt.Integer],
 392            cabc.Sequence[pxt.Integer],
 393        ],
 394    ],
 395    grid_shape: pxt.OpShape,
 396    *,
 397    parallel: bool = False,
 398) -> pxt.OpT:
 399    r"""
 400    Constuct a (possibly-sparse) block-defined operator in COOrdinate format.
 401
 402    A block-defined operator is an operator containing blocks of smaller operators.
 403    Blocks must align on a coarse grid, akin to the COO format used to define sparse arrays.
 404
 405    Parameters
 406    ----------
 407    ops
 408        (data, (i, j)) sequences defining block placement, i.e.
 409
 410        * `data[:]` are :py:attr:`~pyxu.info.ptype.OpT` instances, in any order.
 411        * `i[:]` are the row indices of the block entries on the coarse grid.
 412        * `j[:]` are the column indices of the block entries on the coarse grid.
 413
 414    grid_shape: OpShape
 415        (M, N) shape of the coarse grid.
 416
 417    parallel: bool
 418        If ``true`` , use Dask to evaluate the following methods:
 419
 420        * ``.apply()``
 421        * ``.prox()``
 422        * ``.grad()``
 423        * ``.adjoint()``
 424        * ``.[co]gram().[apply,adjoint]()``
 425
 426    Returns
 427    -------
 428    op: OpT
 429        Block-defined operator. (See below for examples.)
 430
 431    Notes
 432    -----
 433    * Blocks on the same row/column must have the same ``codim`` / ``dim`` s.
 434    * Each row/column of the coarse grid **must** contain at least one entry.
 435    * ``parallel=True`` only parallelizes execution when inputs to listed methods are NUMPY arrays.
 436
 437    .. Warning::
 438
 439       When processing Dask inputs, or when ``parallel=True``, operators which are not thread-safe may produce incorrect
 440       results.
 441       There is no easy way to ensure thread-safety at the level of :py:mod:`~pyxu.operator.blocks` without impacting
 442       performance of all operators involved.
 443       Users are thus responsible for executing block-defined operators correctly, i.e., if thread-unsafe operators are
 444       involved, stick to NumPy/CuPy inputs.
 445
 446    Examples
 447    --------
 448
 449    .. code::
 450
 451       >>> coo_block(
 452       ...     ([A(500,1000), B(1,1000), C(500,500), D(1,3)],  # data
 453       ...      [
 454       ...       [0, 1, 0, 2],  # i
 455       ...       [0, 0, 2, 1],  # j
 456       ...      ]),
 457       ...     grid_shape=(3, 3),
 458       ... )
 459
 460       | coarse_idx |      0       |    1    |      2      |
 461       |------------|--------------|---------|-------------|
 462       |          0 | A(500, 1000) |         | C(500, 500) |
 463       |          1 | B(1, 1000)   |         |             |
 464       |          2 |              | D(1, 3) |             |
 465    """
 466    op = _COOBlock(
 467        ops=ops,
 468        grid_shape=grid_shape,
 469        parallel=parallel,
 470    ).op()
 471    return op

 474def _parallelize(func: cabc.Callable) -> cabc.Callable:
 475    # Parallelize execution of func() under conditions.
 476    #
 477    # * func() must be one of the arithmetic methods [apply,prox,grad,adjoint,pinv]()
 478    # * the `_parallel` attribute must be attached to the instance for it to parallelize execution
 479    #   over NUMPY inputs.
 480
 481    @functools.wraps(func)
 482    def wrapper(*ARGS, **KWARGS):
 483        func_args = pxu.parse_params(func, *ARGS, **KWARGS)
 484
 485        arr = func_args.get("arr", None)
 486        N = pxd.NDArrayInfo
 487        parallelize = ARGS[0]._parallel and (N.from_obj(arr) == N.NUMPY)
 488
 489        # [2022.09.26] Sepand Kashani
 490        # Q: Why do we parallelize only for NUMPY inputs and not CUPY?
 491        # A: Given an arithmetic method `f`, there is no obligation (for DASK inputs) to satisfy the
 492        #    relation `f(arr).chunk_type == arr.chunk_type`.
 493        #    In particular the relationship does not hold for LinFunc.[grad, adjoint]() unless the
 494        #    user provides a special implementation for DASK inputs.
 495        #    Consequence: the sum() / xp.concatenate() instructions in _COOBlock() may:
 496        #    * fail due to array-type mismatch; or
 497        #    * induce unnecessary CPU<>GPU transfers.
 498
 499        if parallelize:
 500            xp = N.DASK.module()
 501            func_args.update(arr=xp.array(arr, dtype=arr.dtype))
 502        else:
 503            pass
 504
 505        out = func(**func_args)
 506        f = {True: pxu.compute, False: lambda _: _}[parallelize]
 507        return f(out)
 508
 509    return wrapper

 512class _COOBlock:  # See coo_block() for a detailed description.
 513    def __init__(
 514        self,
 515        ops: tuple[
 516            cabc.Sequence[pxt.OpT],
 517            tuple[
 518                cabc.Sequence[pxt.Integer],
 519                cabc.Sequence[pxt.Integer],
 520            ],
 521        ],
 522        grid_shape: pxt.OpShape,
 523        parallel: bool,
 524    ):
 525        self._grid_shape = tuple(grid_shape)
 526        self._init_spec(ops)
 527        self._parallel = bool(parallel)
 528
 529    def op(self) -> pxt.OpT:
 530        """
 531        Returns
 532        -------
 533        op: pxt.OpT
 534            Synthesized operator given inputs to
 535            :py:meth:`~pyxu.operator.blocks._COOBlock.__init__`.
 536        """
 537        blk = self._block
 538        if len(blk) == 1:
 539            _, op = blk.popitem()
 540        else:
 541            import pyxu.abc.arithmetic as arithmetic
 542
 543            op = self._infer_op()
 544            op._block = self._block  # embed for introspection
 545            op._block_offset = self._block_offset  # embed for introspection
 546            op._grid_shape = self._grid_shape  # embed for introspection
 547            op._parallel = self._parallel  # embed for introspection
 548            for p in op.properties():
 549                for name in p.arithmetic_methods():
 550                    func = getattr(self.__class__, name)
 551                    setattr(op, name, types.MethodType(func, op))
 552            arithmetic.Rule._propagate_constants(op)
 553        return op
 554
 555    def _init_spec(self, ops):
 556        """
 557        Transform input into standardized form.
 558
 559        Returns
 560        -------
 561        block: (int, int) -> pxt.OpT
 562            coarse_grid index -> Operator at that location.
 563        block_offset: (int, int) -> (int, int)
 564            coarse_grid index -> (top-left) fine_grid index.
 565        """
 566        data, (i, j) = ops
 567        N_row, N_col, N_block = *self._grid_shape, len(data)
 568        msg = "Incorrect COO parametrization"
 569        assert N_block == len(i) == len(j), msg
 570        assert 0 < N_block <= N_row * N_col, msg
 571        assert 0 <= min(i) <= max(i) < N_row, msg
 572        assert 0 <= min(j) <= max(j) < N_col, msg
 573
 574        # block dimensions are compatible.
 575        row = collections.defaultdict(list)  # row_idx -> [block]
 576        col = collections.defaultdict(list)  # col_idx -> [block]
 577        for d, _i, _j in zip(data, i, j):
 578            row[_i].append(d)
 579            col[_j].append(d)
 580        msg = lambda dim, idx, dom: f"All sub-operators on {dim} {idx} must have same {dom} size."
 581        for k, v in row.items():
 582            assert len({_.codim for _ in v}) == 1, msg("row", k, "codomain")
 583        for k, v in col.items():
 584            assert len({_.dim for _ in v}) == 1, msg("column", k, "domain")
 585
 586        # no empty lines/columns in coarse grid.
 587        msg = lambda _: f"Coarse grid contains empty {_}: cannot infer fine-grid dimensions."
 588        assert len(row) == N_row, msg("rows")
 589        assert len(col) == N_col, msg("columns")
 590
 591        # ---------------------------------------------------------------------
 592        # create block
 593        block = dict()  # coarse_grid(int, int) -> pxt.OpT
 594        for d, _i, _j in zip(data, i, j):
 595            block[(_i, _j)] = d.squeeze()
 596        self._block = block
 597
 598        # create block_offset
 599        block_offset = dict()  # coarse_grid(int, int) -> fine_grid(int, int)
 600        row_offset = np.cumsum([row[k][0].codim for k in range(N_row)])
 601        col_offset = np.cumsum([col[k][0].dim for k in range(N_col)])
 602        for i in range(N_row):
 603            for j in range(N_col):
 604                _i = 0 if (i == 0) else row_offset[i - 1]
 605                _j = 0 if (j == 0) else col_offset[j - 1]
 606                block_offset[(i, j)] = _i, _j
 607        self._block_offset = block_offset
 608
 609    def _infer_op(self) -> pxt.OpT:
 610        # Create the abstract COO-operator to which methods will be assigned.
 611        blk = self._block  # shorthand
 612
 613        row = collections.defaultdict(list)
 614        col = collections.defaultdict(list)
 615        for (r, c), op in blk.items():
 616            row[r].append(op)
 617            col[c].append(op)
 618        N_row, N_col = len(row), len(col)
 619        op_codim = sum(ops[0].codim for ops in row.values())
 620        op_dim = sum(ops[0].dim for ops in col.values())
 621
 622        P = pxo.Property
 623        properties = set.intersection(*[set(op.properties()) for op in blk.values()])
 624        if op_codim > 1:
 625            properties -= {
 626                P.FUNCTIONAL,
 627                P.PROXIMABLE,
 628                P.DIFFERENTIABLE_FUNCTION,
 629                P.QUADRATIC,
 630            }
 631
 632        if all(  # block_diag case
 633            [
 634                N_row == N_col == len(blk),
 635                all((r, r) in blk for r in range(N_row)),
 636            ]
 637        ):
 638            pass
 639        elif op_codim == 1:  # hstack case: special treatment of quadratics
 640            if all(op.has(P.QUADRATIC) for op in blk.values()):
 641                properties.add(P.QUADRATIC)
 642            elif any(op.has(P.QUADRATIC) for op in blk.values()):
 643                # possible quadratic if all other terms are linear.
 644                non_quad = [op for op in blk.values() if not op.has(P.QUADRATIC)]
 645                if all(op.has(P.LINEAR) for op in non_quad):
 646                    properties.add(P.QUADRATIC)
 647        else:  # (1) vstack case or (2) arbitrary fill-in case => non-functionals
 648            properties &= {
 649                P.CAN_EVAL,
 650                P.DIFFERENTIABLE,
 651                P.LINEAR,
 652            }
 653            if (op_codim == op_dim) and (P.LINEAR in properties):
 654                properties.add(P.LINEAR_SQUARE)
 655
 656        klass = pxo.Operator._infer_operator_type(properties)
 657        op = klass(shape=(op_codim, op_dim))
 658        return op

 661    @pxrt.enforce_precision(i="arr")
 662    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
 663        # compute blocks
 664        parts = dict()
 665        for idx, op in self._block.items():
 666            offset = self._block_offset[idx][1]
 667            p = op.apply(arr[..., offset : offset + op.dim])
 668            parts[idx] = p
 669
 670        # row-oriented reduction (via +)
 671        rows = collections.defaultdict(list)
 672        for (r, c), p in parts.items():
 673            rows[r].append(p)
 674        cols = [sum(rows[r]) for r in range(len(rows))]
 675
 676        # concatenate to super-column
 677        xp = pxu.get_array_module(arr)
 678        out = xp.concatenate(cols, axis=-1)
 679        return out

 684    def estimate_lipschitz(self, **kwargs) -> pxt.Real:
 685        Ls_all = np.zeros(self._grid_shape)  # squared Lipschitz constant of each block.
 686
 687        no_eval = "__rule" in kwargs
 688        if no_eval:
 689            for (r, c), op in self._block.items():
 690                Ls_all[r, c] = op.lipschitz**2
 691        elif self.has(pxo.Property.LINEAR):
 692            L = self.__class__.estimate_lipschitz(self, **kwargs)
 693            return L
 694        else:
 695            for (r, c), op in self._block.items():
 696                Ls_all[r, c] = op.estimate_lipschitz(**kwargs) ** 2
 697
 698        # [non-linear case] Various upper bounds apply depending on how the blocks are organized:
 699        #   * vertical alignment: L**2 = sum(L_k**2)
 700        #   * horizontal alignment: L**2 = sum(L_k**2)
 701        #   * block-diagonal alignment: L**2 = max(L_k**2)
 702        #   * arbitrary 2D alignment: L**2 = sum(L_k**2)
 703        #     [obtained via vertical+horizontal composition (or vice-versa)]
 704        if np.allclose(Ls_all.sum(), Ls_all.diagonal().sum()):  # block-diag case
 705            L = np.sqrt(Ls_all.max())
 706        else:
 707            L = np.sqrt(Ls_all.sum())
 708        return L

 710    def _expr(self) -> tuple:
 711        class _Block(pxo.Operator):
 712            def __init__(self, idx: tuple[int, int], op: pxt.OpT):
 713                super().__init__(shape=op.shape)
 714                self._name = op._name
 715                self._idx = tuple(idx)
 716                self._op = op
 717
 718            def _expr(self) -> tuple:
 719                head = "block[" + ", ".join(map(str, self._idx)) + "]"
 720                return (head, self._op)
 721
 722        # head = coo_block[grid_shape]
 723        head = "coo_block[" + ", ".join(map(str, self._grid_shape)) + "]"
 724
 725        # tail = sequence of sub-blocks, encapsulated in _Block to obtain hierarchical view.
 726        tail = [_Block(idx=k, op=op) for (k, op) in self._block.items()]
 727        return (head, *tail)

 746    def _quad_spec(self):
 747        if not self.has(pxo.Property.QUADRATIC):
 748            raise NotImplementedError
 749
 750        parts = dict()
 751        for idx, op in self._block.items():
 752            if op.has(pxo.Property.QUADRATIC):
 753                _Q, _c, _t = op._quad_spec()
 754            else:  # op is necessarily LINEAR
 755                from pyxu.operator.linop import NullOp
 756
 757                _Q = NullOp(shape=(op.dim, op.dim)).asop(pxo.PosDefOp)
 758                _c = op
 759                _t = 0
 760            parts[idx] = _Q, _c, _t
 761
 762        parts = [parts[k] for k in sorted(parts.keys())]  # re-ordering
 763        Q, c, t = zip(*parts)
 764        return (block_diag(Q), hstack(c), sum(t))

 789    def estimate_diff_lipschitz(self, **kwargs) -> pxt.Real:
 790        if not self.has(pxo.Property.DIFFERENTIABLE):
 791            raise NotImplementedError
 792
 793        dLs_all = np.zeros(self._grid_shape)  # squared diff-Lipschitz constant of each block.
 794
 795        no_eval = "__rule" in kwargs
 796        if no_eval:
 797            for (r, c), op in self._block.items():
 798                dLs_all[r, c] = op.diff_lipschitz**2
 799        elif self.has(pxo.Property.QUADRATIC):
 800            dL = pxo.QuadraticFunc.estimate_diff_lipschitz(self, **kwargs)
 801            return dL
 802        elif self.has(pxo.Property.LINEAR):
 803            dL = 0
 804            return dL
 805        else:
 806            for (r, c), op in self._block.items():
 807                dLs_all[r, c] = op.estimate_diff_lipschitz(**kwargs) ** 2
 808
 809        # Various upper bounds apply depending on how the blocks are organized:
 810        #   * vertical alignment: dL**2 = sum(dL_k**2)
 811        #   * horizontal alignment: dL**2 = sum(dL_k**2)
 812        #   * block-diagonal alignment: dL**2 = max(dL_k**2)
 813        #   * arbitrary 2D alignment: dL**2 = sum(dL_k**2)
 814        #     [obtained via vertical+horizontal composition (or vice-versa)]
 815        if np.allclose(dLs_all.sum(), dLs_all.diagonal().sum()):  # block-diag case
 816            dL = np.sqrt(dLs_all.max())
 817        else:
 818            dL = np.sqrt(dLs_all.sum())
 819        return dL

 889    def gram(self) -> pxt.OpT:
 890        if not self.has(pxo.Property.LINEAR):
 891            raise NotImplementedError
 892
 893        blk = self._block  # shorthand
 894        N_row, N_col = self._grid_shape
 895
 896        # Determine operator(s) which will occupy position (r,c) on coarse grid.
 897        ops = collections.defaultdict(list)  # coarse_grid(int, int) -> [pxt.OpT]
 898        for r, c in itertools.product(range(N_col), repeat=2):
 899            for k in range(N_row):
 900                if ((k, r) in blk) and ((k, c) in blk):
 901                    if r == c:
 902                        _op = blk[(k, r)].gram()
 903                    else:
 904                        _op = blk[(k, r)].T * blk[(k, c)]
 905                    ops[(r, c)].append(_op)
 906
 907        # ops[(r,c)] should be reduced (via +) to form a single operator per (r,c)-entry.
 908        # It is inefficient however to chain so many operators together via AddRule().
 909        # G.[apply,adjoint]() are thus redefined to improve performance.
 910        data, i, j = [], [], []
 911        for (r, c), _ops in ops.items():
 912            klass = pxo.SelfAdjointOp if (r == c) else pxo.LinOp
 913            _op = klass(shape=_ops[0].shape)  # .[apply|adjoint]() overridden below.
 914            _op._ops = _ops  # embed for introspection
 915
 916            @pxrt.enforce_precision(i="arr")
 917            def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
 918                parts = [op.apply(arr) for op in _._ops]
 919                out = sum(parts)
 920                return out
 921
 922            @pxrt.enforce_precision(i="arr")
 923            def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
 924                parts = [op.adjoint(arr) for op in _._ops]
 925                out = sum(parts)
 926                return out
 927
 928            _op.apply = types.MethodType(op_apply, _op)
 929            _op.adjoint = types.MethodType(op_adjoint, _op)
 930
 931            data.append(_op)
 932            i.append(r)
 933            j.append(c)
 934        G = (
 935            _COOBlock(
 936                ops=(data, (i, j)),
 937                grid_shape=(N_col, N_col),
 938                parallel=self._parallel,
 939            )
 940            .op()
 941            .asop(pxo.SelfAdjointOp)
 942        )
 943        return G.squeeze()
