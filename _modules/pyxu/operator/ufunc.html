 256        self.diff_lipschitz = 3 * np.sqrt(3) / 8
 257        #   max_{x \in R} |arctan''(x)|
 258        # = max_{x \in R} |2x / (1+x^2)^2|
 259        # = 3 \sqrt(3) / 8  [at x = +- 1/\sqrt(3)]

 346class Tanh(pxa.DiffMap):
 347    r"""
 348    Hyperbolic tangent, element-wise.
 349
 350    Notes
 351    -----
 352    * :math:`f(x) = \tanh(x)`
 353    * :math:`f'(x) = 1 - \tanh^{2}(x)`
 354    * :math:`\vert f(x) - f(y) \vert \le L \vert x - y \vert`, with Lipschitz constant :math:`L = 1`.
 355
 356      (Reason: :math:`\vert f'(x) \vert` is bounded by :math:`L` at :math:`x = 0`.)
 357    * :math:`\vert f'(x) - f'(y) \vert \le \partial L \vert x - y \vert`, with diff-Lipschitz constant :math:`\partial L
 358      = 4 / 3 \sqrt{3}`.
 359
 360      (Reason: :math:`\vert f''(x) \vert` is bounded by :math:`\partial L` at :math:`x = \frac{1}{2} \ln(2 \pm
 361      \sqrt{3})`.
 362    """

 365        super().__init__(
 366            dim_shape=dim_shape,
 367            codim_shape=dim_shape,
 368        )
 369        self.lipschitz = 1
 370        self.diff_lipschitz = 4 / (3 * np.sqrt(3))
 371        #   max_{x \in R} |tanh''(x)|
 372        # = max_{x \in R} |-2 tanh(x) [1 - tanh(x)^2|
 373        # = 4 / (3 \sqrt(3))  [at x = ln(2 +- \sqrt(3))]

 513    Notes

 518

 524    """

 527        super().__init__(
 528            dim_shape=dim_shape,
 529            codim_shape=dim_shape,
 530        )
 531        self.lipschitz = np.inf
 532        self.diff_lipschitz = np.inf
 533        self._base = base
