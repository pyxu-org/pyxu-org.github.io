 15def from_source(
 16    cls: pxt.OpC,
 17    shape: pxt.OpShape,
 18    embed: dict = None,
 19    vectorize: pxt.VarName = frozenset(),
 20    vmethod: typ.Union[str, dict] = None,
 21    enforce_precision: pxt.VarName = frozenset(),
 22    **kwargs,
 23) -> pxt.OpT:
 24    r"""
 25    Define an :py:class:`~pyxu.abc.operator.Operator` from low-level constructs.
 26
 27    Parameters
 28    ----------
 29    cls: OpC
 30        Operator sub-class to instantiate.
 31    shape: OpShape
 32        (N, M) operator shape.
 33    embed: dict
 34        (k[str], v[value]) pairs to embed into the created operator.
 35
 36        `embed` is useful to attach extra information to synthesized
 37        :py:class:`~pyxu.abc.operator.Operator` used by arithmetic methods.
 38    kwargs: dict
 39        (k[str], v[callable]) pairs to use as arithmetic methods.
 40
 41        Keys must be entries from :py:meth:`~pyxu.abc.operator.Property.arithmetic_methods`.
 42
 43        Omitted arithmetic attributes/methods default to those provided by `cls`.
 44    vectorize: VarName
 45        Arithmetic methods to vectorize.
 46
 47        `vectorize` is useful if an arithmetic method provided to `kwargs` (ex:
 48        :py:meth:`~pyxu.abc.operator.Map.apply`):
 49
 50        * does not support stacking dimensions; OR
 51        * does not support DASK inputs.
 52    vmethod: str, dict
 53        Vectorization strategy. (See :py:class:`~pyxu.util.operator.vectorize`.)
 54
 55        Different strategies can be applied per arithmetic method via a dictionary.
 56    enforce_precision: VarName
 57        Arithmetic methods to make compliant with Pyxu's runtime FP-precision.
 58
 59        `enforce_precision` is useful if an arithmetic method provided to `kwargs` (ex:
 60        :py:meth:`~pyxu.abc.operator.Map.apply`) is unaware of Pyxu's runtime FP context.
 61
 62    Returns
 63    -------
 64    op: OpT
 65        (N, M) Pyxu-compliant operator.
 66
 67    Notes
 68    -----
 69    * If provided, arithmetic methods must abide exactly to the Pyxu interface.
 70      In particular, the following arithmetic methods, if supplied, **must** have
 71      the following interface:
 72
 73      .. code-block:: python3
 74
 75         def apply(self, arr: pxt.NDArray) -> pxt.NDArray
 76         def grad(self, arr: pxt.NDArray) -> pxt.NDArray
 77         def adjoint(self, arr: pxt.NDArray) -> pxt.NDArray
 78         def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray
 79         def pinv(self, arr: pxt.NDArray, damp: pxt.Real) -> pxt.NDArray
 80
 81      Moreover, the methods above **must** accept ``(..., M)``-shaped inputs for ``arr``.
 82      If this does not hold, consider populating `vectorize`.
 83
 84    * Auto-vectorization consists in decorating `kwargs`-specified arithmetic methods with
 85      :py:func:`~pyxu.util.operator.vectorize`.
 86      Auto-vectorization may be less efficient than explicitly providing a vectorized implementation.
 87
 88    * Enforcing precision consists in decorating `kwargs`-specified arithmetic methods with
 89      :py:func:`~pyxu.runtime.enforce_precision`.
 90      Not all arithmetic methods can be made runtime FP-precision compliant.
 91      It is thus recommended to make arithmetic methods precision-compliant manually.
 92
 93    Examples
 94    --------
 95    Creation of the custom element-wise differential operator :math:`f(\mathbf{x}) = \mathbf{x}^{2}`.
 96
 97    .. code-block:: python3
 98
 99       N = 5
100       f = from_source(
101           cls=pyxu.abc.DiffMap,
102           shape=(N, N),
103           apply=lambda _, arr: arr**2,
104       )
105       x = np.arange(N)
106       y = f(x)  # [0, 1, 4, 9, 16]
107       dL = f.diff_lipschitz  # inf (default value provided by DiffMap class.)
108
109    In practice we know that :math:`f` has a finite-valued diff-Lipschitz constant.
110    It is thus recommended to set it too when instantiating via ``from_source``:
111
112    .. code-block:: python3
113
114       N = 5
115       f = from_source(
116           cls=pyxu.abc.DiffMap,
117           shape=(N, N),
118           embed=dict(  # special form to set (diff-)Lipschitz attributes via from_source()
119               _diff_lipschitz=2,
120           ),
121           apply=lambda _, arr: arr**2,
122       )
123       x = np.arange(N)
124       y = f(x)  # [0, 1, 4, 9, 16]
125       dL = f.diff_lipschitz  # 2  <- instead of inf
126    """
127    if embed is None:
128        embed = dict()
129
130    if isinstance(vectorize, str):
131        vectorize = (vectorize,)
132    vectorize = frozenset(vectorize)
133
134    if isinstance(enforce_precision, str):
135        enforce_precision = (enforce_precision,)
136    enforce_precision = frozenset(enforce_precision)
137
138    src = _FromSource(
139        cls=cls,
140        shape=shape,
141        embed=embed,
142        vectorize=vectorize,
143        vmethod=vmethod,
144        enforce_precision=enforce_precision,
145        **kwargs,
146    )
147    op = src.op()
148    return op

151class _FromSource:  # See from_source() for a detailed description.
152    def __init__(
153        self,
154        cls: pxt.OpC,
155        shape: pxt.OpShape,
156        embed: dict,
157        vectorize: frozenset[str],
158        vmethod: typ.Union[str, dict],
159        enforce_precision: frozenset[str],
160        **kwargs,
161    ):
162        assert cls in pxo._core_operators(), f"Unknown Operator type: {cls}."
163        self._op = cls(shape)  # ensure shape well-formed
164
165        # Arithmetic methods to attach to `_op`.
166        meth = frozenset.union(*[p.arithmetic_methods() for p in pxo.Property])
167        if not (set(kwargs) <= meth):
168            msg_head = "Unknown arithmetic methods:"
169            unknown = set(kwargs) - meth
170            msg_tail = ", ".join([f"{name}()" for name in unknown])
171            raise ValueError(f"{msg_head} {msg_tail}")
172        self._kwargs = kwargs
173
174        # Extra attributes to attach to `_op`.
175        assert isinstance(embed, cabc.Mapping)
176        self._embed = embed
177
178        # Add-on functionality to enable.
179        self._vkwargs = self._parse_vectorize(vectorize, vmethod)
180        self._vectorize = vectorize
181
182        self._ekwargs = self._parse_precision(enforce_precision)
183        self._enforce_fp = enforce_precision
184
185    def op(self) -> pxt.OpT:
186        _op = self._op  # shorthand
187        for p in _op.properties():
188            for name in p.arithmetic_methods():
189                if func := self._kwargs.get(name, False):
190                    # vectorize() & enforce_precision() do NOT kick in for default-provided methods.
191                    # (We assume they are Pyxu-compliant from the start.)
192
193                    if name in self._vectorize:
194                        decorate = pxu.vectorize(**self._vkwargs[name])
195                        func = decorate(func)
196
197                    if name in self._enforce_fp:
198                        decorate = pxrt.enforce_precision(**self._ekwargs[name])
199                        func = decorate(func)
200
201                    setattr(_op, name, types.MethodType(func, _op))
202
203        # Embed extra attributes
204        for name, attr in self._embed.items():
205            setattr(_op, name, attr)
206
207        return _op

209    def _parse_vectorize(
210        self,
211        vectorize: frozenset[str],
212        vmethod: typ.Union[str, dict],
213    ):
214        vsize = dict(  # `codim` hint for vectorize()
215            apply=self._op.codim,
216            prox=self._op.dim,
217            grad=self._op.dim,
218            adjoint=self._op.dim,
219            pinv=self._op.dim,
220        )
221
222        if not (vectorize <= set(vsize)):  # un-recognized arithmetic method
223            msg_head = "Can only vectorize arithmetic methods"
224            msg_tail = ", ".join([f"{name}()" for name in vsize])
225            raise ValueError(f"{msg_head} {msg_tail}")
226
227        vmethod_default = pxu.parse_params(
228            pxu.vectorize,
229            i="bogus",  # doesn't matter
230        )["method"]
231        if vmethod is None:
232            vmethod = vmethod_default
233        if isinstance(vmethod, str):
234            vmethod = {name: vmethod for name in vsize}
235
236        vkwargs = {
237            name: dict(
238                i="arr",  # Pyxu arithmetic methods broadcast along parameter `arr`.
239                method=vmethod.get(name, vmethod_default),
240                codim=vsize[name],
241            )
242            for name in vsize
243        }
244        return vkwargs
245
246    def _parse_precision(self, enforce_precision: frozenset[str]):
247        ekwargs = dict(
248            # Pyxu arithmetic methods enforce FP-precision along these parameters.
249            apply=dict(i="arr"),
250            prox=dict(i=("arr", "tau")),
251            grad=dict(i="arr"),
252            adjoint=dict(i="arr"),
253            pinv=dict(i=("arr", "damp")),
254            svdvals=dict(),
255            trace=dict(),
256        )
257
258        if not (enforce_precision <= set(ekwargs)):
259            msg_head = "Can only enforce precision on arithmetic methods"
260            msg_tail = ", ".join([f"{name}()" for name in ekwargs])
261            raise ValueError(f"{msg_head} {msg_tail}")
262        return ekwargs
