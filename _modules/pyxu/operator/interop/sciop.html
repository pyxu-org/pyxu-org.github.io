18def from_sciop(cls: pxt.OpC, sp_op: spsl.LinearOperator) -> pxt.OpT:
19    r"""
20    Wrap a :py:class:`scipy.sparse.linalg.LinearOperator` as a
21    :py:class:`~pyxu.abc.operator.LinOp` (or sub-classes thereof).
22
23    Parameters
24    ----------
25    sp_op: ~scipy.sparse.linalg.LinearOperator
26        (N, M) Linear CPU/GPU operator compliant with SciPy's interface.
27
28    Returns
29    -------
30    op: OpT
31        (N, M) Pyxu-compliant linear operator.
32    """
33    assert cls.has(pxo.Property.LINEAR)
34
35    if sp_op.dtype not in [_.value for _ in pxrt.Width]:
36        warnings.warn(
37            "Computation may not be performed at the requested precision.",
38            pxw.PrecisionWarning,
39        )

41    # [r]matmat only accepts 2D inputs -> reshape apply|adjoint inputs as needed.
42
43    @pxrt.enforce_precision(i="arr")
44    def op_apply(_, arr: pxt.NDArray) -> pxt.NDArray:
45        sh = arr.shape[:-1]
46        arr = arr.reshape(-1, _.dim)
47        out = _._sp_op.matmat(arr.T).T
48        out = out.reshape(*sh, _.codim)
49        return out
50
51    @pxrt.enforce_precision(i="arr")
52    def op_adjoint(_, arr: pxt.NDArray) -> pxt.NDArray:
53        sh = arr.shape[:-1]
54        arr = arr.reshape(-1, _.codim)
55        out = _._sp_op.rmatmat(arr.T).T
56        out = out.reshape(*sh, _.dim)
57        return out
58
59    def op_asarray(_, **kwargs) -> pxt.NDArray:
60        # Determine XP-module accepted by sci_op, then compute array-representation.
61        for ndi in [
62            pxd.NDArrayInfo.NUMPY,
63            pxd.NDArrayInfo.CUPY,
64        ]:
65            try:
66                cls = _.__class__
67                _A = cls.asarray(_, xp=ndi.module(), dtype=_._sp_op.dtype)
68                break
69            except Exception:
70                pass
71
72        # Cast to user specs.
73        xp = kwargs.get("xp", pxd.NDArrayInfo.NUMPY.module())
74        dtype = kwargs.get("dtype", pxrt.getPrecision().value)
75        A = xp.array(pxu.to_NUMPY(_A), dtype=dtype)
76        return A
77
78    def op_expr(_) -> tuple:
79        return ("from_sciop", _._sp_op)
80
81    op = pxsrc.from_source(
82        cls=cls,
83        shape=sp_op.shape,
84        apply=op_apply,
85        adjoint=op_adjoint,
86        asarray=op_asarray,
87        _expr=op_expr,
88    )
89    op._sp_op = sp_op
90
91    return op
