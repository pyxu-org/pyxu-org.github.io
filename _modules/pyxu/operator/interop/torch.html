 42def _traceable(f):
 43    # Needed to compile functorch transforms. See this issue: https://github.com/pytorch/pytorch/issues/98822
 44    f = dynamo.allow_in_graph(f)
 45
 46    @wraps(f)
 47    def wrapper(*args, **kwargs):
 48        return f(*args, **kwargs)
 49
 50    return wrapper

 53def astensor(arr: pxt.NDArray, requires_grad: bool = False) -> TorchTensor:
 54    r"""
 55    Convert a Numpy-like array into a PyTorch tensor, sharing data, dtype and device.
 56
 57    Parameters
 58    ----------
 59    arr: NDArray
 60        Input array.
 61    requires_grad: bool
 62        If autograd should record operations on the returned tensor.
 63
 64    Returns
 65    -------
 66    tensor: torch.Tensor
 67        Output tensor.
 68
 69    Notes
 70    -----
 71    The returned tensor and input array share the same memory.
 72    Modifications to the tensor will be reflected in the ndarray and vice versa.
 73    The returned tensor is not resizable.
 74    """
 75    if pxd.NDArrayInfo.from_obj(arr) == pxd.NDArrayInfo.CUPY:
 76        with torch.device("cuda", arr.device.id):
 77            return torch.as_tensor(arr).requires_grad_(requires_grad)
 78    else:
 79        return torch.from_numpy(arr).requires_grad_(requires_grad)

 82def asarray(tensor: TorchTensor) -> pxt.NDArray:
 83    r"""
 84    Convert a PyTorch tensor into a Numpy-like array, sharing data, dtype and device.
 85
 86    Parameters
 87    ----------
 88    tensor: torch.Tensor
 89        Input tensor.
 90
 91    Returns
 92    -------
 93    arr: NDArray
 94        Output array.
 95
 96    Notes
 97    -----
 98    The returned array and input tensor share the same memory.
 99    Modifications to the tensor will be reflected in the ndarray and vice versa.
100    """
101    if tensor.get_device() == -1:
102        return tensor.detach().numpy(force=False)
103    else:
104        cp = pxd.NDArrayInfo.CUPY.module()
105        with cp.cuda.Device(tensor.get_device()):
106            return cp.asarray(tensor.detach())

148            unsupported = set(self._kwargs) - self._meth

152    def op(self) -> pxt.OpT:
153        # Idea: modify `**kwargs` from constructor to [when required]:
154        #   1. auto-define omitted methods.            [_infer_missing()]
155        #   2. auto-vectorize via vmap().              [_auto_vectorize()]
156        #   3. JIT-compile via compile().              [_compile()]
157        #   4. TORCH<>NUMPY/CUPY conversions.          [_interface()]
158        #   Note: JIT-compilation is currently deactivated due to the undocumented interaction of torch.func transforms and
159        #   torch.compile. Will be reactivated once torch.func goes out of beta.
160
161        self._infer_missing()
162        self._compile()
163        self._auto_vectorize()
164        t_state, kwargs = self._interface()
165
166        _op = pxsrc.from_source(
167            cls=self._op.__class__,
168            shape=self._op.shape,
169            embed=dict(
170                _batch_size=self._batch_size,
171                _dtype=self._dtype,
172                _jit=self._jit,
173                _enable_warnings=self._enable_warnings,
174                _name=self._name,
175                _meta=self._meta,
176                _coerce=self._coerce,
177                _torch=t_state,
178            ),
179            # vectorize=None,  # see top-level comment.
180            # vmethod=None,    #
181            **kwargs,
182        )
183        return _op

185    def _infer_missing(self):
186        # The following methods must be auto-inferred if missing from `kwargs`:
187        #
188        #     grad(), adjoint()
189        #
190        # Missing methods are auto-inferred via auto-diff rules and added to `_kwargs`.
191        # At the end of _infer_missing(), all torch-funcs required for _interface() have been added to
192        # `_kwargs`.
193        #
194        # Notes
195        # -----
196        # This method does NOT produce vectorized implementations: _auto_vectorize() is responsible
197        # for this.
198        self._vectorize = set(self._vectorize)  # to allow updates below
199
200        nl_difffunc = all(  # non-linear diff-func
201            [
202                self._op.has(pxo.Property.DIFFERENTIABLE_FUNCTION),
203                not self._op.has(pxo.Property.LINEAR),
204            ]
205        )
206        if nl_difffunc and ("grad" not in self._kwargs):
207
208            def f_grad(tensor: TorchTensor) -> TorchTensor:
209                grad = functorch.grad(self._kwargs["apply"])
210                return grad(tensor)
211
212            self._vectorize.add("grad")
213            self._kwargs["grad"] = f_grad
214
215        non_selfadj = all(  # linear, but not self-adjoint
216            [
217                self._op.has(pxo.Property.LINEAR),
218                not self._op.has(pxo.Property.LINEAR_SELF_ADJOINT),
219            ]
220        )
221        if non_selfadj and ("adjoint" not in self._kwargs):
222
223            def f_adjoint(tensor: TorchTensor) -> TorchTensor:
224                # codim -> dim
225                f = self._kwargs["apply"]
226                primal = torch.zeros_like(tensor, shape=(self._op.dim,))
227                _, f_vjp = functorch.vjp(f, primal)
228                out = f_vjp(tensor)[0]  # f_vjp returns a tuple
229                return out  # size dim
230
231            self._vectorize.add("adjoint")
232            self._kwargs["adjoint"] = f_adjoint
233
234        self._vectorize = frozenset(self._vectorize)
235
236    def _compile(self):
237        # JIT-compile user-specified [or _infer_missing()-added] arithmetic methods via torch.compile().
238        #
239        # Modifies `_kwargs` to hold compiled torch-funcs.
240        # Note: Currently deactivated until torch.func goes out of beta.
241
242        if self._jit:
243            for name in self._kwargs:
244                if name in self._meth:
245                    func = self._kwargs[name]  # necessarily torch_func
246                    self._kwargs[name] = torch.compile(func)
247        else:
248            pass
249
250    def _auto_vectorize(self):
251        # Vectorize user-specified [or _infer_missing()-added] arithmetic methods via torch.vmap().
252        #
253        # Modifies `_kwargs` to hold vectorized torch-funcs.
254
255        for name in self._kwargs:
256            if name in self._vectorize:
257                func = self._kwargs[name]  # necessarily torch_func
258                if name in [
259                    "prox",
260                    "pinv",
261                ]:  # These methods have two arguments, but vectorization should be for the first argument only.
262                    self._kwargs[name] = _traceable(torch.vmap(func, in_dims=(0, None), chunk_size=self._batch_size))
263                else:
264                    self._kwargs[name] = _traceable(torch.vmap(func, chunk_size=self._batch_size))
265
266    def _interface(self):
267        # Arithmetic methods supplied in `kwargs`:
268        #
269        #     * take `torch.Tensor` inputs
270        #     * do not have the `self` parameter. (Reason: to be JIT-compatible.)
271        #
272        # This method creates modified arithmetic functions to match Pyxu's API, and `state`
273        # required for them to work.
274        #
275        # Returns
276        # -------
277        # t_state: dict
278        #     Torch functions referenced by wrapper arithmetic methods. (See below.)
279        # kwargs: dict
280        #     Pyxu-compatible functions which can be submitted to interop.from_source().
281
282        t_state = dict()
283        for name, obj in self._kwargs.items():
284            if name in self._meth:
285                t_state[name] = obj  # necessarily torch_func (potentially auto_vec/jitted at this stage)
286
287        kwargs = dict()
288        for name, obj in self._kwargs.items():
289            func = getattr(self.__class__, name)
290            kwargs[name] = func  # necessarily a pyxu_func
291
292        # Special cases.
293        # (Reason: not in `_kwargs` [c.f. from_torch() docstring], but need to be inferred.)
294        for name in (
295            "jacobian",
296            "_quad_spec",
297            "_expr",
298        ):  # "asarray"):
299            kwargs[name] = getattr(self.__class__, name)
300
301        return t_state, kwargs

303    def _coerce(self, arr: pxt.NDArray) -> pxt.NDArray:
304        # Coerce inputs (and raise a warning) in case of precision mis-matches.
305        if self._dtype is not None and (arr.dtype != self._dtype):
306            if self._enable_warnings:
307                msg = f"Precision mis-match! Input array was coerced to {self._dtype} precision automatically."
308                warnings.warn(msg, pxw.PrecisionWarning)
309            arr = arr.astype(self._dtype, copy=False)
310        return arr

314    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:
315        arr = self._coerce(arr)
316        tensor = astensor(arr.reshape(-1, self.dim))
317        func = self._torch["apply"]
318        out = asarray(func(tensor)).reshape(arr.shape[:-1] + (-1,))
319        return out

335    @pxrt.enforce_precision(i=["arr", "tau"])
336    def prox(self, arr: pxt.NDArray, tau: pxt.Real) -> pxt.NDArray:

342    @pxrt.enforce_precision(i=("arr", "damp"))
343    def pinv(self, arr: pxt.NDArray, damp: pxt.Real, **kwargs) -> pxt.NDArray:
344        arr = self._coerce(arr)
345        tensor = astensor(arr.reshape(-1, +self.codim))
346        func = self._torch["pinv"]
347        out = func(tensor, damp)  # positional args only if auto-vectorized.
348        return asarray(out).reshape(arr.shape[:-1] + (-1,))

350    @pxrt.enforce_precision(i="arr", o=False)
351    def jacobian(self, arr: pxt.NDArray) -> pxt.OpT:
352        try:
353            # Use the class' method if available ...
354            klass = self.__class__
355            op = klass.jacobian(self, arr)
356        except NotImplementedError:
357            # ... and fallback to auto-inference if undefined.
358            arr = self._coerce(arr)
359            primal = astensor(arr.reshape(self.dim))
360            f = self._torch["apply"]
361
362            def jf_apply(tan: TorchTensor) -> TorchTensor:
363                return functorch.jvp(f, primals=(primal,), tangents=(tan,))[1]
364
365            def jf_adjoint(cotan: TorchTensor) -> TorchTensor:
366                _, f_vjp = functorch.vjp(f, primal)
367                out = f_vjp(cotan)[0]  # f_vjp returns a tuple
368                return out
369
370            klass = pxo.LinFunc if (self.codim == 1) else pxo.LinOp
371            op = from_torch(
372                apply=jf_apply,
373                shape=self.shape,
374                cls=klass,
375                vectorize=("apply", "adjoint"),
376                batch_size=self._batch_size,
377                jit=self._jit,
378                dtype=self._dtype,
379                enable_warnings=self._enable_warnings,
380                name="TorchJacobian",
381                meta=self,
382                adjoint=jf_adjoint,
383            )
384        return op

386    def _quad_spec(self):
387        if self.has(pxo.Property.QUADRATIC):
388            # auto-infer (Q, c, t)
389            _grad = self._torch["grad"]
390
391            def Q_apply(tensor: TorchTensor) -> TorchTensor:
392                # \grad_{f}(x) = Q x + c = Q x + \grad_{f}(0)
393                # ==> Q x = \grad_{f}(x) - \grad_{f}(0)
394                z = torch.zeros_like(tensor)
395                out = _grad(tensor) - _grad(z)
396                return out
397
398            def c_apply(tensor: TorchTensor) -> TorchTensor:
399                z = torch.zeros_like(tensor)
400                c = _grad(z)
401                out = torch.sum(c * tensor, dtype=tensor.dtype, dim=-1, keepdim=True)
402                return out
403
404            Q = from_torch(
405                apply=Q_apply,
406                shape=(self.dim, self.dim),
407                cls=pxo.PosDefOp,
408                vectorize="apply",
409                batch_size=self._batch_size,
410                jit=self._jit,
411                dtype=self._dtype,
412                enable_warnings=self._enable_warnings,
413                name="TorchQuadSpec_Q",
414                meta=self,
415            )
416
417            c = from_torch(
418                apply=c_apply,
419                shape=(1, self.dim),
420                cls=pxo.LinFunc,
421                vectorize="apply",
422                batch_size=self._batch_size,
423                jit=self._jit,
424                dtype=self._dtype,
425                enable_warnings=self._enable_warnings,
426                name="TorchQuadSpec_c",
427                meta=self,
428            )
429
430            # We cannot know a-priori which backend the supplied torch-apply() function works with.
431            # Consequence: to compute `t`, we must try different backends until one works.
432            f = self._torch["apply"]
433            try:  # test a CPU implementation ...
434                with torch.device("cpu"):
435                    t = float(f(torch.zeros(self.dim)))
436            except Exception:  # ... and use GPU(s) if it doesn't work.
437                for id in range(torch.cuda.device_count()):
438                    try:
439                        with torch.device("gpu", id):
440                            t = float(f(torch.zeros(self.dim)))
441                    except Exception:
442                        continue
443
444            return (Q, c, t)
445        else:
446            raise NotImplementedError
447
448    def _expr(self):
449        torch_funcs = list(self._torch.keys())
450        if self._meta is not None:
451            torch_funcs.append(self._meta)
452        return ("from_torch", *torch_funcs)

455def from_torch(
456    apply: cabc.Callable,
457    shape: pxt.OpShape,
458    cls: pxt.OpC = pxo.Map,
459    vectorize: pxt.VarName = frozenset(),
460    batch_size: typ.Optional[int] = None,
461    jit: bool = False,
462    dtype: typ.Optional[pxt.DType] = None,
463    enable_warnings: bool = True,
464    name: str = "TorchOp",
465    meta=None,
466    **kwargs,
467) -> pxt.OpT:
468    r"""
469    Define an :py:class:`~pyxu.abc.operator.Operator` from Torch-based callables.
470
471    Parameters
472    ----------
473    apply: ~collections.abc.Callable
474        A Python function with single-element Tensor input/output. Defines the :py:meth:`~pyxu.abc.operator.Map.apply`
475        method of the operator: ``apply(x)==op.apply(x)``.
476    shape: OpShape
477        (N,M) shape of the operator, where N and M are the sizes of the output and input Tensors of ``apply`` respectively.
478    cls: OpT
479        Pyxu abstract base class to instantiate from.
480    vectorize: pxt.VarName
481        Arithmetic methods to vectorize.
482
483        `vectorize` is useful if an arithmetic method provided to `kwargs` does not support stacking
484        dimensions.
485    batch_size: int
486        If None (default), vectorized methods are applied as a single map over all inputs.
487        If not None, then compute the vectorized methods `batch_size` samples at a time.
488        Note that ``batch_size=1`` is equivalent to computing the vectorization with a for-loop.
489        If you run into memory issues applying your vectorized methods over stacked inputs, try a non-None `batch_size`.
490    jit: bool
491        Currently has no effect (for future-compatibility only). In the future, if ``True``, then Torch-backed arithmetic methods
492        will be JIT-compiled for better performance.
493    dtype: DType
494        Assumed `dtype` of the Torch-defined arithmetic methods. If `None` the arithmetic methods are assumed precision-agnostic.
495    enable_warnings: bool
496        If ``True``, emit warnings in case of precision/zero-copy issues.
497    name: str
498        Name of the operator.
499    meta: Any
500        Meta information to be provided as tail to :py:class:`~pyxu.abc.operator.Operator._expr`.
501    kwargs: dict
502        Optional   (k[str], v[value | callable]) pairs to use as arithmetic methods.
503
504        Keys are restricted to the following arithmetic methods:
505
506        .. code-block:: python3
507
508           grad(), prox(), pinv(), adjoint()  # methods
509
510        Omitted arithmetic methods default to those provided by `cls`, or are
511        auto-inferred via auto-diff rules.
512
513    Returns
514    -------
515    op: OpT
516        (N, M) Pyxu-compliant operator.
517
518    Notes
519    -----
520    * If provided, arithmetic methods must abide exactly to the interface below:
521
522      .. code-block:: python3
523
524         def apply(tensor: torch.Tensor) -> torch.Tensor
525         def grad(tensor: torch.Tensor) -> torch.Tensor
526         def adjoint(tensor: torch.Tensor) -> torch.Tensor
527         def prox(tensor: torch.Tensor, tau: pxt.Real) -> torch.Tensor
528         def pinv(tensor: torch.Tensor, damp: pxt.Real) -> torch.Tensor
529
530      The arithmetic methods `apply`, `grad`, `prox` **must** accept ``(-1, M)``-shaped inputs for ``tensor``, that is a
531      core dimension of size `M` and an optional stacking/batching dimension of arbitrary size.
532      The arithmetic methods `adjoint`, `pinv` **must** accept ``(-1, N)``-shaped inputs for ``tensor``, that is a core
533      dimension of size `N` and an optional stacking/batching dimension of arbitrary size.
534      If stacking/batching dimensions are not supported for some methods, consider populating `vectorize` accordingly.
535
536    * Auto-vectorization consists in decorating `kwargs`-specified arithmetic methods with :py:func:`torch.vmap`.
537      See the `PyTorch documentation <https://pytorch.org/docs/stable/func.ux_limitations.html#vmap-limitations>`_ for
538      known limitations.
539
540    * All arithmetic methods provided in `kwargs` are decorated using :py:func:`~pyxu.runtime.enforce_precision` to
541      abide by Pyxu's FP-runtime semantics.
542      Note however that Torch does not allow mixed-precision computation, so this wrapper will coerce the input if its
543      precision does not comply with the specified `dtype`.
544      This triggers a warning by default, which can be silenced via the `enable_warnings`.
545
546    * Arithmetic methods are **not currently JIT-ed** even if `jit` is set to ``True``.
547      This is because of the undocumented and currently poor interaction between :py:mod:`torch.func` transforms and
548      :py:func:`torch.compile`.
549      See `this issue <https://github.com/pytorch/pytorch/issues/98822>`_ for additional details.
550
551    * For :py:class:`~pyxu.abc.operator.DiffMap` (or subclasses thereof), the methods
552      :py:meth:`~pyxu.abc.operator.DiffMap.jacobian`, :py:meth:`~pyxu.abc.operator.DiffFunc.grad` and
553      :py:meth:`~pyxu.abc.operator.LinOp.adjoint` are defined implicitly if not provided using the
554      auto-differentiation transforms from :py:mod:`torch.func`.
555      As detailed `on this page <https://pytorch.org/docs/stable/func.ux_limitations.html>`_, such transforms work well
556      on pure functions (that is, functions where the output is completely determined by the input and that do not
557      involve side effects like mutation), but may fail on more complex functions.
558      Moreover, :py:mod:`torch.func` does not yet have full coverage over PyTorch operations.
559      For functions that call a :py:class:`torch.nn.Module`, see `here
560      <https://pytorch.org/docs/stable/func.api.html#utilities-for-working-with-torch-nn-modules>`_ for some utilities.
561
562    .. Warning::
563
564       Operators created with this wrapper do not support Dask inputs for now.
565    """
566    if isinstance(vectorize, str):
567        vectorize = (vectorize,)
568    vectorize = frozenset(vectorize)
569
570    src = _FromTorch(
571        apply=apply,
572        shape=shape,
573        cls=cls,
574        vectorize=vectorize,
575        batch_size=batch_size,
576        jit=jit,
577        dtype=dtype,
578        enable_warnings=enable_warnings,
579        name=name,
580        meta=meta,
581        **kwargs,
582    )
583    op = src.op()
584    return op
