 34def _from_jax(
 35    x: JaxArray,
 36    xp: pxt.ArrayModule = None,
 37) -> pxt.NDArray:
 38    """
 39    JAX -> NUMPY/CUPY conversion.
 40
 41    The transform is always zero-copy, but it is not easy to check this condition for all array
 42    types (contiguous, views, etc.) and backends (NUMPY, CUPY).
 43
 44    [More info] https://github.com/google/jax/issues/1961#issuecomment-875773326
 45    """
 46    N = pxd.NDArrayInfo  # shorthand
 47
 48    if xp is None:
 49        xp = N.default().module()
 50
 51    if xp not in (N.NUMPY.module(), N.CUPY.module()):
 52        raise pxw.BackendWarning("Only NUMPY/CUPY inputs are supported.")
 53
 54    y = xp.asarray(x)
 55    return y

 58def _to_jax(x: pxt.NDArray, enable_warnings: bool = True) -> JaxArray:
 59    """
 60    NUMPY/CUPY -> JAX conversion.
 61
 62    Conversion is zero-copy when possible, i.e. 16-byte alignment, on the right device, etc.
 63
 64    [More info] https://github.com/google/jax/issues/4486#issuecomment-735842976
 65    """
 66    N = pxd.NDArrayInfo  # shorthand
 67    W, cW = pxrt.Width, pxrt.CWidth  # shorthand
 68
 69    ndi = N.from_obj(x)
 70    if ndi == N.DASK:
 71        raise pxw.BackendWarning("DASK inputs are unsupported.")
 72
 73    supported_dtype = set(w.value for w in W) | set(w.value for w in cW)
 74    if x.dtype not in supported_dtype:
 75        msg = "For safety reasons, _to_jax() only accepts pyxu.runtime.[C]Width-supported dtypes."
 76        raise pxw.PrecisionWarning(msg)
 77
 78    xp = ndi.module()
 79    if ndi == N.NUMPY:
 80        dev_type = "cpu"
 81        f_wrap = jnp.asarray
 82    elif ndi == N.CUPY:
 83        dev_type = "gpu"
 84        x = xp.require(x, requirements="C")  # JAX-DLPACK only supports contiguous arrays [2023.04.05]
 85        f_wrap = jnp.from_dlpack
 86    else:
 87        raise ValueError("Unknown NDArray category.")
 88    dev = jax.devices(dev_type)[0]
 89    with jax.default_device(dev):
 90        y = f_wrap(x)
 91
 92    same_dtype = x.dtype == y.dtype
 93    same_mem = xp.byte_bounds(x)[0] == y.device_buffer.unsafe_buffer_pointer()
 94    if not (same_dtype and same_mem) and enable_warnings:
 95        msg = "\n".join(
 96            [
 97                "_to_jax(): a zero-copy conversion did not take place.",
 98                "[More info] https://jax.readthedocs.io/en/latest/notebooks/Common_Gotchas_in_JAX.html#double-64bit-precision",
 99                "[More info] https://github.com/google/jax/issues/4486#issuecomment-735842976",
100            ]
101        )
102        warnings.warn(msg, pxw.PrecisionWarning)
103    return y

106def from_jax(
107    cls: pxt.OpC,
108    shape: pxt.OpShape,
109    vectorize: pxt.VarName = frozenset(),
110    jit: bool = False,
111    enable_warnings: bool = True,
112    **kwargs,
113) -> pxt.OpT:
114    r"""
115    Define an :py:class:`~pyxu.abc.operator.Operator` from low-level constructs.
116
117    Parameters
118    ----------
119    cls: OpC
120        Operator sub-class to instantiate.
121    shape: OpShape
122        (N, M) operator shape.
123    kwargs: dict
124        (k[str], v[callable]) pairs to use as arithmetic methods.
125
126        Keys are restricted to the following arithmetic methods:
127
128        .. code-block:: python3
129
130           apply(), grad(), prox(), pinv(), adjoint()
131
132        Omitted arithmetic methods default to those provided by `cls`, or are
133        auto-inferred via auto-diff rules.
134    vectorize: VarName
135        Arithmetic methods to vectorize.
136
137        `vectorize` is useful if an arithmetic method provided to `kwargs` does not support stacking
138        dimensions.
139    jit: bool
140        If ``True``, JIT-compile JAX-backed arithmetic methods for better performance.
141    enable_warnings: bool
142        If ``True``, emit warnings in case of precision/zero-copy issues.
143
144    Returns
145    -------
146    op: OpT
147        (N, M) Pyxu-compliant operator.
148
149    Notes
150    -----
151    * If provided, arithmetic methods must abide exactly to the interface below:
152
153      .. code-block:: python3
154
155         def apply(arr: jax.Array) -> jax.Array
156         def grad(arr: jax.Array) -> jax.Array
157         def adjoint(arr: jax.Array) -> jax.Array
158         def prox(arr: jax.Array, tau: pxt.Real) -> jax.Array
159         def pinv(arr: jax.Array, damp: pxt.Real) -> jax.Array
160
161      Moreover, the methods above **must** accept ``(..., M)``-shaped inputs for ``arr``.
162      If this does not hold, consider populating `vectorize`.
163
164    * Auto-vectorization consists in decorating `kwargs`-specified arithmetic methods with
165      :py:func:`jax.numpy.vectorize`.
166
167    * All arithmetic methods provided in `kwargs` are decorated using
168      :py:func:`~pyxu.runtime.enforce_precision` to abide by Pyxu's FP-runtime semantics.
169      Note however that JAX enforces `32-bit arithmetic <https://jax.readthedocs.io/en/latest/notebooks/Common_Gotchas_in_JAX.html#double-64bit-precision>`_
170      by default, and this constraint cannot be changed at runtime.
171      As such, to allow zero-copy transfers between JAX and NUMPY/CUPY arrays, it is advised to
172      perform computations in single-precision mode.
173      This can be achieved locally using the :py:class:`~pyxu.runtime.Precision` context manager.
174      (See example.)
175      Alternatively, precision/copy-related warnings can be silenced via the `enable_warnings`
176      option of :py:func:`~pyxu.operator.interop.jax.from_jax`.
177
178    * Inferred arithmetic methods are not JIT-ed by default since the operation is error-prone
179      depending on how :py:meth:`~pyxu.abc.operator.Map.apply` is defined.
180      If :py:meth:`~pyxu.abc.operator.Map.apply` supplied to
181      :py:func:`~pyxu.operator.interop.jax.from_jax` is JIT-friendly, then consider enabling `jit`.
182
183    Examples
184    --------
185    Create the custom differential map :math:`f: \mathbb{R}^{2} \to \mathbb{R}^{3}`:
186
187    .. math::
188
189       f(x, y) =
190       \left[
191           \sin(x) + \cos(y),
192           \cos(x) - \sin(y),
193           \sin(x) + \cos(x)
194       \right]
195
196    .. code-block:: python3
197
198       import pyxu.abc as pxa
199       import pyxu.runtime as pxrt
200       import pyxu.operator.interop as pxi
201       import jax, jax.numpy as jnp
202       import numpy as np
203
204       @jax.jit
205       def j_apply(arr: jax.Array) -> jax.Array:
206           x, y = arr[0], arr[1]
207           o1 = jnp.sin(x) + jnp.cos(y)
208           o2 = jnp.cos(x) - jnp.sin(y)
209           o3 = jnp.sin(x) + jnp.cos(x)
210           out = jnp.r_[o1, o2, o3]
211           return out
212
213       op = pxi.from_jax(
214           cls=pxa.DiffMap,
215           shape=(3, 2),
216           vectorize="apply",  # j_apply() does not work on stacked inputs
217                               # --> let JAX figure it out automatically.
218           apply=j_apply,
219       )
220       with pxrt.Precision(pxrt.Width.SINGLE):
221           rng = np.random.default_rng(0)
222           x = rng.normal(size=(5,3,4,2))
223           y1 = op.apply(x)  # (5,3,4,3)
224
225           x = rng.normal(size=(2,))
226           opJ = op.jacobian(x)  # JAX auto-infers the Jacobian for you.
227
228           v = rng.normal(size=(5,2))
229           w = rng.normal(size=(4,3))
230           y2 = opJ.apply(v)  # (5,3)
231           y3 = opJ.adjoint(w)  # (4,2)
232    """
233    if isinstance(vectorize, str):
234        vectorize = (vectorize,)
235    vectorize = frozenset(vectorize)
236
237    src = _FromJax(
238        cls=cls,
239        shape=shape,
240        vectorize=vectorize,
241        jit=bool(jit),
242        enable_warnings=bool(enable_warnings),
243        **kwargs,
244    )
245    op = src.op()
246    return op

249class _FromJax(pxsrc._FromSource):
250    # supported methods in __init__(**kwargs)
251    _meth = frozenset({"apply", "grad", "prox", "pinv", "adjoint"})

253    def __init__(  # See from_jax() for a detailed description.
254        self,
255        cls: pxt.OpC,
256        shape: pxt.OpShape,
257        vectorize: frozenset[str],
258        jit: bool,
259        enable_warnings: bool,
260        **kwargs,
261    ):
262        super().__init__(
263            cls=cls,
264            shape=shape,
265            embed=dict(),  # jax-funcs are state-free.
266            vectorize=vectorize,
267            vmethod=None,  # pyxu.util.vectorize() not used for jax-funcs
268            enforce_precision=frozenset(),  # will be applied manually in op() ...
269            **kwargs,
270        )
271        self._enforce_fp = frozenset(self._ekwargs)  # ... based on this list
272        self._jit = jit
273        self._enable_warnings = enable_warnings
274
275        # Only a subset of arithmetic methods allowed from from_jax().
276        if not (set(self._kwargs) <= self._meth):
277            msg_head = "Unsupported arithmetic methods:"
278            unsupported = set(self._kwargs) - self._meth
279            msg_tail = ", ".join([f"{name}()" for name in unsupported])
280            raise ValueError(f"{msg_head} {msg_tail}")

282    def op(self) -> pxt.OpT:
283        # Idea: modify `**kwargs` from constructor to [when required]:
284        #   1. auto-define omitted methods.            [_infer_missing()]
285        #   2. auto-vectorize via vmap().              [_auto_vectorize()]
286        #   3. JIT & JAX<>NUMPY/CUPY conversions.      [_interface()]
287        #   4. delegate enforce_precision() calls to from_source().
288        self._infer_missing()
289        self._auto_vectorize()
290        j_state, kwargs = self._interface()
291
292        _op = pxsrc.from_source(
293            cls=self._op.__class__,
294            shape=self._op.shape,
295            embed=dict(
296                _jax=j_state,
297                _enable_warnings=self._enable_warnings,
298                _jit=self._jit,
299            ),
300            # vectorize=None,  # see top-level comment.
301            # vmethod=None,    #
302            enforce_precision=self._enforce_fp,
303            **kwargs,
304        )
305        return _op

307    def _infer_missing(self):
308        # The following methods must be auto-inferred if missing from `kwargs`:
309        #
310        #     grad(), adjoint()
311        #
312        # Missing methods are auto-inferred via auto-diff rules and added to `_kwargs`.
313        # At the end of _infer_missing(), all jax-funcs required for _interface() have been added to
314        # `_kwargs`.
315        #
316        # Notes
317        # -----
318        # This method does NOT produce vectorized implementations: _auto_vectorize() is responsible
319        # for this.
320        self._vectorize = set(self._vectorize)  # to allow updates below
321
322        nl_difffunc = all(  # non-linear diff-func
323            [
324                self._op.has(pxo.Property.DIFFERENTIABLE_FUNCTION),
325                not self._op.has(pxo.Property.LINEAR),
326            ]
327        )
328        if nl_difffunc and ("grad" not in self._kwargs):
329
330            def f_grad(arr: JaxArray) -> JaxArray:
331                f = self._kwargs["apply"]
332                y, f_vjp = jax.vjp(f, arr)
333                v = jnp.ones_like(y)
334                out = f_vjp(v)[0]  # f_vjp() returns a tuple
335                return out
336
337            self._vectorize.add("grad")
338            self._kwargs["grad"] = f_grad
339
340        non_selfadj = all(  # linear, but not self-adjoint
341            [
342                self._op.has(pxo.Property.LINEAR),
343                not self._op.has(pxo.Property.LINEAR_SELF_ADJOINT),
344            ]
345        )
346        if non_selfadj and ("adjoint" not in self._kwargs):
347
348            def f_adjoint(arr: JaxArray) -> JaxArray:
349                f = self._kwargs["apply"]
350                x = jnp.zeros_like(arr, shape=(self._op.dim,))
351                _, f_vjp = jax.vjp(f, x)
352                out = f_vjp(arr)[0]  # f_vjp() returns a tuple
353                return out
354
355            self._vectorize.add("adjoint")
356            self._kwargs["adjoint"] = f_adjoint
357
358        self._vectorize = frozenset(self._vectorize)
359
360    def _auto_vectorize(self):
361        # Vectorize user-specified [or _infer_missing()-added] arithmetic methods via jax.vmap().
362        #
363        # Modifies `_kwargs` to hold vectorized jax-funcs.
364        vkwargs = dict(  # kwargs to jax.numpy.vectorize()
365            apply=dict(signature="(n)->(m)"),
366            adjoint=dict(signature="(n)->(m)"),
367            grad=dict(signature="(n)->(n)"),
368            prox=dict(signature="(n)->(n)", excluded={1}),
369            pinv=dict(signature="(n)->(m)", excluded={1}),
370        )
371
372        for name in self._kwargs:
373            if name in self._vectorize:
374                func = self._kwargs[name]  # necessarily jax_func
375                vectorize = functools.partial(
376                    jax.numpy.vectorize,
377                    **vkwargs[name],
378                )
379                self._kwargs[name] = vectorize(func)

381    def _interface(self):
382        # Arithmetic methods supplied in `kwargs`:
383        #
384        #     * take `jax.Array` inputs
385        #     * do not have the `self` parameter. (Reason: to be JIT-compatible.)
386        #
387        # This method creates modified arithmetic functions to match Pyxu's API, and `state`
388        # required for them to work.
389        #
390        # Returns
391        # -------
392        # j_state: dict
393        #     Jax functions referenced by wrapper arithmetic methods. (See below.)
394        #     These functions are JIT-compiled if specified.
395        # kwargs: dict
396        #     Pyxu-compatible functions which can be submitted to interop.from_source().
397        j_state = dict()
398        for name, obj in self._kwargs.items():
399            if name in self._meth:
400                if self._jit:
401                    obj = jax.jit(obj)
402                j_state[name] = obj  # necessarily jax_func
403
404        kwargs = dict()
405        for name, obj in self._kwargs.items():
406            func = getattr(self.__class__, name)
407            kwargs[name] = func  # necessarily a pyxu_func
408
409        # Special cases.
410        # (Reason: not in `_kwargs` [c.f. from_jax() docstring], but need to be inferred.)
411        for name in ("jacobian", "_quad_spec", "asarray", "_expr"):
412            kwargs[name] = getattr(self.__class__, name)
413
414        return j_state, kwargs

417    def apply(self, arr: pxt.NDArray) -> pxt.NDArray:

457    @pxrt.enforce_precision(i="arr", o=False)
458    def jacobian(self, arr: pxt.NDArray) -> pxt.OpT:
459        try:
460            # Use the class' method if available ...
461            klass = self.__class__
462            op = klass.jacobian(self, arr)
463        except NotImplementedError:
464            # ... and fallback to auto-inference if undefined.
465            f = self._jax["apply"]
466            j_arr = _to_jax(arr, enable_warnings=self._enable_warnings)
467
468            # define forward: [1] explains why jvp() is a better fit than linearize().
469            # [1] https://jax.readthedocs.io/en/latest/_autosummary/jax.linearize.html
470            _fwd = functools.partial(jax.jvp, f, (j_arr,))
471            f_fwd = lambda arr: _fwd((arr,))[1]  # jax returns a tuple
472
473            # define adjoint: [2] explains benefits of linear_transpose() over vjp().
474            # [2] https://jax.readthedocs.io/en/latest/_autosummary/jax.linear_transpose.html
475            hint = types.SimpleNamespace(shape=(self.dim,), dtype=arr.dtype)
476            _adj = jax.linear_transpose(f_fwd, hint)
477            f_adj = lambda arr: _adj(arr)[0]  # jax returns a tuple
478
479            klass = pxo.LinFunc if (self.codim == 1) else pxo.LinOp
480            op = from_jax(
481                cls=klass,
482                shape=self.shape,
483                vectorize=("apply", "adjoint"),
484                jit=self._jit,
485                enable_warnings=self._enable_warnings,
486                apply=f_fwd,
487                adjoint=f_adj,
488            )
489        return op

491    def _quad_spec(self):
492        if self.has(pxo.Property.QUADRATIC):
493            # auto-infer (Q, c, t)
494            def _grad(arr: JaxArray) -> JaxArray:
495                # Just like jax.grad(f)(arr), but works with (1,)-valued functions.
496                # [jax.grad(f) expects scalar outputs.]
497                f = self._jax["apply"]
498                y, f_vjp = jax.vjp(f, arr)
499                v = jnp.ones_like(y)
500                out = f_vjp(v)[0]  # f_vjp() returns a tuple
501                return out
502
503            # vectorize & JIT internal function
504            _grad = jnp.vectorize(_grad, signature="(m)->(m)")
505            if self._jit:
506                _grad = jax.jit(_grad)
507
508            def Q_apply(arr: JaxArray) -> JaxArray:
509                # \grad_{f}(x) = Q x + c = Q x + \grad_{f}(0)
510                # ==> Q x = \grad_{f}(x) - \grad_{f}(0)
511                z = jnp.zeros_like(arr)
512                out = _grad(arr) - _grad(z)
513                return out
514
515            def c_apply(arr: JaxArray) -> JaxArray:
516                z = jnp.zeros_like(arr)
517                c = _grad(z)
518                out = jnp.sum(c * arr, axis=-1, keepdims=True)
519                return out
520
521            Q = from_jax(
522                cls=pxo.PosDefOp,
523                shape=(self.dim, self.dim),
524                vectorize="apply",
525                jit=self._jit,
526                enable_warnings=self._enable_warnings,
527                apply=Q_apply,
528            )
529            c = from_jax(
530                cls=pxo.LinFunc,
531                shape=(1, self.dim),
532                vectorize="apply",
533                jit=self._jit,
534                enable_warnings=self._enable_warnings,
535                apply=c_apply,
536            )
537
538            # We cannot know a-priori which backend the supplied jax-apply() function works with.
539            # Consequence: to compute `t`, we must try different backends until one works.
540            f = self._jax["apply"]
541            try:  # test a CPU implementation ...
542                with jax.default_device(jax.devices("cpu")[0]):
543                    t = float(f(jnp.zeros(self.dim)))
544            except Exception:  # ... and use GPU if it doesn't work.
545                with jax.default_device(jax.devices("gpu")[0]):
546                    t = float(f(jnp.zeros(self.dim)))
547
548            return (Q, c, t)
549        else:
550            raise NotImplementedError

552    def asarray(self, **kwargs) -> pxt.NDArray:
553        if self.has(pxo.Property.LINEAR):
554            # (1) Lin[Op,Func].asarray() assumes the operator is precision-agnostic.
555            #     This condition does not hold for JAX arrays. (See from_jax() notes.)
556            #
557            # (2) If the operator is backend-specific (i.e. only works with CUPY), then we have no
558            #     way to determine which `xp` value use in the generic LinOp.asarray()
559            #     implementation without a potential fail.
560            #
561            # Consequence:
562            # (1) May need to cast Lin[Op,Func].asarray()'s output.
563            # (2) We must try different `xp` values until one works.
564            N = pxd.NDArrayInfo  # shorthand
565            dtype_orig = kwargs.get("dtype", pxrt.getPrecision().value)
566            xp_orig = kwargs.get("xp", N.default().module())
567
568            klass = self.__class__
569            try:
570                A = klass.asarray(self, dtype=dtype_orig, xp=N.NUMPY.module())
571            except Exception:
572                A = klass.asarray(self, dtype=dtype_orig, xp=N.CUPY.module())
573
574            # Not the most efficient method, but fail-proof
575            A = xp_orig.array(pxu.to_NUMPY(A), dtype=dtype_orig)
576            return A
577        else:
578            raise NotImplementedError
