 39    def type(self) -> type:
 40        """Array type associated to a backend."""
 41        if self.name == "NUMPY":
 42            return numpy.ndarray
 43        elif self.name == "DASK":
 44            return dask.array.core.Array
 45        elif self.name == "CUPY":
 46            return cupy.ndarray if CUPY_ENABLED else type(None)
 47        else:
 48            raise ValueError(f"No known array type for {self.name}.")

 67    def module(self, linalg: bool = False) -> types.ModuleType:
 68        """
 69        Python module associated to an array backend.
 70
 71        Parameters
 72        ----------
 73        linalg: bool
 74            Return the linear-algebra submodule with identical API to :py:mod:`numpy.linalg`.
 75        """
 76        if self.name == "NUMPY":
 77            xp = numpy
 78            xpl = xp.linalg
 79        elif self.name == "DASK":
 80            xp = dask.array
 81            xpl = xp.linalg
 82        elif self.name == "CUPY":
 83            xp = cupy if CUPY_ENABLED else None
 84            xpl = xp if (xp is None) else xp.linalg
 85        else:
 86            raise ValueError(f"No known module(s) for {self.name}.")
 87        return xpl if linalg else xp

105    def type(self) -> type:
106        """Array type associated to a backend."""
107        if self.name == "SCIPY_SPARSE":
108            # All `*matrix` classes descend from `spmatrix`.
109            return scipy.sparse.spmatrix
110        elif self.name == "PYDATA_SPARSE":
111            return sparse.SparseArray
112        elif self.name == "CUPY_SPARSE":
113            return cupyx.scipy.sparse.spmatrix if CUPY_ENABLED else type(None)
114        else:
115            raise ValueError(f"No known array type for {self.name}.")

149def supported_array_types() -> cabc.Collection[type]:
150    """List of all supported dense array types in current Pyxu install."""
151    data = set()
152    for ndi in NDArrayInfo:
153        if (ndi != NDArrayInfo.CUPY) or CUPY_ENABLED:
154            data.add(ndi.type())
155    return tuple(data)
